792 /*==============================================================================================================================
793 
794  Finds the end point of a coastline-normal line, given the start point on the vector coastline. All co-ordinates are in the external CRS
795 
796 ===============================================================================================================================*/
797 int CSimulation::nGetCoastNormalEndPoint(int const nCoast, int const nStartCoastPoint, int const nCoastSize, CGeom2DPoint const* pPtStart, double const dLineLength, CGeom2DPoint* pPtEnd, CGeom2DIPoint* pPtiEnd)
798 {
799  // If at beginning or end of coast, need special treatment for points before and points after
800  int
801  nCoastPointBefore = tMax(nStartCoastPoint-1, 0),
802  nCoastPointAfter = tMin(nStartCoastPoint+1, nCoastSize-1);
803 
804  // Get the y = a * x + b equation of the straight line linking the coastline points before and after 'this' coastline point
805  CGeom2DPoint PtBefore = *m_VCoast[nCoast].pPtGetCoastlinePointExtCRS(nCoastPointBefore); // PtBefore has coordinates in external CRS
806  CGeom2DPoint PtAfter = *m_VCoast[nCoast].pPtGetCoastlinePointExtCRS(nCoastPointAfter); // PtAfter has coordinates in external CRS
807 
808  // For this linking line, slope a = (y2 - y1) / (x2 - x1)
809  double
810  dYDiff = PtAfter.dGetY() - PtBefore.dGetY(),
811  dXDiff = PtAfter.dGetX() - PtBefore.dGetX();
812 
813  double dXEnd1 = 0, dXEnd2 = 0, dYEnd1 = 0, dYEnd2 = 0;
814 
815  if (bFPIsEqual(dYDiff, 0, TOLERANCE))
816  {
817  // The linking line runs W-E or E-W, so a straight line at right angles to this runs N-S or S-N. Calculate the two possible end points for this coastline-normal profile
818  dXEnd1 = dXEnd2 = pPtStart->dGetX();
819  dYEnd1 = pPtStart->dGetY() + dLineLength;
820  dYEnd2 = pPtStart->dGetY() - dLineLength;
821  }
822  else if (bFPIsEqual(dXDiff, 0, TOLERANCE))
823  {
824  // The linking line runs N-S or S-N, so a straight line at right angles to this runs W-E or E-W. Calculate the two possible end points for this coastline-normal profile
825  dYEnd1 = dYEnd2 = pPtStart->dGetY();
826  dXEnd1 = pPtStart->dGetX() + dLineLength;
827  dXEnd2 = pPtStart->dGetX() - dLineLength;
828  }
829  else
830  {
831  // The linking line runs neither W-E nor N-S so we have to work a bit harder to find the end-point of the coastline-normal profile
832  double dA = dYDiff / dXDiff;
833 
834  // Now calculate the equation of the straight line which is perpendicular to this linking line
835  double
836  dAPerp = -1 / dA,
837  dBPerp = pPtStart->dGetY() - (dAPerp * pPtStart->dGetX());
838 
839  // Calculate the end point of the profile: first do some substitution then rearrange as a quadratic equation i.e. in the form Ax^2 + Bx + C = 0 (see http://math.stackexchange.com/questions/228841/how-do-i-calculate-the-intersections-of-a-straight-line-and-a-circle)
840  double
841  dQuadA = 1 + (dAPerp * dAPerp),
842  dQuadB = 2 * ((dBPerp * dAPerp) - (dAPerp * pPtStart->dGetY()) - pPtStart->dGetX()),
843  dQuadC = ((pPtStart->dGetX() * pPtStart->dGetX()) + (pPtStart->dGetY() * pPtStart->dGetY()) + (dBPerp * dBPerp) - (2 * pPtStart->dGetY() * dBPerp) - (dLineLength * dLineLength));
844 
845  // Solve for x and y using the quadratic formula x = (−B ± sqrt(B^2 − 4AC)) / 2A
846  double dDiscriminant = (dQuadB * dQuadB) - (4 * dQuadA * dQuadC);
847  if (dDiscriminant < 0)
848  {
849  LogStream << ERR << "timestep " << m_ulIter << ": discriminant < 0 when finding profile end point on coastline " << nCoast << ", from coastline point " << nStartCoastPoint << "), ignored" << endl;
850  return RTN_ERR_NO_SOLUTION_FOR_ENDPOINT;
851  }
852 
853  dXEnd1 = (-dQuadB + sqrt(dDiscriminant)) / (2 * dQuadA);
854  dYEnd1 = (dAPerp * dXEnd1) + dBPerp;
855  dXEnd2 = (-dQuadB - sqrt(dDiscriminant)) / (2 * dQuadA);
856  dYEnd2 = (dAPerp * dXEnd2) + dBPerp;
857  }
858 
859  // We have two possible solutions, so decide which of the two endpoints to use then create the profile end-point (coordinates in external CRS)
860  int nSeaHand = m_VCoast[nCoast].nGetSeaHandedness(); // Assumes handedness is either 0 or 1 (i.e. not -1)
861  *pPtEnd = PtChooseEndPoint(nSeaHand, &PtBefore, &PtAfter, dXEnd1, dYEnd1, dXEnd2, dYEnd2);
862 
863  // Check that pPtiEnd is not off the grid. Note that pPtiEnd is NOT (necessarily) a cell centroid
864  pPtiEnd->SetXY(nRound(dExtCRSXToGridX(pPtEnd->dGetX())), nRound(dExtCRSYToGridY(pPtEnd->dGetY())));
865  if (! bIsWithinValidGrid(pPtiEnd))
866  {
867  // The end point is off the grid, so constrain it to be within the valid grid
868  CGeom2DIPoint PtiStart(nRound(dExtCRSXToGridX(pPtStart->dGetX())), nRound(dExtCRSYToGridY(pPtStart->dGetY())));
869  KeepWithinValidGrid(&PtiStart, pPtiEnd);
870 
871  pPtEnd->SetX(dGridCentroidXToExtCRSX(pPtiEnd->nGetX()));
872  pPtEnd->SetY(dGridCentroidYToExtCRSY(pPtiEnd->nGetY()));
873 
874 // LogStream << m_ulIter << ": changed endpoint for profile, is now [" << PtiEnd.nGetX() << "][" << PtiEnd.nGetY() << "] = {" << pPtEnd->dGetX() << ", " << pPtEnd->dGetY() << "}. The profile starts at coastline point " << nStartCoastPoint << " = {" << pPtStart->dGetX() << ", " << pPtStart->dGetY() << "}" << endl;
875 // return RTN_ERR_PROFILE_ENDPOINT_IS_OFFGRID;
876  }
877 
878  return RTN_OK;
879 }
