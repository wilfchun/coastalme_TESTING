162 void CSimulation::StartClock(void)
163 {
164  // First start the 'CPU time' clock ticking
165  if (static_cast<clock_t>(-1) == clock())
166  {
167  // There's a problem with the clock, but continue anyway
168  LogStream << NOTE << "CPU time not available" << endl;
169  m_dCPUClock = -1;
170  }
171  else
172  {
173  // All OK, so get the time in m_dClkLast (this is needed to check for clock rollover on long runs)
174  m_dClkLast = static_cast<double>(clock());
175  m_dClkLast -= CLOCK_T_MIN; // necessary if clock_t is signed to make m_dClkLast unsigned
176  }
177 
178  // And now get the actual time we started
179  m_tSysStartTime = time(nullptr);
180 }

246 
247 ==============================================================================================================================*/
248 double CSimulation::dGetTimeMultiplier(string const* strIn)
249 {
250  // First decide what the time units are
251  int nTimeUnits = nDoTimeUnits(strIn);
252 
253  // Then return the correct multiplier, since m_dTimeStep is in hours
254  switch (nTimeUnits)
255  {
256  case TIME_UNKNOWN:
257  return TIME_UNKNOWN;
258  break;
259 
260  case TIME_HOURS:
261  return 1; // Multiplier for hours
262  break;
263 
264  case TIME_DAYS:
265  return 24; // Multiplier for days -> hours
266  break;
267 
268  case TIME_MONTHS:
269  return 24 * 30.416667; // Multiplier for months -> hours (assume 30 + 5/12 day months, no leap years)
270  break;
271 
272  case TIME_YEARS:
273  return 24 * 365.25; // Multiplier for years -> hours
274  break;
275  }
276 
277  return 0;
278 }

1318 {

1320  m_dSimElapsed += m_dTimeStep;
1321 
1322  if (m_dSimElapsed >= m_dSimDuration)
1323  {
1324  // It is time to quit
1325  m_dSimElapsed = m_dSimDuration;
1326  AnnounceProgress();
1327  return true;

1331  m_ulIter++;

1333

1335  if (0 == m_ulIter % CLOCK_CHECK_ITERATION)
1336  DoCPUClockReset();
1337

1517 /*==============================================================================================================================
1518 
1519  Resets the CPU clock timer to prevent it 'rolling over', as can happen during long runs. This is a particularly problem under Unix systems where the value returned by clock() is defined in microseconds (for compatibility with systems that have CPU clocks with much higher resolution) i.e. CLOCKS_PER_SEC is 1000000 rather than the more usual 1000. In this case, the value returned from clock() will wrap around after accumulating only 2147 seconds of CPU time (about 36 minutes).
1520 
1521 ==============================================================================================================================*/
1522 void CSimulation::DoCPUClockReset(void)
1523 {
1524  if (static_cast<clock_t>(-1) == clock())
1525  {
1526  // Error
1527  LogStream << "CPU time not available" << endl;
1528  m_dCPUClock = -1;
1529  return;
1530  }
1531 
1532  // OK, so carry on
1533  double dClkThis = static_cast<double>(clock());
1534  dClkThis -= CLOCK_T_MIN; // necessary when clock_t is signed, to make dClkThis unsigned
1535 
1536  if (dClkThis < m_dClkLast)
1537  {
1538  // Clock has 'rolled over'
1539  m_dCPUClock += (CLOCK_T_RANGE + 1 - m_dClkLast); // this elapsed before rollover
1540  m_dCPUClock += dClkThis; // this elapsed after rollover
1541 
1542 #ifdef CLOCKCHECK
1543  // For debug purposes
1544  LogStream << "Rolled over: dClkThis=" << dClkThis << " m_dClkLast=" << m_dClkLast << endl << "\t" << " before rollover=" << (CLOCK_T_RANGE + 1 - m_dClkLast) << endl << "\t" << " after rollover=" << dClkThis << endl << "\t" << " ADDED=" << (CLOCK_T_RANGE + 1 - m_dClkLast + dClkThis) << endl;
1545 #endif
1546  }
1547  else
1548  {
1549  // No rollover
1550  m_dCPUClock += (dClkThis - m_dClkLast);
1551 
1552 #ifdef CLOCKCHECK
1553  // For debug purposes
1554  LogStream << "No rollover: dClkThis=" << dClkThis << " m_dClkLast=" << m_dClkLast << " ADDED=" << dClkThis - m_dClkLast << endl;
1555 #endif
1556  }
1557 
1558  // Reset for next time
1559  m_dClkLast = dClkThis;
1560 }
