414 bool CGeomCell::bIsSeaIncBeach(void) const
415 {
416  if (m_bInContiguousSea)
417  // Sea
418  return true;
419 
420  double
421  dWaterLevel = m_pGrid->pGetSim()->CSimulation::dGetThisIterSWL(),
422  dSedTop = m_VdAllHorizonTopElev.back();
423 
424  // Beach
425  if ((m_VLayerAboveBasement.back().dGetUnconsolidatedThickness() > 0) && ((dSedTop - m_pGrid->pGetSim()->CSimulation::dGetMaxBeachElevAboveSWL()) < dWaterLevel))
426  return true;
427 
428  return false;

465 void CGeomCell::CalcAllLayerElevsAndD50(void)
466 {
467  m_VdAllHorizonTopElev.clear();
468  m_VdAllHorizonTopElev.push_back(m_dBasementElevation); // Elevation of top of the basement
469 
470  // Calculate the elevation of the top of all other layers
471  int m = 0;
472  for (unsigned int n = 0; n < m_VLayerAboveBasement.size(); n++)
473  m_VdAllHorizonTopElev.push_back(m_VLayerAboveBasement[n].dGetTotalThickness() + m_VdAllHorizonTopElev[m++]); // Elevation of top of layer n
474 
475  // Now calculate the d50 of the topmost unconsolidated sediment layer with non-zero thickness
476  m_dUnconsD50 = DBL_NODATA;
477  for (int n = static_cast<int>(m_VLayerAboveBasement.size())-1; n >= 0; n--)
478  {
479  double dUnconsThick = m_VLayerAboveBasement[n].dGetUnconsolidatedThickness();
480  if (dUnconsThick > 0)
481  {
482  // This is a layer with non-zero thickness of unconsolidated sediment
483  CRWCellSediment* pUnconsSedLayer = m_VLayerAboveBasement[n].pGetUnconsolidatedSediment();
484  double
485  dFineProp = pUnconsSedLayer->dGetFine() / dUnconsThick,
486  dSandProp = pUnconsSedLayer->dGetSand() / dUnconsThick,
487  dCoarseProp = pUnconsSedLayer->dGetCoarse() / dUnconsThick;
488 
489  // Calculate d50 for the unconsolidated sediment
490  m_dUnconsD50 = (dFineProp * m_pGrid->pGetSim()->dGetD50Fine()) + (dSandProp * m_pGrid->pGetSim()->dGetD50Sand()) + (dCoarseProp * m_pGrid->pGetSim()->dGetD50Coarse());
491 
492  break;
493  }
494  }
495 }

498 int CGeomCell::nGetLayerAtElev(double const dElev) const
499 {
501  if (dElev < m_VdAllHorizonTopElev[0])
502  return ELEV_IN_BASEMENT;
503 
504  for (unsigned int nLayer = 1; nLayer < m_VdAllHorizonTopElev.size(); nLayer++)
505  {
506  if ((m_VLayerAboveBasement[nLayer-1].dGetTotalThickness() > 0) && (dElev >= m_VdAllHorizonTopElev[nLayer-1]) && (dElev <= m_VdAllHorizonTopElev[nLayer]))
507  return (nLayer-1);
508  }
509 
510  return ELEV_ABOVE_SEDIMENT_TOP;
511 }
