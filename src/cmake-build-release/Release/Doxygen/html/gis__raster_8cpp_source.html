52  Reads a raster DEM of basement elevation data to the Cell array
53 
54 ===============================================================================================================================*/
55 int CSimulation::nReadRasterBasementDEM(void)
56 {
57  // Use GDAL to create a dataset object, which then opens the DEM file
58  GDALDataset *pGDALDataset = (GDALDataset *)GDALOpen(m_strInitialBasementDEMFile.c_str(), GA_ReadOnly);
59  if (NULL == pGDALDataset)
60  {
61  // Can't open file (note will already have sent GDAL error message to stdout)
62  cerr << ERR << "cannot open " << m_strInitialBasementDEMFile << " for input: " << CPLGetLastErrorMsg() << endl;
63  return RTN_ERR_DEMFILE;
64  }
65 
66  // Opened OK, so get GDAL basement DEM dataset information
67  m_strGDALBasementDEMDriverCode = pGDALDataset->GetDriver()->GetDescription();
68  m_strGDALBasementDEMDriverDesc = pGDALDataset->GetDriver()->GetMetadataItem(GDAL_DMD_LONGNAME);
69  m_strGDALBasementDEMProjection = pGDALDataset->GetProjectionRef();
70 
71  // If we have reference units, then check that they are in meters (note US spelling)
72  if (!m_strGDALBasementDEMProjection.empty())
73  {
74  string strTmp = strToLower(&m_strGDALBasementDEMProjection);
75  if (strTmp.find("meter") == string::npos)
76  {
77  // error: x-y values must be in metres
78  cerr << ERR << "GIS file x-y values (" << m_strGDALBasementDEMProjection << ") in " << m_strInitialBasementDEMFile << " must be in metres" << endl;
79  return RTN_ERR_DEMFILE;
80  }
81  }
82 
83  // Now get dataset size, and do some rudimentary checks
84  m_nXGridMax = pGDALDataset->GetRasterXSize();
85  if (m_nXGridMax == 0)
86  {
87  // Error: silly number of columns specified
88  cerr << ERR << "invalid number of columns (" << m_nXGridMax << ") in " << m_strInitialBasementDEMFile << endl;
89  return RTN_ERR_DEMFILE;
90  }
91 
92  m_nYGridMax = pGDALDataset->GetRasterYSize();
93  if (m_nYGridMax == 0)
94  {
95  // Error: silly number of rows specified
96  cerr << ERR << "invalid number of rows (" << m_nYGridMax << ") in " << m_strInitialBasementDEMFile << endl;
97  return RTN_ERR_DEMFILE;
98  }
99 
100  // Get geotransformation info (see http://www.gdal.org/classGDALDataset.html)
101  if (CE_Failure == pGDALDataset->GetGeoTransform(m_dGeoTransform))
102  {
103  // Can't get geotransformation (note will already have sent GDAL error message to stdout)
104  cerr << ERR << CPLGetLastErrorMsg() << " in " << m_strInitialBasementDEMFile << endl;
105  return RTN_ERR_DEMFILE;
106  }
107 
108  // Get the X and Y cell sizes, in external CRS units. Note that while the cell is supposed to be square, it may not be exactly so due to oddities with some GIS calculations
109  double dCellSideX = tAbs(m_dGeoTransform[1]);
110  double dCellSideY = tAbs(m_dGeoTransform[5]);
111 
112  // Check that the cell is more or less square
113  if (!bFPIsEqual(dCellSideX, dCellSideY, 1e-2))
114  {
115  // Error: cell is not square enough
116  cerr << ERR << "cell is not square in " << m_strInitialBasementDEMFile << ", is " << dCellSideX << " x " << dCellSideY << endl;
117  return (RTN_ERR_RASTER_FILE_READ);
118  }
119 
120  // Calculate the average length of cell side, the cell's diagonal, and the area of a cell (in external CRS units)
121  m_dCellSide = (dCellSideX + dCellSideY) / 2.0;
122  m_dCellArea = m_dCellSide * m_dCellSide;
123  m_dCellDiagonal = hypot(m_dCellSide, m_dCellSide);
124 
125  // And calculate the inverse values
126  m_dInvCellSide = 1 / m_dCellSide;
127  m_dInvCellDiagonal = 1 / m_dCellDiagonal;
128 
129  // Save some values in external CRS
130  m_dNorthWestXExtCRS = m_dGeoTransform[0] - (m_dGeoTransform[1] / 2);
131  m_dNorthWestYExtCRS = m_dGeoTransform[3] - (m_dGeoTransform[5] / 2);
132  m_dSouthEastXExtCRS = m_dGeoTransform[0] + (m_nXGridMax * m_dGeoTransform[1]) + (m_dGeoTransform[1] / 2);
133  m_dSouthEastYExtCRS = m_dGeoTransform[3] + (m_nYGridMax * m_dGeoTransform[5]) + (m_dGeoTransform[5] / 2);
134 
135  // And calc the grid area in external CRS units
136  m_dExtCRSGridArea = tAbs(m_dNorthWestXExtCRS - m_dSouthEastXExtCRS) * tAbs(m_dNorthWestYExtCRS * m_dSouthEastYExtCRS);
137 
138  // Now get GDAL raster band information
139  GDALRasterBand *pGDALBand = pGDALDataset->GetRasterBand(1);
140  int nBlockXSize = 0, nBlockYSize = 0;
141  pGDALBand->GetBlockSize(&nBlockXSize, &nBlockYSize);
142  m_strGDALBasementDEMDataType = GDALGetDataTypeName(pGDALBand->GetRasterDataType());
143 
144  // If we have value units, then check them
145  string strUnits = pGDALBand->GetUnitType();
146  if ((!strUnits.empty()) && (strUnits.find("m") == string::npos))
147  {
148  // Error: value units must be m
149  cerr << ERR << "DEM vertical units are (" << strUnits << " ) in " << m_strInitialBasementDEMFile << ", should be 'm'" << endl;
150  return RTN_ERR_DEMFILE;
151  }
152 
153  // If present, get the missing value (NODATA) setting
154  CPLPushErrorHandler(CPLQuietErrorHandler); // Needed to get next line to fail silently, if it fails
155  double dMissingValue = pGDALBand->GetNoDataValue(); // Will fail for some formats
156  CPLPopErrorHandler();
157 
158  if (dMissingValue != m_dMissingValue)
159  {
160  cerr << " " << NOTE << "NODATA value in " << m_strInitialBasementDEMFile << " is " << dMissingValue << "\n instead using CoastalME's default floating-point NODATA value " << m_dMissingValue << endl;
161  }
162 
163  // Next allocate memory for a 2D array of raster cell objects: tell the user what is happening
164  AnnounceAllocateMemory();
165  int nRet = m_pRasterGrid->nCreateGrid();
166  if (nRet != RTN_OK)
167  return nRet;
168 
169  // Allocate memory for a 1D floating-point array, to hold the scan line for GDAL
170  double *pdScanline = new double[m_nXGridMax];
171  if (NULL == pdScanline)
172  {
173  // Error, can't allocate memory
174  cerr << ERR << "cannot allocate memory for " << m_nXGridMax << " x 1D array" << endl;
175  return (RTN_ERR_MEMALLOC);
176  }
177 
178  // Now read in the data
179  for (int j = 0; j < m_nYGridMax; j++)
180  {
181  // Read scanline
182  if (CE_Failure == pGDALBand->RasterIO(GF_Read, 0, j, m_nXGridMax, 1, pdScanline, m_nXGridMax, 1, GDT_Float64, 0, 0, NULL))
183  {
184  // Error while reading scanline
185  cerr << ERR << CPLGetLastErrorMsg() << " in " << m_strInitialBasementDEMFile << endl;
186  return RTN_ERR_DEMFILE;
187  }
188 
189  // All OK, so read scanline into cell elevations (including any missing values)
190  for (int i = 0; i < m_nXGridMax; i++)
191  {
192  double dTmp = pdScanline[i];
193 
194  if (!isfinite(dTmp)) // Deal with any NaN values
195  dTmp = m_dMissingValue;
196 
197  m_pRasterGrid->m_Cell[i][j].SetBasementElev(dTmp);
198  }
199  }
200 
201  // Finished, so get rid of dataset object
202  GDALClose(pGDALDataset);
203 
204  // Get rid of memory allocated to this array
205  delete[] pdScanline;
206 
207  return RTN_OK;
208 }

218  vector<CGeom2DIPoint> VPtiBoundingBoxCorner;
219 
220  // Start with the top (north) edge
221  bool bFound = false;
222  for (int nX = 0; nX < m_nXGridMax; nX++)
223  {
224  if (bFound)
225  break;
226 
227  for (int nY = 0; nY < m_nYGridMax; nY++)
228  {

230  {
231  CGeom2DIPoint PtiTmp(nX, nY);
232  VPtiBoundingBoxCorner.push_back(PtiTmp);
233  bFound = true;
234  break;
235  }
236  }
237  }
238 
239  if (!bFound)
240  {
241  if (m_nLogFileDetail >= LOG_FILE_MOST_DETAIL)
242  LogStream << m_ulIter << ": north (top) " << endl;
243  return RTN_ERR_BOUNDING_BOX;
244  }
245 
246  // Do the same for the right (east) edge
247  bFound = false;
248  for (int nY = 0; nY < m_nYGridMax; nY++)
249  {
250  if (bFound)
251  break;
252 
253  for (int nX = m_nXGridMax - 1; nX >= 0; nX--)
254  {
255  if (!m_pRasterGrid->m_Cell[nX][nY].bBasementElevIsMissingValue())
256  {
257  CGeom2DIPoint PtiTmp(nX, nY);
258  VPtiBoundingBoxCorner.push_back(PtiTmp);
259  bFound = true;
260  break;
261  }
262  }
263  }
264 
265  if (!bFound)
266  {
267  if (m_nLogFileDetail >= LOG_FILE_MOST_DETAIL)
268  LogStream << m_ulIter << ": east (right) edge of bounding box not found" << endl;
269  return RTN_ERR_BOUNDING_BOX;
270  }
271 
272  // Do the same for the south (bottom) edge
273  bFound = false;
274  for (int nX = m_nXGridMax - 1; nX >= 0; nX--)
275  {
276  if (bFound)
277  break;
278 
279  for (int nY = m_nYGridMax - 1; nY >= 0; nY--)
280  {
281  if (!m_pRasterGrid->m_Cell[nX][nY].bBasementElevIsMissingValue())
282  {
283  CGeom2DIPoint PtiTmp(nX, nY);
284  VPtiBoundingBoxCorner.push_back(PtiTmp);
285  bFound = true;
286  break;
287  }
288  }
289  }
290 
291  if (!bFound)
292  {
293  if (m_nLogFileDetail >= LOG_FILE_MOST_DETAIL)
294  LogStream << m_ulIter << ": south (bottom) edge of bounding box not found" << endl;
295  return RTN_ERR_BOUNDING_BOX;
296  }
297 
298  // And finally repeat for the west (left) edge
299  bFound = false;
300  for (int nY = m_nYGridMax - 1; nY >= 0; nY--)
301  {
302  if (bFound)
303  break;
304 
305  for (int nX = 0; nX < m_nXGridMax; nX++)
306  {
307  if (!m_pRasterGrid->m_Cell[nX][nY].bBasementElevIsMissingValue())
308  {
309  CGeom2DIPoint PtiTmp(nX, nY);
310  VPtiBoundingBoxCorner.push_back(PtiTmp);
311  bFound = true;
312  break;
313  }
314  }
315  }
316 
317  if (!bFound)
318  {
319  if (m_nLogFileDetail >= LOG_FILE_MOST_DETAIL)
320  LogStream << m_ulIter << ": west (left) edge of bounding box not found" << endl;
321  return RTN_ERR_BOUNDING_BOX;
322  }
323 
324  // OK, so we have a point on each side of the grid, so start at this point and find the edges of the bounding box. Go round in a clockwise direction: top (north) edge first
325  for (int nX = VPtiBoundingBoxCorner[0].nGetX(); nX <= VPtiBoundingBoxCorner[1].nGetX(); nX++)
326  {
327  bFound = false;
328  for (int nY = VPtiBoundingBoxCorner[0].nGetY(); nY < m_nYGridMax; nY++)
329  {
330  if (m_pRasterGrid->m_Cell[nX][nY].bBasementElevIsMissingValue())
331  {
332  m_ulMissingValueBasementCells++;
333  continue;
334  }
335

338 
339  m_VEdgeCell.push_back(CGeom2DIPoint(nX, nY));
340  m_VEdgeCellEdge.push_back(NORTH);
341 
342  bFound = true;
343  break;
344  }
345 
346  if (!bFound)
347  {
348  if (m_nLogFileDetail >= LOG_FILE_MIDDLE_DETAIL)
349  LogStream << m_ulIter << ": could not find a bounding box edge cell for grid column " << nX << endl;
350  return RTN_ERR_BOUNDING_BOX;
351  }
352  }
353 
354  // Right (east) edge
355  for (int nY = VPtiBoundingBoxCorner[1].nGetY(); nY <= VPtiBoundingBoxCorner[2].nGetY(); nY++)
356  {
357  bFound = false;
358  for (int nX = VPtiBoundingBoxCorner[1].nGetX(); nX >= 0; nX--)
359  {

372  bFound = true;
373  break;
374  }
375 
376  if (!bFound)
377  {
378  if (m_nLogFileDetail >= LOG_FILE_MIDDLE_DETAIL)
379  LogStream << m_ulIter << ": could not find a bounding box edge cell for grid row " << nY << endl;
380  return RTN_ERR_BOUNDING_BOX;
381  }
382  }
383 
384  // Bottom (south) edge
385  for (int nX = VPtiBoundingBoxCorner[2].nGetX(); nX >= VPtiBoundingBoxCorner[3].nGetX(); nX--)
386  {
387  bFound = false;
388  for (int nY = VPtiBoundingBoxCorner[2].nGetY(); nY >= 0; nY--)
389  {
390  if (m_pRasterGrid->m_Cell[nX][nY].bBasementElevIsMissingValue())
391  {
392  m_ulMissingValueBasementCells++;
393  continue;
394  }
395 
396  // Found a bounding box edge cell
397  m_pRasterGrid->m_Cell[nX][nY].SetBoundingBoxEdge(SOUTH);
398 
399  m_VEdgeCell.push_back(CGeom2DIPoint(nX, nY));
400  m_VEdgeCellEdge.push_back(SOUTH);
401 
402  bFound = true;
403  break;
404  }
405 
406  if (!bFound)
407  {
408  if (m_nLogFileDetail >= LOG_FILE_MIDDLE_DETAIL)
409  LogStream << m_ulIter << ": could not find a bounding box edge cell for grid column " << nX << endl;
410  return RTN_ERR_BOUNDING_BOX;
411  }
412  }
413 
414  // Left (west) edge
415  for (int nY = VPtiBoundingBoxCorner[3].nGetY(); nY >= VPtiBoundingBoxCorner[0].nGetY(); nY--)
416  {
417  for (int nX = VPtiBoundingBoxCorner[3].nGetX(); nX < m_nXGridMax - 1; nX++)
418  //for (int nX = VPtiBoundingBoxCorner[3].nGetX(); nX < VPtiBoundingBoxCorner[3].nGetX(); nX++) //APayo April 2018
419  {
420  if (m_pRasterGrid->m_Cell[nX][nY].bBasementElevIsMissingValue())
421  {
422  m_ulMissingValueBasementCells++;
423  continue;
424  }
425 
426  // Found a bounding box edge cell
427  m_pRasterGrid->m_Cell[nX][nY].SetBoundingBoxEdge(WEST);
428 
429  m_VEdgeCell.push_back(CGeom2DIPoint(nX, nY));
430  m_VEdgeCellEdge.push_back(WEST);
431

434  }

436  if (!bFound)

439  LogStream << m_ulIter << ": could not find a bounding box edge cell for grid row " << nY << endl;
440  return RTN_ERR_BOUNDING_BOX;
441  }
442  }
443 
444  return RTN_OK;
