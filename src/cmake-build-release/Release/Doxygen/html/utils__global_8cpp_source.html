87  Checks to see if a string can be read as a valid integer, from https://stackoverflow.com/questions/2844817/how-do-i-check-if-a-c-string-is-an-int
88 
89 ==============================================================================================================================*/
90 bool bIsStringValidInt(string& str)
91 {
92  // Trim leading whitespace
93  size_t nPos = str.find_first_not_of(" \t");
94  if (nPos != string::npos)
95  str = str.substr(nPos);
96 
97  // If the first character is the sign, remove it
98  if ((str[0] == '-') || (str[0] == '+'))
99  str.erase(0, 1);
100 
101  // Now check that the string contains only numbers
102  return (str.find_first_not_of("0123456789") == string::npos);
103 }
