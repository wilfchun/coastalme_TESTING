111  dWindowTotY += pLineIn->dGetYAt(k);

119  case SOUTH:

138  int nTmpWindow = 0;

187 // LTemp.SetYAt(i, dY);

189  }
190  }
191  }

203 CGeomLine CSimulation::LSmoothCoastRunningMean(CGeomLine* pLineIn, int const nStartEdge, int const nEndEdge) const
204 {
205  // Note that m_nCoastSmoothWindow must be odd (have already checked this)
206  int nHalfWindow = m_nCoastSmoothWindow / 2;
207 
208  // Make a copy of the unsmoothed CGeomLine
209  int nSize = pLineIn->nGetSize();
210  CGeomLine LTemp;
211  LTemp = *pLineIn;
212 
213  // Apply the running mean smoothing filter, with a variable window size at both ends of the line
214  for (int i = 0; i < nSize; i++)
215  {
216  bool bNearStartEdge = false, bNearEndEdge = false;
217  int nTmpWindow = 0;
218  double dWindowTotX = 0, dWindowTotY = 0;
219  for (int j = -nHalfWindow; j < m_nCoastSmoothWindow - nHalfWindow; j++)
220  {
221  // For points at both ends of the coastline, use a smaller window
222  int k = i+j;
223 
224  if (k < 0)
225  {
226  bNearStartEdge = true;
227  continue;
228  }
229 
230  if (k >= nSize)
231  {
232  bNearEndEdge = true;
233  continue;
234  }
235 
236  dWindowTotX += pLineIn->dGetXAt(k);
237  dWindowTotY += pLineIn->dGetYAt(k);
238  nTmpWindow++;
239  }
240 
241  if (bNearStartEdge)
242  {
243  // We are near the start edge
244  switch (nStartEdge)
245  {
246  case NORTH:
247  case SOUTH:
248  // Don't apply the filter in the y direction
249  LTemp.SetXAt(i, dWindowTotX / static_cast<double>(nTmpWindow));
250  break;
251 
252  case EAST:
253  case WEST:
254  // Don't apply the filter in the x direction
255  LTemp.SetYAt(i, dWindowTotY / static_cast<double>(nTmpWindow));
256  break;
257  }
258  }
259  else if (bNearEndEdge)
260  {
261  // We are near the end edge
262  switch (nEndEdge)
263  {
264  case NORTH:
265  case SOUTH:
266  // Don't apply the filter in the y direction
267  LTemp.SetXAt(i, dWindowTotX / static_cast<double>(nTmpWindow));
268  break;
269 
270  case EAST:
271  case WEST:
272  // Don't apply the filter in the x direction
273  LTemp.SetYAt(i, dWindowTotY / static_cast<double>(nTmpWindow));
274  break;
275  }
276  }
277  else
278  {
279  // Not near any edge, apply both x and y filters
280  LTemp[i] = CGeom2DPoint(dWindowTotX / nTmpWindow, dWindowTotY / nTmpWindow);
281 // LTemp.SetXAt(i, dWindowTotX / static_cast<double>(nTmpWindow));
282 // LTemp.SetYAt(i, dWindowTotY / static_cast<double>(nTmpWindow));
283  }
284  }
285 
286  // Return the smoothed CGeomLine
287  return LTemp;
288 }

338  CalcSavitzkyGolay uses LULinearSolve and LUDecomp. It returns dFilterCoeffsArray[nWindowSize], a set of Savitzky-Golay filter coefficients in wrap-around order. nLeft is the number of leftward data points used, nRight is the number of rightward data points, making the total number of data points used = nLeft + nRight + 1. nDerivOrder is the order of the derivative desired (e.g. nDerivOrder = 0 for smoothed function), nSmoothPolyOrder is the order of the smoothing polynomial also equal to the highest conserved moment; usual values are nSmoothPolyOrder = 2 or nSmoothPolyOrder = 4
339 
340 ==============================================================================================================================*/
341 void CSimulation::CalcSavitzkyGolay(double dFilterCoeffsArray[], int const nWindowSize, int const nLeft, int const nRight, int const nDerivOrder, int const nSmoothPolyOrder)
342 {
343  // Some sanity checks
344  if ((nWindowSize < nLeft + nRight + 1) || (nLeft < 0) || (nRight < 0) || (nDerivOrder > nSmoothPolyOrder) || (nSmoothPolyOrder > static_cast<int>(SAVGOL_POLYNOMIAL_MAX_ORDER)) || (nLeft + nRight < nSmoothPolyOrder))
345  {
346  cerr << ERR << "Error in arguments to CalcSavitzkyGolay" << endl;
347  return;
348  }
349 
350  // Initialise arrays (including index 0 for tidiness)
351  int nIndexArray[SAVGOL_POLYNOMIAL_MAX_ORDER+2];
352  Matrix dMatrix;
353  double dSolutionArray[SAVGOL_POLYNOMIAL_MAX_ORDER+2];
354  for (int i = 0; i <= SAVGOL_POLYNOMIAL_MAX_ORDER+1; i++)
355  {
356  for (int j = 0; j <= SAVGOL_POLYNOMIAL_MAX_ORDER+1; j++)
357  dMatrix[i][j] = 0;
358  dSolutionArray[i] = 0;
359  nIndexArray[i] = 0;
360  }
361 
362  for (int ipj = 0; ipj <= 2 * nSmoothPolyOrder ; ipj++)
363  {
364  // Set up the equations for the desired least squares fit
365  double dSum = 0;
366  if (ipj == 0)
367  dSum = 1;
368  for (int k = 1; k <= nRight; k++)
369  dSum += pow(k, ipj);
370  for (int k = 1; k <= nLeft; k++)
371  dSum += pow(-k, ipj);
372  int mm = tMin(ipj, 2 * nSmoothPolyOrder - ipj);
373  int imj = -mm;
374  do
375  {
376  dMatrix[1 + (ipj + imj) / 2][1 + (ipj - imj) / 2] = dSum;
377  imj += 2;
378  }
379  while (imj <= mm);
380  }
381 
382  // Solve them using LU decomposition
383  int nDCode = 0, nICode = 0;
384  LUDecomp(dMatrix, nSmoothPolyOrder+1, SAVGOL_POLYNOMIAL_MAX_ORDER+1, nIndexArray, &nDCode, &nICode);
385 
386  // Right-hand side vector is unit vector, depending on which derivative we want
387  dSolutionArray[nDerivOrder+1] = 1;
388 
389  // Backsubstitute, giving one row of the inverse matrix
390  LULinearSolve(dMatrix, nSmoothPolyOrder +1, nIndexArray, dSolutionArray);
391 
392  // Zero the output array (it may be bigger than the number of coefficients). Again include index 0 for tidiness
393 // for (int n = 0; n < SAVGOL_POLYNOMIAL_MAX_ORDER+1; n++)
394 // for (int n = 0; n <= nWindowSize; n++)
395 // dFilterCoeffsArray[n] = 0;
396 
397  for (int n = -nLeft; n <= nRight; n++)
398  {
399  // Each Savitzky-Golay coefficient is the dot product of powers of an integer with the inverse matrix row
400  double dSum = dSolutionArray[1];
401  double dFac = 1;
402  for (int m = 1; m <= nSmoothPolyOrder ; m++)
403  {
404  dFac *= n;
405  dSum += dSolutionArray[m+1] * dFac;
406  }
407 
408  // Store in wrap-around order
409  int nInd = ((nWindowSize - n) % nWindowSize) + 1;
410  dFilterCoeffsArray[nInd] = dSum;
411  }
412 }

417  Given an N x N matrix A, this routine replaces it by the LU decomposition of a rowwise permutation of itself. A and N are input. nIndexArray is an output vector which records the row permutation effected by the partial pivoting; D is output as -1 or 1, depending on whether the number of row interchanges was even or odd, respectively. This routine is used in combination with LULinearSolve to solve linear equations or to invert a matrix. Returns with nICode = 1 if matrix is singular
418 
419 ==============================================================================================================================*/
420 void LUDecomp(Matrix A, int const N, int const np, int nIndexArray[], int* nDCode, int* nICode)
421 {
422  if (N >= np)
423  {
424  cerr << ERR << "in LUDecomp" << endl;
425  return;
426  }
427 
428  double TINY = 1e-12;
429  double AMAX, DUM, SUM;
430  double* VV = new double[np];
431  int I, IMAX = 0, J, K;
432 
433  *nDCode = 1;
434  *nICode = 0;
435 
436  for (I = 1; I <= N; I++)
437  {
438  AMAX = 0.0;
439  for (J = 1; J <= N; J++)
440  if (tAbs(A[I][J]) > AMAX)
441  AMAX = tAbs(A[I][J]);
442 
443  if (AMAX < TINY)
444  {
445  *nICode = 1;
446  delete[] VV;
447  return;
448  }
449 
450  VV[I] = 1.0 / AMAX;
451  }
452 
453  for (J = 1; J <= N; J++)
454  {
455  for (I = 1; I < J; I++)
456  {
457  SUM = A[I][J];
458  for (K = 1; K < I; K++)
459  SUM -= A[I][K] * A[K][J];
460  A[I][J] = SUM;
461  }
462 
463  AMAX = 0.0;
464  for (I = J; I <= N; I++)
465  {
466  SUM = A[I][J];
467  for (K = 1; K < J; K++)
468  SUM -= A[I][K] * A[K][J];
469 
470  A[I][J] = SUM;
471  DUM = VV[I] * tAbs(SUM);
472  if (DUM >= AMAX)
473  {
474  IMAX = I;
475  AMAX = DUM;
476  }
477  }
478 
479  if (J != IMAX)
480  {
481  for (K = 1; K <= N; K++)
482  {
483  DUM = A[IMAX][K];
484  A[IMAX][K] = A[J][K];
485  A[J][K] = DUM;
486  }
487 
488  *nDCode = -(*nDCode);
489  VV[IMAX] = VV[J];
490  }
491 
492  nIndexArray[J] = IMAX;
493  if (tAbs(A[J][J]) < TINY)
494  A[J][J] = TINY;
495 
496  if (J != N)
497  {
498  DUM = 1.0 / A[J][J];
499  for (I = J + 1; I <= N; I++)
500  A[I][J] *= DUM;
501  }
502  }
503 
504  delete[] VV;

510  Solves the set of N linear equations A . X = B. Here A is input, not as the matrix A but rather as its LU decomposition, determined by the routine LUDecomp. nIndexArray is input as the permutation vector returned by LUDecomp. B is input as the right-hand side vector B, and returns with the solution vector X. A, N and nIndexArray are not modified by this routine and can be used for successive calls with different right-hand sides. This routine is also efficient for plain matrix inversion
511

514 {

516  double SUM;
517 
518  for (int I = 1; I <= N; I++)
519  {
520  int LL = nIndexArray[I];
521  SUM = B[LL];
522  B[LL] = B[I];
523  if (II != 0)
524  for (int J = II; J < I; J++)
525  SUM -= A[I][J] * B[J];
526  else if (SUM != 0.0)
527  II = I;

531  for (int I = N; I > 0; I--)
