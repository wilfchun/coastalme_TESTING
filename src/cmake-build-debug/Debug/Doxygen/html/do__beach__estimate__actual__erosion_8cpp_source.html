590  }

615  Estimates actual (supply-limited) erosion of unconsolidated beach sediment on a single cell, returns the depth-equivalents of fine, sand and coarse sediment which could be removed
616

618 void CSimulation::EstimateActualBeachErosionOnCell(int const nX, int const nY, int const nThisLayer, double const dPotentialErosionOnPolygon, double& dFine, double& dSand, double& dCoarse)
619 {
620  // Find out how much unconsolidated sediment we have available on this cell
621  double
622  dExistingAvailableFine = m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(nThisLayer)->pGetUnconsolidatedSediment()->dGetFine(),
623  dExistingAvailableSand = m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(nThisLayer)->pGetUnconsolidatedSediment()->dGetSand(),
624  dExistingAvailableCoarse = m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(nThisLayer)->pGetUnconsolidatedSediment()->dGetCoarse();
625 
626  // Is there any unconsolidated sediment on this cell?
627  if ((dExistingAvailableFine + dExistingAvailableSand + dExistingAvailableCoarse) < SEDIMENT_ELEV_TOLERANCE)
628  return;
629 
630  // We have some unconsolidated sediment, so partition the total lowering for this cell between the three size fractions: do this by relative erodibility
631  int
632  nFineWeight = (dExistingAvailableFine > 0 ? 1 : 0),
633  nSandWeight = (dExistingAvailableSand > 0 ? 1 : 0),
634  nCoarseWeight = (dExistingAvailableCoarse > 0 ? 1 : 0);
635 
636  double dTotErodibility = (nFineWeight * m_dFineErodibilityNormalized) + (nSandWeight * m_dSandErodibilityNormalized) + (nCoarseWeight * m_dCoarseErodibilityNormalized);
637 
638  if (nFineWeight)
639  {
640  // Pretend to erode some fine-sized consolidated sediment
641  double dFineLowering = (m_dFineErodibilityNormalized * dPotentialErosionOnPolygon) / dTotErodibility;
642 
643  // Make sure we don't get -ve amounts left on the cell
644  dFine = tMin(dExistingAvailableFine, dFineLowering);
645  }
646 
647  if (nSandWeight)
648  {
649  // Pretend to erode some sand-sized consolidated sediment
650  double dSandLowering = (m_dSandErodibilityNormalized * dPotentialErosionOnPolygon) / dTotErodibility;
651 
652  // Make sure we don't get -ve amounts left on the source cell
653  dSand = tMin(dExistingAvailableSand, dSandLowering);
654  }
655 
656  if (nCoarseWeight)
657  {
658  // Erode some coarse-sized consolidated sediment
659  double dCoarseLowering = (m_dCoarseErodibilityNormalized * dPotentialErosionOnPolygon) / dTotErodibility;
660 
661  // Make sure we don't get -ve amounts left on the source cell
662  dCoarse = tMin(dExistingAvailableCoarse, dCoarseLowering);
663  }
664 
665 // LogStream << "\tEstimating actual erosion on [" << nX << "][" << nY << "] dPotentialErosionOnPolygon = " << dPotentialErosionOnPolygon << " dTotActualErosion = " << dTotActualErosion << " dFine = " << dFine << " dSand = " << dSand << " dCoarse = " << dCoarse << endl;
