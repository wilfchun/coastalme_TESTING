43 /*
44  Notes re. co-ordinate systems used
45 
46  1. In the raster CRS, cell[0][0] is at the top left (NW) corner of the grid. Raster grid co-oordinate [0][0] is actually the top left (NW) corner of this cell.
47 
48  2. We assume that the grid CRS and external CRS have parallel axes. If they have not, see http://www.gdal.org/classGDALDataset.html which says that:
49 
50  To convert between pixel/line (P,L) raster space, and projection coordinates (Xp,Yp) space
51  Xp = padfTransform[0] + P*padfTransform[1] + L*padfTransform[2];
52  Yp = padfTransform[3] + P*padfTransform[4] + L*padfTransform[5];
53 
54  In a north-up image, padfTransform[1] is the pixel width, and padfTransform[5] is the pixel height. The upper left corner of the upper left pixel is at position
55  (padfTransform[0], padfTransform[3]).
56 
57  3. Usually, raster grid CRS values are integer, i.e. they refer to a point which is at the centroid of a cell. They may also be -ve or greater than m_nXGridMax-1 i.e. may refer to a point which lies outside any cell of the raster grid.
58 
59 */

243 /*==============================================================================================================================
244 
245  Given two points in the grid CRS (the points assumed not to be coincident), this routine modifies the value of the second point so that it is on a line joining the original two points and is a valid cell within the raster grid
246 
247  However in some cases (e.g. if the first point is at the edge of the valid part of the raster grid) then the second cell will be coincident with the first cell, and the line joining them is thus of zero length. The calling routine has to be able to handle this
248 
249 ===============================================================================================================================*/
250 void CSimulation::KeepWithinValidGrid(int nX0, int nY0, int &nX1, int &nY1) const
251 {
252  // Safety check: make sure that the first point is within the valid grid
253  if (nX0 >= m_nXGridMax)
254  nX0 = m_nXGridMax - 1;
255  else if (nX0 < 0)
256  nX0 = 0;
257 
258  if (nY0 >= m_nYGridMax)
259  nY0 = m_nYGridMax - 1;
260  else if (nY0 < 0)
261  nY0 = 0;
262 
263  // OK let's go
264  int
265  nDiffX = nX0 - nX1,
266  nDiffY = nY0 - nY1;
267 
268  if (nDiffX == 0)
269  {
270  // The two points have the same x co-ords, so we just need to constrain the y co-ord
271  if (nY1 < nY0)
272  {
273  nY1 = -1;
274 
275  do
276  {
277  nY1++;
278  } while (m_pRasterGrid->m_Cell[nX1][nY1].bBasementElevIsMissingValue());
279 
280  return;
281  }
282  else
283  {
284  nY1 = m_nYGridMax;
285 
286  do
287  {
288  nY1--;
289  } while (m_pRasterGrid->m_Cell[nX1][nY1].bBasementElevIsMissingValue());
290 
291  return;
292  }
293  }
294  else if (nDiffY == 0)
295  {
296  // The two points have the same y co-ords, so we just need to constrain the x co-ord
297  if (nX1 < nX0)
298  {
299  nX1 = -1;
300 
301  do
302  {
303  nX1++;
304  } while (m_pRasterGrid->m_Cell[nX1][nY1].bBasementElevIsMissingValue());
305 
306  return;
307  }
308  else
309  {
310  nX1 = m_nXGridMax;
311 
312  do
313  {
314  nX1--;
315  } while (m_pRasterGrid->m_Cell[nX1][nY1].bBasementElevIsMissingValue());
316 
317  return;
318  }
319  }
320  else
321  {
322  // The two points have different x co-ords and different y co-ords, so we have to work harder. First find which of the co-ords is the greatest distance outside the grid, and constrain that co-ord for efficiency (since this will reduce the number of times round the loop). Note that both may be inside the grid, if the incorrect co-ord is in the invalid margin, in which case arbitrarily contrain the x co-ord
323  int
324  nXDistanceOutside = 0,
325  nYDistanceOutside = 0;
326 
327  if (nX1 < 0)
328  nXDistanceOutside = -nX1;
329  else if (nX1 >= m_nXGridMax)
330  nXDistanceOutside = nX1 - m_nXGridMax + 1;
331 
332  if (nY1 < 0)
333  nYDistanceOutside = -nY1;
334  else if (nY1 >= m_nYGridMax)
335  nXDistanceOutside = nY1 - m_nYGridMax + 1;
336 
337  if (nXDistanceOutside >= nYDistanceOutside)
338  {
339  // Constrain the x co-ord
340  if (nX1 < nX0)
341  {
342  // The incorrect x co-ord is less than the correct x co-ord: constrain it and find the y co-ord
343  nX1 = -1;
344 
345  do
346  {
347  nX1++;
348 
349  nY1 = nY0 + nRound(((nX1 - nX0) * nDiffY) / static_cast<double>(nDiffX));
350  } while ((nY1 < 0) || (nY1 >= m_nYGridMax) || (m_pRasterGrid->m_Cell[nX1][nY1].bBasementElevIsMissingValue()));
351 
352  return;
353  }
354 
355  else
356  {
357  // The incorrect x co-ord is greater than the correct x-co-ord: constrain it and find the y co-ord
358  nX1 = m_nXGridMax;
359 
360  do
361  {
362  nX1--;
363 
364  nY1 = nY0 + nRound(((nX1 - nX0) * nDiffY) / static_cast<double>(nDiffX));
365  } while ((nY1 < 0) || (nY1 >= m_nYGridMax) || (m_pRasterGrid->m_Cell[nX1][nY1].bBasementElevIsMissingValue()));
366 
367  return;
368  }
369  }
370  else
371  {
372  // Constrain the y co-ord
373  if (nY1 < nY0)
374  {
375  // The incorrect y co-ord is less than the correct y-co-ord: constrain it and find the x co-ord
376  nY1 = -1;
377 
378  do
379  {
380  nY1++;
381 
382  nX1 = nX0 + nRound(((nY1 - nY0) * nDiffX) / static_cast<double>(nDiffY));
383  } while ((nX1 < 0) || (nX1 >= m_nXGridMax) || (m_pRasterGrid->m_Cell[nX1][nY1].bBasementElevIsMissingValue()));
384 
385  return;
386  }
387 
388  else
389  {
390  // The incorrect y co-ord is greater than the correct y co-ord: constrain it and find the x co-ord
391  nY1 = m_nYGridMax;
392 
393  do
394  {
395  nY1--;
396 
397  nX1 = nX0 + nRound(((nY1 - nY0) * nDiffX) / static_cast<double>(nDiffY));
398  } while ((nX1 < 0) || (nX1 >= m_nXGridMax) || (m_pRasterGrid->m_Cell[nX1][nY1].bBasementElevIsMissingValue()));
399 
400  return;
401  }
402  }
403  }
404 }

538 
539  Returns an integer point (grid CRS) which is the centroid of a polygon, given by a vector of grid CRS points. From https://stackoverflow.com/questions/2792443/finding-the-centroid-of-a-polygon
540 
541 ===============================================================================================================================*/
542 CGeom2DIPoint CSimulation::PtiPolygonCentroid(vector<CGeom2DIPoint> *pVIn)
543 {
544  CGeom2DIPoint PtiCentroid(0, 0);
545  int
546  nSize = static_cast<int>(pVIn->size()),
547  nX0 = 0, // Current vertex X
548  nY0 = 0, // Current vertex Y
549  nX1 = 0, // Next vertex X
550  nY1 = 0; // Next vertex Y
551 
552  double
553  dA = 0, // Partial signed area
554  dSignedArea = 0.0;
555 
556  // For all vertices except last
557  for (int i = 0; i < nSize - 1; ++i)
558  {
559  nX0 = pVIn->at(i).nGetX();
560  nY0 = pVIn->at(i).nGetY();
561  nX1 = pVIn->at(i + 1).nGetX();
562  nY1 = pVIn->at(i + 1).nGetY();
563 
564  dA = (nX0 * nY1) - (nX1 * nY0);
565  dSignedArea += dA;
566  PtiCentroid.AddXAddY((nX0 + nX1) * dA, (nY0 + nY1) * dA);
567  }
568 
569  // Do last vertex separately to avoid performing an expensive modulus operation in each iteration
570  nX0 = pVIn->at(nSize - 1).nGetX();
571  nY0 = pVIn->at(nSize - 1).nGetY();
572  nX1 = pVIn->at(0).nGetX();
573  nY1 = pVIn->at(0).nGetY();
574 
575  dA = (nX0 * nY1) - (nX1 * nY0);
576  dSignedArea += dA;
577  PtiCentroid.AddXAddY((nX0 + nX1) * dA, (nY0 + nY1) * dA);
578 
579  dSignedArea *= 0.5;
580  PtiCentroid.DivXDivY(6.0 * dSignedArea, 6.0 * dSignedArea);
581 
582  return PtiCentroid;
583 }

729 /*==============================================================================================================================
730 
731  Returns the signed angle BAC (in radians) subtended between three CGeom2DIPoints B A C. From http://stackoverflow.com/questions/3057448/angle-between-3-vertices
732 
733 *==============================================================================================================================*/
734 double CSimulation::dAngleSubtended(CGeom2DIPoint const *pPtiA, CGeom2DIPoint const *pPtiB, CGeom2DIPoint const *pPtiC)
735 {
736  double
737  dXDistBtoA = pPtiB->nGetX() - pPtiA->nGetX(),
738  dYDistBtoA = pPtiB->nGetY() - pPtiA->nGetY(),
739  dXDistCtoA = pPtiC->nGetX() - pPtiA->nGetX(),
740  dYDistCtoA = pPtiC->nGetY() - pPtiA->nGetY(),
741  dDotProduct = dXDistBtoA * dXDistCtoA + dYDistBtoA * dYDistCtoA,
742  dPseudoCrossProduct = dXDistBtoA * dYDistCtoA - dYDistBtoA * dXDistCtoA,
743  dAngle = atan2(dPseudoCrossProduct, dDotProduct);
744 
745  return dAngle;
746 }
