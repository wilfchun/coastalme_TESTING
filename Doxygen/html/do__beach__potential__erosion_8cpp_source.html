58 /*===============================================================================================================================
59 
60  Uses either the CERC equation or the Kamphuis (1990) equation to calculate potential (unconstrained) sediment movement between polygons
61 
62 ===============================================================================================================================*/
63 void CSimulation::DoAllPotentialBeachErosion(void)
64 {
65  // Do this for each coast
66  for (int nCoast = 0; nCoast < static_cast<int>(m_VCoast.size()); nCoast++)
67  {
68  int nNumPolygons = m_VCoast[nCoast].nGetNumPolygons();
69 
70  // Create a vector of pairs: the first value of the pair is the profile index, the second is the seaward length of that profile
71  vector<pair<int, double> > prVPolygonLength;
72  for (int nPoly = 0; nPoly < nNumPolygons; nPoly++)
73  prVPolygonLength.push_back(make_pair(nPoly, m_VCoast[nCoast].dGetPolygonLength(nPoly)));
74 
75  // Sort this pair vector in ascending order, so that the polygons with the shortest length (i.e. the most concave polygons) are first
76  sort(prVPolygonLength.begin(), prVPolygonLength.end(), bPolygonLengthPairCompare);
77 
78  // Do this for every coastal polygon in sequence of coastline concavity
79  for (int n = 0; n < nNumPolygons; n++)
80  {
81  int nThisPoly = prVPolygonLength[n].first;
82 
83  CGeomCoastPolygon* pPolygon = m_VCoast[nCoast].pGetPolygon(nThisPoly);
84 
85  // Calculate the average breaking wave height and angle along this polygon's segment of coastline
86  int
87  nStartNormal = pPolygon->nGetUpCoastProfile(),
88  nEndNormal = pPolygon->nGetDownCoastProfile(),
89  nCoastStartPoint = m_VCoast[nCoast].pGetProfile(nStartNormal)->nGetNumCoastPoint(),
90  nCoastEndPoint = m_VCoast[nCoast].pGetProfile(nEndNormal)->nGetNumCoastPoint(),
91  nCoastPoints = 0,
92  nActiveZonePoints = 0;
93 
94  double
95  dAvgBreakingWaveHeight = 0,
96  dAvgBreakingWaveOrientation = 0,
97  dAvgDeepWaterWavePeriod = 0,
98  dAvgFluxOrientation = 0,
99  dAvgBreakingDepth = 0,
100  dAvgBreakingDist = 0;
101 
102  // Calculate the average tangent to the polygon's coast segment, the average wave breaking height, the average depth of breaking, and the average distance of breaking, for this coast segment
103  for (int nCoastPoint = nCoastStartPoint; nCoastPoint < nCoastEndPoint-1; nCoastPoint++)
104  {
105  nCoastPoints++;
106  dAvgFluxOrientation += m_VCoast[nCoast].dGetFluxOrientation(nCoastPoint);
107 
108 
109  double dThisBreakingWaveHeight = m_VCoast[nCoast].dGetBreakingWaveHeight(nCoastPoint);
110  if (dThisBreakingWaveHeight != DBL_NODATA)
111  {
112  // We are in the active zone
113  nActiveZonePoints++;
114  dAvgBreakingWaveHeight += dThisBreakingWaveHeight;
115 
116  double dThisBreakingWaveOrientation = m_VCoast[nCoast].dGetBreakingWaveOrientation(nCoastPoint);
117  double dThisDeepWaterWavePeriod = m_VCoast[nCoast].dGetDeepWaterWavePeriod(nCoastPoint);
118 
119  dAvgBreakingWaveOrientation += dThisBreakingWaveOrientation;
120  dAvgDeepWaterWavePeriod += dThisDeepWaterWavePeriod;
121 
122  dAvgBreakingDepth += m_VCoast[nCoast].dGetDepthOfBreaking(nCoastPoint);
123 
124  dAvgBreakingDist += (m_VCoast[nCoast].nGetBreakingDistance(nCoastPoint) * m_dCellSide);
125  }
126  }
127 
128  // Calc the averages
129  dAvgFluxOrientation /= nCoastPoints;
130 
131  if (nActiveZonePoints > 0)
132  {
133  // Only calculate sediment movement if the polygon has at least one coast point in its coastline segment which is in the active zone
134  dAvgBreakingWaveHeight /= nActiveZonePoints;
135  dAvgBreakingWaveOrientation /= nActiveZonePoints;
136  dAvgDeepWaterWavePeriod /= nActiveZonePoints;
137  dAvgBreakingDepth /= nActiveZonePoints;
138  dAvgBreakingDist /= nActiveZonePoints;
139 
140  // Get the coast handedness, and (based on the average tangent) calculate the direction towards which a coastline-normal profile points
141  int nSeaHand = m_VCoast[nCoast].nGetSeaHandedness();
142  double dNormalOrientation;
143  if (nSeaHand == RIGHT_HANDED)
144  dNormalOrientation = dKeepWithin360(dAvgFluxOrientation - 90);
145  else
146  dNormalOrientation = dKeepWithin360(dAvgFluxOrientation + 90);
147 
148  // Determine dThetaBr, the angle between the coastline-normal orientation and the breaking wave orientation (the direction FROM which the waves move). This tells us whether the sediment movement is up-coast (-ve) or down-coast (+ve)
149  double dThetaBr = dNormalOrientation - dAvgBreakingWaveOrientation;
150 
151  bool bDownCoast = true;
152  if (dThetaBr < 0)
153  bDownCoast = false;
154 
155  // And save the direction of sediment movement in the polygon object
156  pPolygon->SetDownCoastThisTimestep(bDownCoast);
157 
158  // Now that we have the direction of sediment movement, normalize dThetaBr to be always +ve so that subsequent calculations are clearer
159  dThetaBr = tAbs(dThetaBr);
160 
161  // Safety check: not sure why we need this, but get occasional big values
162  dThetaBr = tMin(dThetaBr, 90.0);
163 
164  // Calculate the immersed weight of sediment transport
165  double dImmersedWeightTransport = 0;
166  if (m_nBeachErosionDepositionEquation == UNCONS_SEDIMENT_EQUATION_CERC)
167  {
168  /*
169  Use the CERC equation (Komar and Inman, 1970; USACE, 1984), this describes the immersive weight transport of sand (i.e. sand transport in suspension). Depth-integrated alongshore volumetric sediment transport is a function of breaking wave height Hb and angle αb:
170 
171  Qls = Kls * Hb^(5/2) * sin(2 * αb)
172 
173  where Kls is a transport coefficient which varies between 0.4 to 0.79
174  */
175  dImmersedWeightTransport = m_dKLS * pow(dAvgBreakingWaveHeight, 2.5) * sin((PI / 180) * 2 * dThetaBr);
176  }
177  else if (m_nBeachErosionDepositionEquation == UNCONS_SEDIMENT_EQUATION_KAMPHUIS)
178  {
179  /*
180  Use the Kamphuis (1990) equation to estimate the immersive weight transport of sand in kg/s:
181 
182  Qls = 2.33 * (Tp^(1.5)) * (tanBeta^(0.75)) * (d50^(-0.25)) * (Hb^2) * (sin(2 * αb)^(0.6))
183 
184  where:
185 
186  Tp = peak wave period
187  tanBeta = beach slope, defined as the ratio of the water depth at the breaker line and the distance from the still water beach line to the breaker line
188  d50 = median particle size in surf zone (m)
189  */
190 
191  if (dAvgBreakingDist > 0)
192  {
193  double dBeachSlope = dAvgBreakingDepth / dAvgBreakingDist;
194 
195  double dD50 = pPolygon->dGetAvgUnconsD50();
196  if (dD50 > 0)
197  // Note that we use a calibration constant here (m_dKamphuis)
198  dImmersedWeightTransport = m_dKamphuis * 2.33 * pow(dAvgDeepWaterWavePeriod, 1.5) * pow(dBeachSlope, 0.75) * pow(dD50, -0.25) * pow(dAvgBreakingWaveHeight, 2) * pow(sin((PI / 180) * 2 * dThetaBr), 0.6);
199  }
200  }
201 
202  // Convert from immersed weight rate to bulk volumetric (sand and voids) transport rate (m3/s)
203  double dSedimentVol = m_dInmersedToBulkVolumetric * dImmersedWeightTransport;
204 
205  // Convert to a volume during this timestep
206  dSedimentVol *= (m_dTimeStep * 3600);
207 
208  // Convert to a depth in m
209  double dSedimentDepth = dSedimentVol / m_dCellArea;
210 
211  // If this is just a tiny depth, do nothing
212  if (dSedimentDepth < SEDIMENT_ELEV_TOLERANCE)
213  dSedimentDepth = 0;
214 
215 // LogStream << m_ulIter << ": polygon = " << nThisPoly << " nActiveZonePoints = " << nActiveZonePoints << " dAvgBreakingWaveHeight = " << dAvgBreakingWaveHeight << " dAvgFluxOrientation = " << dAvgFluxOrientation << " dNormalOrientation = " << dNormalOrientation << " dAvgBreakingWaveOrientation = " << dAvgBreakingWaveOrientation << " potential sediment transport this timestep = " << dSedimentDepth << " m " << (bDownCoast ? "DOWN" : "UP") << " coast" << endl;
216 
217 
218  // Store the potential erosion value for this polygon
219  pPolygon->AddDeltaPotentialTotalSediment(-dSedimentDepth);
220 // LogStream << "\tPotential erosion on polygon " << nThisPoly << " -dSedimentDepth = " << -dSedimentDepth << endl;
221  }
222 // else
223 // LogStream << m_ulIter << ": polygon = " << nThisPoly << " NOT IN ACTIVE ZONE dAvgFluxOrientation = " << dAvgFluxOrientation << endl;
224  }
225  }
226 }
