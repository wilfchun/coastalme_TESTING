56 /*===============================================================================================================================
57 
58  Uses either the CERC equation or the Kamphuis (1990) equation to calculate potential (unconstrained) sediment movement between polygons
59 
60 ===============================================================================================================================*/
61 void CSimulation::DoAllPotentialBeachErosion(void)
62 {
63  // Do this for each coast
64  for (int nCoast = 0; nCoast < static_cast<int>(m_VCoast.size()); nCoast++)
65  {
66  int nNumPolygons = m_VCoast[nCoast].nGetNumPolygons();
67 
68  // Create a vector of pairs: the first value of the pair is the profile index, the second is the seaward length of that profile
69  vector<pair<int, double>> prVPolygonLength;
70  for (int nPoly = 0; nPoly < nNumPolygons; nPoly++)
71  prVPolygonLength.push_back(make_pair(nPoly, m_VCoast[nCoast].dGetPolygonLength(nPoly)));
72 
73  // Sort this pair vector in ascending order, so that the polygons with the shortest length (i.e. the most concave polygons) are first
74  sort(prVPolygonLength.begin(), prVPolygonLength.end(), bPolygonLengthPairCompare);
75 
76  // Do this for every coastal polygon in sequence of coastline concavity
77  for (int n = 0; n < nNumPolygons; n++)
78  {
79  int nThisPoly = prVPolygonLength[n].first;
80 
81  CGeomCoastPolygon *pPolygon = m_VCoast[nCoast].pGetPolygon(nThisPoly);
82 
83  // Calculate the average breaking wave height and angle along this polygon's segment of coastline
84  int
85  nStartNormal = pPolygon->nGetUpCoastProfile(),
86  nEndNormal = pPolygon->nGetDownCoastProfile(),
87  nCoastStartPoint = m_VCoast[nCoast].pGetProfile(nStartNormal)->nGetNumCoastPoint(),
88  nCoastEndPoint = m_VCoast[nCoast].pGetProfile(nEndNormal)->nGetNumCoastPoint(),
89  nCoastPoints = 0,
90  nActiveZonePoints = 0;
91 
92  double
93  dAvgBreakingWaveHeight = 0,
94  dAvgBreakingWaveAngle = 0,
95  dAvgDeepWaterWavePeriod = 0,
96  dAvgFluxOrientation = 0,
97  dAvgBreakingDepth = 0,
98  dAvgBreakingDist = 0;
99 
100  // Calculate the average tangent to the polygon's coast segment, the average wave breaking height, the average depth of breaking, and the average distance of breaking, for this coast segment
101  for (int nCoastPoint = nCoastStartPoint; nCoastPoint < nCoastEndPoint - 1; nCoastPoint++)
102  {
103  nCoastPoints++;
104  dAvgFluxOrientation += m_VCoast[nCoast].dGetFluxOrientation(nCoastPoint);
105 
106  double dThisBreakingWaveHeight = m_VCoast[nCoast].dGetBreakingWaveHeight(nCoastPoint);
107  if (dThisBreakingWaveHeight != DBL_NODATA)
108  {
109  // We are in the active zone
110  nActiveZonePoints++;
111  dAvgBreakingWaveHeight += dThisBreakingWaveHeight;
112 
113  double dThisBreakingWaveAngle = m_VCoast[nCoast].dGetBreakingWaveAngle(nCoastPoint);
114  double dThisDeepWaterWavePeriod = m_VCoast[nCoast].dGetCoastDeepWaterWavePeriod(nCoastPoint);
115 
116  dAvgBreakingWaveAngle += dThisBreakingWaveAngle;
117  dAvgDeepWaterWavePeriod += dThisDeepWaterWavePeriod;
118 
119  dAvgBreakingDepth += m_VCoast[nCoast].dGetDepthOfBreaking(nCoastPoint);
120 
121  dAvgBreakingDist += (m_VCoast[nCoast].nGetBreakingDistance(nCoastPoint) * m_dCellSide);
122  }
123  }
124 
125  // Calc the averages
126  dAvgFluxOrientation /= nCoastPoints;
127 
128  if (nActiveZonePoints > 0)
129  {
130  // Only calculate sediment movement if the polygon has at least one coast point in its coastline segment which is in the active zone
131  dAvgBreakingWaveHeight /= nActiveZonePoints;
132  dAvgBreakingWaveAngle /= nActiveZonePoints;
133  dAvgDeepWaterWavePeriod /= nActiveZonePoints;
134  dAvgBreakingDepth /= nActiveZonePoints;
135  dAvgBreakingDist /= nActiveZonePoints;
136 
137  // Get the coast handedness, and (based on the average tangent) calculate the direction towards which a coastline-normal profile points
138  int nSeaHand = m_VCoast[nCoast].nGetSeaHandedness();
139  double dNormalOrientation;
140  if (nSeaHand == RIGHT_HANDED)
141  dNormalOrientation = dKeepWithin360(dAvgFluxOrientation - 90);
142  else
143  dNormalOrientation = dKeepWithin360(dAvgFluxOrientation + 90);
144 
145  // Determine dThetaBr, the angle between the coastline-normal orientation and the breaking wave orientation (the direction FROM which the waves move). This tells us whether the sediment movement is up-coast (-ve) or down-coast (+ve)
146  double dThetaBr = dNormalOrientation - dAvgBreakingWaveAngle;
147 
148  bool bDownCoast = true;
149  if (dThetaBr < 0)
150  bDownCoast = false;
151 
152  // And save the direction of sediment movement in the polygon object
153  pPolygon->SetDownCoastThisIter(bDownCoast);
154 
155  // Now that we have the direction of sediment movement, normalize dThetaBr to be always +ve so that subsequent calculations are clearer
156  dThetaBr = tAbs(dThetaBr);
157 
158  // Safety check: not sure why we need this, but get occasional big values
159  dThetaBr = tMin(dThetaBr, 90.0);
160 
161  // Calculate the immersed weight of sediment transport
162  double dImmersedWeightTransport = 0;
163  if (m_nBeachErosionDepositionEquation == UNCONS_SEDIMENT_EQUATION_CERC)
164  {
165  /*
166  Use the CERC equation (Komar and Inman, 1970; USACE, 1984), this describes the immersive weight transport of sand (i.e. sand transport in suspension). Depth-integrated alongshore volumetric sediment transport is a function of breaking wave height Hb and angle αb:
167 
168  Qls = Kls * Hb^(5/2) * sin(2 * αb)
169 
170  where Kls is a transport coefficient which varies between 0.4 to 0.79
171  */
172  dImmersedWeightTransport = m_dKLS * pow(dAvgBreakingWaveHeight, 2.5) * sin((PI / 180) * 2 * dThetaBr);
173  }
174  else if (m_nBeachErosionDepositionEquation == UNCONS_SEDIMENT_EQUATION_KAMPHUIS)
175  {
176  /*
177  Use the Kamphuis (1990) equation to estimate the immersive weight transport of sand in kg/s:
178 
179  Qls = 2.33 * (Tp^(1.5)) * (tanBeta^(0.75)) * (d50^(-0.25)) * (Hb^2) * (sin(2 * αb)^(0.6))
180 
181  where:
182 
183  Tp = peak wave period
184  tanBeta = beach slope, defined as the ratio of the water depth at the breaker line and the distance from the still water beach line to the breaker line
185  d50 = median particle size in surf zone (m)
186  */
187 
188  if (dAvgBreakingDist > 0)
189  {
190  double dBeachSlope = dAvgBreakingDepth / dAvgBreakingDist;
191 
192  double dD50 = pPolygon->dGetAvgUnconsD50();
193  if (dD50 > 0)
194  // Note that we use a calibration constant here (m_dKamphuis)
195  dImmersedWeightTransport = m_dKamphuis * 2.33 * pow(dAvgDeepWaterWavePeriod, 1.5) * pow(dBeachSlope, 0.75) * pow(dD50, -0.25) * pow(dAvgBreakingWaveHeight, 2) * pow(sin((PI / 180) * 2 * dThetaBr), 0.6);
196  }
197  }
198 
199  // Convert from immersed weight rate to bulk volumetric (sand and voids) transport rate (m3/s)
200  double dSedimentVol = m_dInmersedToBulkVolumetric * dImmersedWeightTransport;
201 
202  // Convert to a volume during this timestep
203  dSedimentVol *= (m_dTimeStep * 3600);
204 
205  // Convert to a depth in m
206  double dSedimentDepth = dSedimentVol / m_dCellArea;
207 
208  // If this is just a tiny depth, do nothing
209  if (dSedimentDepth < SEDIMENT_ELEV_TOLERANCE)
210  dSedimentDepth = 0;
211 
212  // LogStream << m_ulIter << ": polygon = " << nThisPoly << " nActiveZonePoints = " << nActiveZonePoints << " dAvgBreakingWaveHeight = " << dAvgBreakingWaveHeight << " dAvgFluxOrientation = " << dAvgFluxOrientation << " dNormalOrientation = " << dNormalOrientation << " dAvgBreakingWaveAngle = " << dAvgBreakingWaveAngle << " potential sediment transport this timestep = " << dSedimentDepth << " m " << (bDownCoast ? "DOWN" : "UP") << " coast" << endl;
213 
214  // Store the potential erosion value for this polygon
215  pPolygon->AddDeltaPotentialTotalSediment(-dSedimentDepth);
216  // LogStream << "\tPotential erosion on polygon " << nThisPoly << " -dSedimentDepth = " << -dSedimentDepth << endl;
217  }
218  // else
219  // LogStream << m_ulIter << ": polygon = " << nThisPoly << " NOT IN ACTIVE ZONE dAvgFluxOrientation = " << dAvgFluxOrientation << endl;
220  }
221  }
222 }
