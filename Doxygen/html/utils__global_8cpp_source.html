98 Checks to see if a string can be read as a valid integer, from https://stackoverflow.com/questions/2844817/how-do-i-check-if-a-c-string-is-an-int
99 
100 ==============================================================================================================================*/
101 bool bIsStringValidInt(string& str)
102 {
103  // Trim leading whitespace
104  size_t nPos = str.find_first_not_of(" \t");
105  if (nPos != string::npos)
106  str = str.substr(nPos);
107 
108  // If the first character is the sign, remove it
109  if ((str[0] == '-') || (str[0] == '+'))
110  str.erase(0, 1);
111 
112  // Now check that the string contains only numbers
113  return (str.find_first_not_of("0123456789") == string::npos);
114 }
