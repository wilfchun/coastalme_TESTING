40 int CSimulation::nUpdateGrid(void)
41 {
42  // Go through all cells in the raster grid and calculate some this-timestep totals
43  m_dThisIterTopElevMax = -DBL_MAX;
44  m_dThisIterTopElevMin = DBL_MAX;
45  for (int nX = 0; nX < m_nXGridMax; nX++)
46  {
47  for (int nY = 0; nY < m_nYGridMax; nY++)
48  {
49  if (m_pRasterGrid->m_Cell[nX][nY].bIsCoastline())
50  m_ulThisIterNumCoastCells++;
51 
52  if (m_pRasterGrid->m_Cell[nX][nY].bIsInContiguousSea())
53  {
54  // Is a sea cell
55  m_dThisIterTotSeaDepth += m_pRasterGrid->m_Cell[nX][nY].dGetSeaDepth();
56  }
57 
58  double dTopElev = m_pRasterGrid->m_Cell[nX][nY].dGetOverallTopElev();
59 
60  // Get highest and lowest elevations of the top surface of the DEM
61  if (dTopElev > m_dThisIterTopElevMax)
62  m_dThisIterTopElevMax = dTopElev;
63 
64  if (dTopElev < m_dThisIterTopElevMin)
65  m_dThisIterTopElevMin = dTopElev;
66  }
67  }
68 
69  // No sea cells?
70  if (m_ulThisIterNumSeaCells == 0)
71  // All land, assume this is an error
72  return RTN_ERR_NOSEACELLS;
73 
74  // Now go through all cells again and sort out suspended sediment load
75  double dSuspPerSeaCell = static_cast<double>(m_ldGTotSuspendedSediment / m_ulThisIterNumSeaCells);
76  for (int nX = 0; nX < m_nXGridMax; nX++)
77  {
78  for (int nY = 0; nY < m_nYGridMax; nY++)
79  {
80  if (m_pRasterGrid->m_Cell[nX][nY].bIsInContiguousSea())
81  m_pRasterGrid->m_Cell[nX][nY].SetSuspendedSediment(dSuspPerSeaCell);
82  }
83  }
84 
85  // Go along each coastline and update the grid with landform attributes, ready for next timestep
86  for (int i = 0; i < static_cast<int>(m_VCoast.size()); i++)
87  {
88  for (int j = 0; j < m_VCoast[i].nGetCoastlineSize(); j++)
89  {
90  int nRet = nLandformToGrid(i, j);
91  if (nRet != RTN_OK)
92  return nRet;
93  }
94  }
95 
96  return RTN_OK;
97 }
