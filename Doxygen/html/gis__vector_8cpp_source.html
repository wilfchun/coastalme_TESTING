49 int CSimulation::nReadVectorGISFile(int const nDataItem)
50 {
51  int
52  nMaxLayer = 0,
53  nNeedGeometry = 0;
54 
55  string
56  strGISFile,
57  strGeometry;
58 
59  // Set up file name and constraints
60  switch (nDataItem)
61  {
62  case (DEEP_WATER_WAVE_STATIONS_VEC):
63  strGISFile = m_strDeepWaterWaveStationsShapefile;
64  nMaxLayer = DEEP_WATER_WAVE_STATIONS_MAX_LAYER;
65  nNeedGeometry = DEEP_WATER_WAVE_STATIONS_POINT_GEOMETRY;
66  break;
67 
68  case (SEDIMENT_INPUT_EVENT_LOCATION_VEC):
69  strGISFile = m_strSedimentInputEventShapefile;
70  nMaxLayer = SEDIMENT_INPUT_EVENT_LOCATION_MAX_LAYER;
71  if (m_bSedimentInputAtPoint || m_bSedimentInputAtCoast)
72  nNeedGeometry = SEDIMENT_INPUT_EVENT_LOCATION_POINT_GEOMETRY;
73  else if (m_bSedimentInputAlongLine)
74  nNeedGeometry = SEDIMENT_INPUT_EVENT_LOCATION_LINE_GEOMETRY;
75  break;
76  }
77 
78  // Open the GDAL/OGR datasource
79  GDALDataset *pOGRDataSource = (GDALDataset *)GDALOpenEx(strGISFile.c_str(), GDAL_OF_VECTOR, NULL, NULL, NULL);
80  if (pOGRDataSource == NULL)
81  {
82  // Can't open file (note will already have sent GDAL error message to stdout)
83  cerr << ERR << "cannot open " << strGISFile << " for input: " << CPLGetLastErrorMsg() << endl;
84  return RTN_ERR_VECTOR_FILE_READ;
85  }
86 
87  // Find out number of layers, and compare with the required number
88  int nLayer = pOGRDataSource->GetLayerCount();
89  if (nLayer > nMaxLayer)
90  LogStream << WARN << "need " << nMaxLayer << (nMaxLayer > 1 ? "layers" : "layer") << " in " << strGISFile << ", " << nLayer << " found. Only the first " << nMaxLayer << (nMaxLayer > 1 ? "layers" : "layer") << " will be read." << endl;
91 
92  for (int n = 0; n < nMaxLayer; n++)
93  {
94  // Open this layer
95  OGRLayer *pOGRLayer;
96  pOGRLayer = pOGRDataSource->GetLayer(n);
97 
98  // Get features from the layer
99  OGRFeature *pOGRFeature;
100 
101  // Make sure we are at the beginning of the layer
102  pOGRLayer->ResetReading();
103 
104  // Now iterate for all features in the layer
105  while ((pOGRFeature = pOGRLayer->GetNextFeature()) != NULL)
106  {
107  // First get the geometry for this feature
108  OGRGeometry *pOGRGeometry;
109  pOGRGeometry = pOGRFeature->GetGeometryRef();
110  if (pOGRGeometry == NULL)
111  {
112  cerr << ERR << " null geometry in " << strGISFile << "." << endl;
113  return RTN_ERR_VECTOR_FILE_READ;
114  }
115 
116  // Now get the geometry type
117  int
118  nGeometry = wkbFlatten(pOGRGeometry->getGeometryType()),
119  nThisGeometry = 0;
120 
121  switch (nGeometry)
122  {
123  case wkbPoint:
124  nThisGeometry = VEC_GEOMETRY_POINT;
125  strGeometry = "point";
126  break;
127 
128  case wkbLineString:
129  nThisGeometry = VEC_GEOMETRY_LINE;
130  strGeometry = "line";
131  break;
132 
133  case wkbPolygon:
134  nThisGeometry = VEC_GEOMETRY_POLYGON;
135  strGeometry = "polygon";
136  break;
137 
138  default:
139  // NOTE may need wkbMultiLineString or similar for channel network
140  nThisGeometry = VEC_GEOMETRY_OTHER;
141  strGeometry = "other";
142  break;
143  }
144 
145  // Have we got the expected geometry type?
146  if (nThisGeometry != nNeedGeometry)
147  {
148  // Error, we do not have the desired geometry
149  string strNeedGeometry;
150  switch (nNeedGeometry)
151  {
152  case VEC_GEOMETRY_POINT:
153  strNeedGeometry = "point";
154  break;
155 
156  case VEC_GEOMETRY_LINE:
157  strNeedGeometry = "line";
158  break;
159 
160  case VEC_GEOMETRY_POLYGON:
161  strNeedGeometry = "polygon";
162  break;
163 
164  case VEC_GEOMETRY_OTHER:
165  strNeedGeometry = "other";
166  break;
167  }
168 
169  cerr << strGeometry << " data found in " << strGISFile << ", but " << strNeedGeometry << " data is needed" << endl;
170  return RTN_ERR_VECTOR_FILE_READ;
171  }
172 
173  // The geometry type is OK, so process the geometry data
174  int nPoints = 0;
175  OGRPoint *pOGRPoint;
176  OGRLineString *pOGRLineString;
177  switch (nDataItem)
178  {
179  case (DEEP_WATER_WAVE_STATIONS_VEC):
180  // Point data
181  pOGRPoint = (OGRPoint *)pOGRGeometry;
182 
183  // Convert the wave station co-ords to grid CRS and store them: we will use these in the spatial interpolation of deep water waves
184  m_VdDeepWaterWaveStationX.push_back(dExtCRSXToGridX(pOGRPoint->getX()));
185  m_VdDeepWaterWaveStationY.push_back(dExtCRSYToGridY(pOGRPoint->getY()));
186  break;
187 
188  case (SEDIMENT_INPUT_EVENT_LOCATION_VEC):
189  if (m_bSedimentInputAtPoint || m_bSedimentInputAtCoast)
190  {
191  // Point data
192  pOGRPoint = (OGRPoint *)pOGRGeometry;
193 
194  // Convert the point co-ords to grid CRS and store them
195  m_VdSedimentInputLocationX.push_back(dExtCRSXToGridX(pOGRPoint->getX()));
196  m_VdSedimentInputLocationY.push_back(dExtCRSYToGridY(pOGRPoint->getY()));
197  }
198  else if (m_bSedimentInputAlongLine)
199  {
200  // Line data
201  pOGRLineString = (OGRLineString *)pOGRGeometry;
202 
203  nPoints = pOGRLineString->getNumPoints();
204  for (int i = 0; i < nPoints; i++)
205  {
206  // Convert the co-ords for each point in the line to grid CRS and store them
207  m_VdSedimentInputLocationX.push_back(dExtCRSXToGridX(pOGRLineString->getX(i)));
208  m_VdSedimentInputLocationY.push_back(dExtCRSYToGridY(pOGRLineString->getY(i)));
209  }
210  }
211  break;
212  }
213 
214  // Now get the attributes of this feature
215  OGRFeatureDefn *pOGRFeatureDefn = pOGRLayer->GetLayerDefn();
216 
217  int
218  nFieldIndex = -1,
219  nID = -1;
220 
221  switch (nDataItem)
222  {
223  case (DEEP_WATER_WAVE_STATIONS_VEC):
224  // First get the station ID
225  nFieldIndex = pOGRFeatureDefn->GetFieldIndex(DEEP_WATER_WAVE_STATION_ID.c_str());
226  if (nFieldIndex == -1)
227  {
228  // Can't find this field in the vector file
229  cerr << ERR << "cannot find " << DEEP_WATER_WAVE_STATION_ID << " field in " << strGISFile << ": " << CPLGetLastErrorMsg() << endl;
230  return RTN_ERR_VECTOR_FILE_READ;
231  }
232 
233  // Get the Station ID for this point
234  nID = pOGRFeature->GetFieldAsInteger(nFieldIndex);
235  m_VnDeepWaterWaveStationID.push_back(nID);
236 
237  break;
238 
239  case (SEDIMENT_INPUT_EVENT_LOCATION_VEC):
240  // First get the station ID
241  nFieldIndex = pOGRFeatureDefn->GetFieldIndex(SEDIMENT_INPUT_EVENT_LOCATION_ID.c_str());
242  if (nFieldIndex == -1)
243  {
244  // Can't find this field in the vector file
245  cerr << ERR << "cannot find " << SEDIMENT_INPUT_EVENT_LOCATION_ID << " field in " << strGISFile << ": " << CPLGetLastErrorMsg() << endl;
246  return RTN_ERR_VECTOR_FILE_READ;
247  }
248 
249  // Get the Station ID for this point (note that we read it as an integer, not caring what type of field is actually in the shapefile)
250  nID = pOGRFeature->GetFieldAsInteger(nFieldIndex);
251 
252  if (m_bSedimentInputAtPoint || m_bSedimentInputAtCoast)
253  // Save the ID for the point
254  m_VnSedimentInputLocationID.push_back(nID);
255 
256  else if (m_bSedimentInputAlongLine)
257  {
258  // Save the ID for every point in the line
259  for (int i = 0; i < nPoints; i++)
260  m_VnSedimentInputLocationID.push_back(nID);
261  }
262 
263  break;
264  }
265 
266  // Get rid of the Feature object
267  OGRFeature::DestroyFeature(pOGRFeature);
268  }
269  }
270 
271  // Save some info, to be shown in the text output
272  switch (nDataItem)
273  {
274  case (DEEP_WATER_WAVE_STATIONS_VEC):
275  m_strOGRDWWVDriverCode = pOGRDataSource->GetDriverName();
276  m_strOGRDWWVDataType = "integer";
277  m_strOGRDWWVGeometry = strGeometry;
278  break;
279 
280  case (SEDIMENT_INPUT_EVENT_LOCATION_VEC):
281  m_strOGRSedInputDriverCode = pOGRDataSource->GetDriverName();
282  m_strOGRSedInputGeometry = "integer";
283  m_strOGRSedInputDataType = strGeometry;
284  break;
285  }
286 
287  // Clean up: get rid of the data source object
288  GDALClose(pOGRDataSource);
289 
290  return RTN_OK;
291 }
