198 int CRWCoast::nGetCoastPointGivenCell(CGeom2DIPoint *pPtiCell)
199 {
200  for (int nCoastPoint = 0; nCoastPoint < m_ILCellsMarkedAsCoastline.nGetSize(); nCoastPoint++)
201  {
202  if (m_ILCellsMarkedAsCoastline[nCoastPoint] == pPtiCell)
203  {
204  return nCoastPoint;
205  }
206  }
207 
208  // This cell is not under a coastline, so try the adjacent cells
209  int
210  n = -1,
211  nX = pPtiCell->nGetX(),
212  nY = pPtiCell->nGetY(),
213  nXAdj = 0,
214  nYAdj = 0;
215 
216  while (n <= 7)
217  {
218  switch (++n)
219  {
220  case 0:
221  nXAdj = nX;
222  nYAdj = nY - 1;
223  break;
224  case 1:
225  nXAdj = nX + 1;
226  nYAdj = nY - 1;
227  break;
228  case 2:
229  nXAdj = nX + 1;
230  nYAdj = nY;
231  break;
232  case 3:
233  nXAdj = nX + 1;
234  nYAdj = nY + 1;
235  break;
236  case 4:
237  nXAdj = nX;
238  nYAdj = nY + 1;
239  break;
240  case 5:
241  nXAdj = nX - 1;
242  nYAdj = nY + 1;
243  break;
244  case 6:
245  nXAdj = nX - 1;
246  nYAdj = nY;
247  break;
248  case 7:
249  nXAdj = nX - 1;
250  nYAdj = nY - 1;
251  break;
252  }
253 
254  CGeom2DIPoint PtiTmp(nXAdj, nYAdj);
255  for (int nCoastPoint = 0; nCoastPoint < m_ILCellsMarkedAsCoastline.nGetSize(); nCoastPoint++)
256  {
257  if (m_ILCellsMarkedAsCoastline[nCoastPoint] == &PtiTmp)
258  {
259  *pPtiCell = PtiTmp;
260  return nCoastPoint;
261  }
262  }
263  }
264 
265  return INT_NODATA;
266 }

399 {

408  return INT_NODATA;
