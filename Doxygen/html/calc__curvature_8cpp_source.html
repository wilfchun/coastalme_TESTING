47 void CSimulation::DoCoastCurvature(int const nCoast, int const nHandedness)
48 {
49  int nCoastSize = m_VCoast[nCoast].nGetCoastlineSize();
50 
51  // Start with detailed curvature, do every point on the coastline, apart from the first and last points
52  for (int nThisCoastPoint = 1; nThisCoastPoint < (nCoastSize-1); nThisCoastPoint++)
53  {
54  // Calculate the signed curvature based on this point, and the points before and after
55  double dCurvature = dCalcCurvature(nHandedness, m_VCoast[nCoast].pPtGetCoastlinePointExtCRS(nThisCoastPoint-1), m_VCoast[nCoast].pPtGetCoastlinePointExtCRS(nThisCoastPoint), m_VCoast[nCoast].pPtGetCoastlinePointExtCRS(nThisCoastPoint+1));
56 
57  // Set the detailed curvature
58  m_VCoast[nCoast].SetDetailedCurvature(nThisCoastPoint, dCurvature);
59  }
60 
61  // Set the curvature for the first and last coastline points
62  double dTemp = m_VCoast[nCoast].dGetDetailedCurvature(1);
63  m_VCoast[nCoast].SetDetailedCurvature(0, dTemp);
64 
65  dTemp = m_VCoast[nCoast].dGetDetailedCurvature(nCoastSize-2);
66  m_VCoast[nCoast].SetDetailedCurvature(nCoastSize-1, dTemp);
67 
68  // Now create the smoothed curvature
69  int const WINDOWSIZE = 11; // TODO make this a user setting, must be odd
70  int const nHalfWindow = WINDOWSIZE / 2;
71 
72  // Apply a running mean smoothing filter, with a variable window size at both ends of the line
73  for (int i = 0; i < nCoastSize; i++)
74  {
75  int nTmpWindow = 0;
76  double dWindowTot = 0;
77  for (int j = -nHalfWindow; j < WINDOWSIZE - nHalfWindow; j++)
78  {
79  // For points at both ends of the coastline, use a smaller window
80  int const k = i+j;
81 
82  if ((k < 0) || (k >= nCoastSize))
83  continue;
84 
85  dWindowTot += m_VCoast[nCoast].dGetDetailedCurvature(k);
86  nTmpWindow++;
87  }
88 
89  m_VCoast[nCoast].SetSmoothCurvature(i, dWindowTot / static_cast<double>(nTmpWindow));
90  }
91 
92  // Now calculate the mean and standard deviation of each set of curvature values
93  vector<double>* pVDetailed = m_VCoast[nCoast].pVGetDetailedCurvature();
94 
95  double
96  dSum = accumulate(pVDetailed->begin(), pVDetailed->end(), 0.0),
97  dMean = dSum / static_cast<double>(pVDetailed->size());
98 
99  m_VCoast[nCoast].SetDetailedCurvatureMean(dMean);
100 
101  double
102  dSquareSum = inner_product(pVDetailed->begin(), pVDetailed->end(), pVDetailed->begin(), 0.0),
103  dSTD = sqrt(dSquareSum / static_cast<double>(pVDetailed->size()) - dMean * dMean);
104 
105  m_VCoast[nCoast].SetDetailedCurvatureSTD(dSTD);
106 
107  vector<double>* pVSmooth = m_VCoast[nCoast].pVGetSmoothCurvature();
108  dSum = accumulate(pVSmooth->begin(), pVSmooth->end(), 0.0),
109  dMean = dSum / static_cast<double>(pVSmooth->size());
110  m_VCoast[nCoast].SetSmoothCurvatureMean(dMean);
111  dSquareSum = inner_product(pVSmooth->begin(), pVSmooth->end(), pVSmooth->begin(), 0.0),
112  dSTD = sqrt(dSquareSum / static_cast<double>(pVSmooth->size()) - dMean * dMean);
113  m_VCoast[nCoast].SetSmoothCurvatureSTD(dSTD);
114 
115  int
116  nMaxConvexSmoothedCoastPoint = -1,
117  nMaxConvexDetailedCoastPoint = -1;
118  double
119  dMaxConvexDetailed = DBL_MAX,
120  dMaxConvexSmoothed = DBL_MAX;
121  for (int mm = 0; mm < nCoastSize; mm++)
122  {
123 // CGeom2DIPoint PtiThis = *m_VCoast[nCoast].pPtiGetCellMarkedAsCoastline(mm);
124 // CGeom2DPoint PtThis = *m_VCoast[nCoast].pPtGetCoastlinePointExtCRS(mm);
125 
126 // LogStream << m_ulIteration << ": at coast point (" << mm << ") [" << PtiThis.nGetX() << "][" << PtiThis.nGetY() << "] = {" << PtThis.dGetX() << ", " << PtThis.dGetY() << "} curvature = " << m_VCoast[nCoast].dGetDetailedCurvature(mm) << " smoothed curvature = " << m_VCoast[nCoast].dGetSmoothCurvature(mm) << endl;
127 
128  if (m_VCoast[nCoast].dGetDetailedCurvature(mm) < dMaxConvexDetailed)
129  {
130  dMaxConvexDetailed = m_VCoast[nCoast].dGetDetailedCurvature(mm);
131  nMaxConvexDetailedCoastPoint = mm;
132  }
133 
134  if (m_VCoast[nCoast].dGetSmoothCurvature(mm) < dMaxConvexSmoothed)
135  {
136  dMaxConvexSmoothed = m_VCoast[nCoast].dGetSmoothCurvature(mm);
137  nMaxConvexSmoothedCoastPoint = mm;
138  }
139 
140  }
141 
142  if (dMaxConvexDetailed == 0)
143  {
144  // We have a straight-line coast, so set the point of maximum detailed convexity at the coast mid-point
145  nMaxConvexDetailedCoastPoint =
146  nMaxConvexSmoothedCoastPoint = nCoastSize / 2;
147 
148  m_VCoast[nCoast].SetDetailedCurvature(nMaxConvexDetailedCoastPoint, STRAIGHT_COAST_MAX_DETAILED_CURVATURE);
149 
150  // And set the point of maximum smoothed convexity at the same point
151  m_VCoast[nCoast].SetSmoothCurvature(nMaxConvexSmoothedCoastPoint, STRAIGHT_COAST_MAX_SMOOTH_CURVATURE);
152  }

168 /*===============================================================================================================================
169 
170 Calculates signed Menger curvature (https://en.wikipedia.org/wiki/Menger_curvature#Definition) from three points on a line. Returns +ve values for concave, -ve for convex, and zero if the points are co-linear. Curvature is multiplied by 1000 to give easier-to-read numbers
171 
172 ===============================================================================================================================*/
173 double CSimulation::dCalcCurvature(int const nHandedness, CGeom2DPoint const* pPtBefore, CGeom2DPoint const* pPtThis, CGeom2DPoint const* pPtAfter)
174 {
175  double
176  dAreax4 = 2 * dTriangleAreax2(pPtBefore, pPtThis, pPtAfter),
177  dDist1 = dGetDistanceBetween(pPtBefore, pPtThis),
178  dDist2 = dGetDistanceBetween(pPtThis, pPtAfter),
179  dDist3 = dGetDistanceBetween(pPtBefore, pPtAfter),
180  dCurvature = dAreax4 / (dDist1 * dDist2 * dDist3);
181 
182  // Reverse if left-handed
183  int nShape = (nHandedness == LEFT_HANDED ? 1 : -1);
184 
185  return (dCurvature * nShape * 1000);
186 }
