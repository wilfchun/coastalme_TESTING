88 void CGeomMultiLine::InsertLineSegment(int const nSegment)
89 {
90 // assert(nSegment < m_prVVLineSegment.size());
91 
92  // The new vector of pairs is identical to the existing vector of pairs i.e. we inherit profile/line seg details from the previous line seg
93  vector<pair<int, int> > prVPrev = m_prVVLineSegment[nSegment];
94 
95  // Store the profile numbers that are in this existing vector of pairs, these are the profiles that will be affected by this insertion
96  vector<int> nVProfsAffected;
97  for (unsigned int i = 0; i < prVPrev.size(); i++)
98  nVProfsAffected.push_back(prVPrev[i].first);
99 
100  vector<vector<pair<int, int> > >::iterator it;
101  it = m_prVVLineSegment.begin();
102 
103  m_prVVLineSegment.insert(it+nSegment+1, prVPrev);
104 
105  // Must now increment the profile's own line seg numbers, but only for those profile numbers which were affected by the insertion. Do this for the new line seg and every line seg after that
106  for (unsigned int m = nSegment+1; m < m_prVVLineSegment.size(); m++)
107  {
108  for (unsigned int n = 0; n < m_prVVLineSegment[m].size(); n++)
109  {
110  for (unsigned int i = 0; i < nVProfsAffected.size(); i++)
111  {
112  if (m_prVVLineSegment[m][n].first == nVProfsAffected[i])
113  m_prVVLineSegment[m][n].second++;
114  }
115  }
116  }
117 }
