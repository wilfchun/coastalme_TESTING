67 int CSimulation::nSetAllCoastpointDeepWaterWaveValues(void)
68 {
69  // For each coastline, put a value for deep water wave height and direction at each coastline point
70  int
71  nNextProfile,
72  nDistFromPrevProfile = 0,
73  nDistToNextProfile = 0;
74 
75  double
76  dPrevProfileDeepWaterWaveHeight = 0,
77  dPrevProfileDeepWaterWaveAngle = 0,
78  dPrevProfileDeepWaterWavePeriod = 0,
79  dNextProfileDeepWaterWaveHeight = 0,
80  dNextProfileDeepWaterWaveAngle = 0,
81  dNextProfileDeepWaterWavePeriod = 0,
82  dDist = 0;
83 
84  for (int nCoast = 0; nCoast < static_cast<int>(m_VCoast.size()); nCoast++)
85  {
86  for (int nPoint = 0; nPoint < m_VCoast[nCoast].nGetCoastlineSize(); nPoint++)
87  {
88  // We are going down-coast
89  if (m_VCoast[nCoast].bIsProfileStartPoint(nPoint))
90  {
91  // OK, a coastline-normal profile begins at this coastline point, so set the deep water wave values at this coastline point to be the values at the seaward end of the coastline normal
92  int nProfile = m_VCoast[nCoast].nGetProfileNumber(nPoint);
93  CGeomProfile *pProfile = m_VCoast[nCoast].pGetProfile(nProfile);
94 
95  double
96  dThisDeepWaterWaveHeight = pProfile->dGetProfileDeepWaterWaveHeight(),
97  dThisDeepWaterWaveAngle = pProfile->dGetProfileDeepWaterWaveAngle(),
98  dThisDeepWaterWavePeriod = pProfile->dGetProfileDeepWaterWavePeriod();
99 
100  m_VCoast[nCoast].SetCoastDeepWaterWaveHeight(nPoint, dThisDeepWaterWaveHeight);
101  m_VCoast[nCoast].SetCoastDeepWaterWaveAngle(nPoint, dThisDeepWaterWaveAngle);
102  m_VCoast[nCoast].SetCoastDeepWaterWavePeriod(nPoint, dThisDeepWaterWavePeriod);
103 
104  // Reset for next time
105  nDistFromPrevProfile = 0;
106  dPrevProfileDeepWaterWaveHeight = dThisDeepWaterWaveHeight;
107  dPrevProfileDeepWaterWaveAngle = dThisDeepWaterWaveAngle;
108  dPrevProfileDeepWaterWavePeriod = dThisDeepWaterWavePeriod;
109 
110  // Find the next profile
111  nNextProfile = m_VCoast[nCoast].nGetDownCoastProfileNumber(nProfile);
112  if (nNextProfile == INT_NODATA)
113  {
114  // We are at the end of the coast
115  break;
116  }
117 
118  CGeomProfile *pNextProfile = m_VCoast[nCoast].pGetProfile(nNextProfile);
119 
120  // And the distance (in along-coast points) to the next profile
121  nDistToNextProfile = pNextProfile->nGetNumCoastPoint() - nPoint;
122  dDist = nDistToNextProfile;
123 
124  // And the next profile's deep water wave values
125  dNextProfileDeepWaterWaveHeight = pNextProfile->dGetProfileDeepWaterWaveHeight();
126  dNextProfileDeepWaterWaveAngle = pNextProfile->dGetProfileDeepWaterWaveAngle();
127  dNextProfileDeepWaterWavePeriod = pNextProfile->dGetProfileDeepWaterWavePeriod();
128 
129  // LogStream << m_ulIter << ": coast point = " << nPoint << " IS PROFILE START, dThisDeepWaterWaveHeight = " << dThisDeepWaterWaveHeight << ", dThisDeepWaterWaveAngle = " << dThisDeepWaterWaveAngle << endl;
130  }
131 
132  else
133  {
134  // This coast point is not the start of a coastline normal, so set the deep water wave values at this coastline point to be a weighted average of those from the up-coast and down-coast profiles
135  nDistFromPrevProfile++;
136  nDistToNextProfile--;
137 
138  double
139  dPrevWeight = (dDist - nDistFromPrevProfile) / dDist,
140  dNextWeight = (dDist - nDistToNextProfile) / dDist,
141  dThisDeepWaterWaveHeight = (dPrevWeight * dPrevProfileDeepWaterWaveHeight) + (dNextWeight * dNextProfileDeepWaterWaveHeight),
142  dThisDeepWaterWaveAngle = dKeepWithin360((dPrevWeight * dPrevProfileDeepWaterWaveAngle) + (dNextWeight * dNextProfileDeepWaterWaveAngle)),
143  dThisDeepWaterWavePeriod = (dPrevWeight * dPrevProfileDeepWaterWavePeriod) + (dNextWeight * dNextProfileDeepWaterWavePeriod);
144 
145  m_VCoast[nCoast].SetCoastDeepWaterWaveHeight(nPoint, dThisDeepWaterWaveHeight);
146  m_VCoast[nCoast].SetCoastDeepWaterWaveAngle(nPoint, dThisDeepWaterWaveAngle);
147  m_VCoast[nCoast].SetCoastDeepWaterWavePeriod(nPoint, dThisDeepWaterWavePeriod);
148 
149  // LogStream << m_ulIter << ": coast point = " << nPoint << " dThisDeepWaterWaveHeight = " << dThisDeepWaterWaveHeight << " dThisDeepWaterWaveAngle = " << dThisDeepWaterWaveAngle << endl;
150  }
151  }
152  }
153 
154  return RTN_OK;
155 }
