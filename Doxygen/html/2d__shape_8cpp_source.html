133 CGeom2DPoint CA2DShape::PtGetCentroid(void)
134 {
135  int nVertexCount = static_cast<int>(m_VPoints.size());
136  double
137  dSignedArea = 0,
138  dCentroidX = 0,
139  dCentroidY = 0;
140 
141  // For all vertices
142  for (int i = 0; i < nVertexCount; ++i)
143  {
144  double
145  dXThis = m_VPoints[i].dGetX(),
146  dYThis = m_VPoints[i].dGetY(),
147  dXNext = m_VPoints[(i+1) % nVertexCount].dGetX(),
148  dYNext = m_VPoints[(i+1) % nVertexCount].dGetY();
149 
150  double dA = (dXThis * dYNext) - (dXNext * dYThis);
151  dSignedArea += dA;
152 
153  dCentroidX += (dXThis + dXNext) * dA;
154  dCentroidY += (dYThis + dYNext) * dA;
155  }
156 
157  dSignedArea *= 0.5;
158  dCentroidX /= (6 * dSignedArea);
159  dCentroidY /= (6 * dSignedArea);
160 
161  return (CGeom2DPoint(dCentroidX, dCentroidY));
162 }
