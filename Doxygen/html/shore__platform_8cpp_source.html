46  Does platform erosion on all coastlines by first calculating platform erosion on coastline-normal profiles, then extrapolating this to cells between the profiles
47 
48 ===============================================================================================================================*/
49 int CSimulation::nDoAllShorePlatFormErosion(void)
50 {
51  // TODO Only do potential erosion if cell is in a polygon
52 
53  static bool bForward = true;
54 
55  // Do this for each coast
56  for (int nCoast = 0; nCoast < static_cast<int>(m_VCoast.size()); nCoast++)
57  {
58  int const nNumProfiles = m_VCoast[nCoast].nGetNumProfiles();
59 
60  // Calculate potential erosion on every coastline-normal profile. Can do this in the original, curvature-related, sequence of profiles
61  int nProfile = 0;
62  for ((bForward ? nProfile = 0 : nProfile = (nNumProfiles-1)); (bForward ? nProfile < nNumProfiles : nProfile >= 0); (bForward ? nProfile++ : nProfile--))
63  {
64  // Calculate potential platform erosion along the length of this profile
65  int const nRet = nCalcPotentialPlatformErosionOnProfile(nCoast, nProfile);
66  if (nRet != RTN_OK)
67  return nRet;
68  }
69 
70  // Calculate potential platform erosion between the coastline-normal profiles. Do this in along-coastline sequence
71  int nProfIndex = 0;
72  for ((bForward ? nProfIndex = 0 : nProfIndex = (nNumProfiles-1)); (bForward ? nProfIndex < nNumProfiles : nProfIndex >= 0); (bForward ? nProfIndex++ : nProfIndex--))
73  {
74  // Calculate potential erosion for sea cells between this profile and the next profile (or up to the edge of the grid) on these cells
75  int nRet = nCalcPotentialPlatformErosionBetweenProfiles(nCoast, nProfIndex, DIRECTION_DOWNCOAST);
76  if (nRet != RTN_OK)
77  return nRet;
78 
79  nRet = nCalcPotentialPlatformErosionBetweenProfiles(nCoast, nProfIndex, DIRECTION_UPCOAST);
80  if (nRet != RTN_OK)
81  return nRet;
82  }
83  }
84 
85  // If desired, swap direction for next timestep
86  if (m_bErodeShorePlatformAlternateDirection)
87  bForward = !bForward;
88 
89  // Fills in 'holes' in the potential platform erosion i.e. orphan cells which get omitted because of rounding problems
90  FillPotentialPlatformErosionHoles();
91 
92  // Do the same for beach protection
93  FillInBeachProtectionHoles();
94 
95  // Finally calculate actual platform erosion on all sea cells (both on profiles, and between profiles)
96  for (int nX = 0; nX < m_nXGridMax; nX++)
97  {
98  for (int nY = 0; nY < m_nYGridMax; nY++)
99  {
100  if (m_pRasterGrid->m_Cell[nX][nY].bPotentialPlatformErosion())
101  // Calculate actual (supply-limited) shore platform erosion on each cell that has potential platform erosion, also add the eroded sand/coarse sediment to that cells's polygon, ready to be redistributed within the polygon during beach erosion/deposition
102  DoActualShorePlatformErosionOnCell(nX, nY);
103  }
104  }
105 
106  if (m_nLogFileDetail >= LOG_FILE_MIDDLE_DETAIL)
107  {
108  LogStream << endl << m_ulIter << ": potential shore platform erosion (m^3) = " << m_dThisIterPotentialPlatformErosion * m_dCellArea << " (on profiles = " << m_dTotPotentialPlatformErosionOnProfiles * m_dCellArea << ", between profiles = " << m_dTotPotentialPlatformErosionBetweenProfiles * m_dCellArea << ")" << endl;
109 
110  LogStream << m_ulIter << ": actual shore platform erosion (m^3) = " << (m_dThisIterActualPlatformErosionFine + m_dThisIterActualPlatformErosionSand + m_dThisIterActualPlatformErosionCoarse) * m_dCellArea << " (fine = " << m_dThisIterActualPlatformErosionFine * m_dCellArea << ", sand = " << m_dThisIterActualPlatformErosionSand * m_dCellArea << ", coarse = " << m_dThisIterActualPlatformErosionCoarse * m_dCellArea << ")" << endl;
111  }
112 
113  return RTN_OK;
114 }

117 /*==============================================================================================================================
118 
119  Calculates potential (i.e. unconstrained by available sediment) erosional lowering of the shore platform for a single coastline-normal profile, due to wave action. This routine uses a behavioural rule to modify the original surface elevation profile geometry, in which erosion rate/slope = f(d/Db) based on Walkden & Hall (2005). Originally coded in Matlab, by Andres Payo
120 
121 ==============================================================================================================================*/
122 int CSimulation::nCalcPotentialPlatformErosionOnProfile(int const nCoast, int const nProfile)
123 {
124  CGeomProfile* const pProfile = m_VCoast[nCoast].pGetProfile(nProfile);
125 
126  // Only work on this profile if it is problem-free TODO ANDRES Or if it has just hit dry land?
127  if (! pProfile->bOKIncStartAndEndOfCoast()) // || (pProfile->nGetProblemCode() == PROFILE_DRYLAND))
128  return RTN_OK;
129 
130  // Get the length of the profile (in cells) and the index of the coast point at which this profile starts
131  int const
132  nProfSize = pProfile->nGetNumCellsInProfile(),
133  nCoastPoint = pProfile->nGetNumCoastPoint();
134 
135  // Get the breaking depth for this profile from the coastline point
136  double dDepthOfBreaking = m_VCoast[nCoast].dGetDepthOfBreaking(nCoastPoint);
137  if (dDepthOfBreaking == DBL_NODATA)
138  // This profile is not in the active zone, so no platform erosion here
139  return RTN_OK;
140 
141 // LogStream << m_ulIter << ": ON PROFILE nProfile = " << nProfile << " dDepthOfBreaking = " << dDepthOfBreaking << endl;
142 
143  // Get the height of the associated breaking wave from the coast point: this height is used in beach protection calcs
144  double const dBreakingWaveHeight = m_VCoast[nCoast].dGetBreakingWaveHeight(nCoastPoint);
145 
146  // Calculate the length of the profile in external CRS units
147  int const nSegments = pProfile->nGetProfileSize()-1;
148  double dProfileLenXY = 0;
149  for (int nSeg = 0; nSeg < nSegments; nSeg++)
150  {
151  // Do once for every line segment
152  double const
153  dSegStartX = pProfile->pPtGetPointInProfile(nSeg)->dGetX(),
154  dSegStartY = pProfile->pPtGetPointInProfile(nSeg)->dGetY(),
155  dSegEndX = pProfile->pPtGetPointInProfile(nSeg+1)->dGetX(), // Is OK
156  dSegEndY = pProfile->pPtGetPointInProfile(nSeg+1)->dGetY();
157 
158  double const dSegLen = hypot(dSegStartX - dSegEndX, dSegStartY - dSegEndY);
159  dProfileLenXY += dSegLen;
160  }
161 
162  // Next calculate the average distance between profile points, again in external CRS units. Assume that the sample points are equally spaced along the profile (not quite true)
163  double const dSpacingXY = dProfileLenXY / (nProfSize-1);
164 
165  // Set up vectors for the coastline-normal profile elevations. The length of this vector line is given by the number of cells 'under' the profile. Thus each point on the vector relates to a single cell in the grid. This assumes that all points on the profile vector are equally spaced (not quite true, depends on the orientation of the line segments which comprise the profile)
166  // The elevation of each of these profile points is the elevation of the centroid of the cell that is 'under' the point. However we cannot always be confident that this is the 'true' elevation of the point on the vector since (unless the profile runs planview N-S or W-E) the vector does not always run exactly through the centroid of the cell
167  vector<double>
168  VdProfileZ(nProfSize, 0), // Initial (pre-erosion) elevation of both consolidated and unconsolidated sediment for cells 'under' the profile
169  VdProfileDistXY(nProfSize, 0), // Along-profile distance measured from the coast, in external CRS units
170  dVConsProfileZ(nProfSize, 0), // Initial (pre-erosion) elevation of consolidated sediment only for cells 'under' the profile
171  dVConsZDiff(nProfSize, 0),
172  dVConsSlope(nProfSize, 0);
173 
174  for (int i = 0; i < nProfSize; i++)
175  {
176  int const
177  nX = pProfile->pPtiVGetCellsInProfile()->at(i).nGetX(),
178  nY = pProfile->pPtiVGetCellsInProfile()->at(i).nGetY();
179 
180  // Get the number of the highest layer with non-zero thickness
181  int const nTopLayer = m_pRasterGrid->m_Cell[nX][nY].nGetTopNonZeroLayerAboveBasement();
182 
183  // Safety check
184  if (nTopLayer == INT_NODATA)
185  return RTN_ERR_NO_TOP_LAYER;
186 
187  if (nTopLayer == NO_NONZERO_THICKNESS_LAYERS)
188  // TODO we are down to basement, check with Andres what to do
189  return RTN_OK;
190 
191  // Get the elevation for consolidated sediment only on this cell
192  dVConsProfileZ[i] = m_pRasterGrid->m_Cell[nX][nY].dGetConsSedTopForLayerAboveBasement(nTopLayer);
193 
194  // Get the elevation for both consolidated and unconsolidated sediment on this cell
195  VdProfileZ[i] = m_pRasterGrid->m_Cell[nX][nY].dGetSedimentTopElev();
196 
197  // And store the X-Y plane distance from the start of the profile
198  VdProfileDistXY[i] = i * dSpacingXY;
199  }
200 
201  for (int i = 0; i < nProfSize-1; i++)
202  {
203  // For the consolidated-only profile, get the Z differences (already in external CRS units)
204  dVConsZDiff[i] = dVConsProfileZ[i] - dVConsProfileZ[i+1];
205 
206  // Calculate dZ/dXY, the Z slope (i.e. rise over run) in the XY direction. Note that we use the elevation difference on the seaward side of 'this' point
207  dVConsSlope[i] = dVConsZDiff[i] / dSpacingXY;
208  }
209 
210  // Sort out the final value
211  dVConsSlope[nProfSize-1] = dVConsSlope[nProfSize-2];
212 
213  if (m_nProfileSmoothWindow > 0)
214  {
215  // Smooth the vector of slopes for the consolidated-only profile
216  dVConsSlope = dVSmoothProfileSlope(&dVConsSlope);
217  }
218 
219  vector<double>
220  dVProfileDepthOverDB(nProfSize, 0), // Depth over wave breaking depth at the coastline-normal sample points
221  dVProfileErosionPotential(nProfSize, 0); // Erosion potential at the coastline-normal sample points
222 
223  // Calculate the erosion potential along this profile using the shape function
224  double dTotalErosionPotential = 0;
225  for (int i = 0; i < nProfSize; i++)
226  {
227  // Use the actual depth of water here (i.e. the depth to the top of the unconsolidated sediment, including the thickness of consolidated sediment beneath it)
228  dVProfileDepthOverDB[i] = m_dThisIterSWL - VdProfileZ[i];
229  dVProfileDepthOverDB[i] /= dDepthOfBreaking;
230 
231  // Constrain dDepthOverDB[i] to be between 0 (can get small -ve values due to rounding errors) and m_dDepthOverDBMax
232  dVProfileDepthOverDB[i] = tMax(dVProfileDepthOverDB[i], 0.0);
233  dVProfileDepthOverDB[i] = tMin(dVProfileDepthOverDB[i], m_dDepthOverDBMax);
234 
235  // And then use the look-up table to find the value of erosion potential at this point on the profile
236  dVProfileErosionPotential[i] = dLookUpErosionPotential(dVProfileDepthOverDB[i]);
237 
238  // If erosion potential (a -ve value) is tiny, set it to zero
239  if (dVProfileErosionPotential[i] > -SEDIMENT_ELEV_TOLERANCE)
240  dVProfileErosionPotential[i] = 0;
241 
242  // Keep track of the total erosion potential for this profile
243  dTotalErosionPotential += dVProfileErosionPotential[i];
244  }
245 
246  // Constrain erosion potential at every point on the profile, so that the integral of erosion potential on the whole profile is unity (Walkden and Hall 2005). Note that here, erosion potential is -ve so we must constrain to -1
247  for (int i = 0; i < nProfSize; i++)
248  {
249  if (dTotalErosionPotential < 0)
250  dVProfileErosionPotential[i] /= (-dTotalErosionPotential);
251  }
252 
253  vector<double>
254  dVRecessionXY(nProfSize, 0),
255  dVSCAPEXY(nProfSize, 0);
256 
257  // Calculate recession at every point on the coastline-normal profile
258  for (int i = 0; i < nProfSize; i++)
259  {
260  // dRecession = dForce * (dBeachProtection / dR) * dErosionPotential * dSlope * dTime
261  // where:
262  // dVRecession [m] is the landward migration distance defined in the profile relative (XY) CRS
263  // dForce is given by Equation 4 in Walkden & Hall, 2005
264  // dVBeachProtection [1] is beach protection factor [1, 0] = [no protection, fully protected]. (This is calculated later, see dCalcBeachProtectionFactor())
265  // dVR [m^(9/4)s^(2/3)] is the material strength and some hydrodynamic constant
266  // dVProfileErosionPotential [?] is the erosion potential at each point along the profile
267  // dVSlope [1] is the along-profile slope
268  // m_dTimeStep * 3600 [s] is the time interval in seconds
269  //
270  // dRecession is horizontal recession (along the XY direction):
271  //
272  // dVRecessionXY[i] = (dForce * dVBeachProtection[i] * dVErosionPotentialFunc[i] * dVSlope[i] * m_dTimeStep * 3600) / dVR[i]
273  //
274  // XY recession must be -ve or zero. If it is +ve then it represents accretion not erosion, which must be described by a different set of equations. So we also need to constrain XY recession to be <= 0
275  dVRecessionXY[i] = tMin(m_VCoast[nCoast].dGetWaveEnergyatBreaking(nCoastPoint) * dVProfileErosionPotential[i] * dVConsSlope[i] / m_dR, 0.0);
276  dVSCAPEXY[i] = VdProfileDistXY[i] - dVRecessionXY[i];
277  }
278 
279  vector<double> dVChangeElevZ(nProfSize, 0);
280 
281  // We have calculated the XY-plane recession at every point on the profile, so now convert this to a change in Z-plane elevation at every inundated point on the profile (not the coast point). Again we use the elevation difference on the seaward side of 'this' point
282  for (int i = 1; i < nProfSize-1; i++)
283  {
284  // Vertical lowering dZ = dXY * tan(a), where tan(a) is the slope of the SCAPE profile in the XY direction
285  double
286  dSCAPEHorizDist = dVSCAPEXY[i+1] - dVSCAPEXY[i],
287  dSCAPEVertDist = dVConsProfileZ[i] - dVConsProfileZ[i+1],
288  dSCAPESlope = dSCAPEVertDist / dSCAPEHorizDist,
289  dDeltaZ = dVRecessionXY[i] * dSCAPESlope;
290 
291  int const
292  nX = pProfile->pPtiVGetCellsInProfile()->at(i).nGetX(),
293  nY = pProfile->pPtiVGetCellsInProfile()->at(i).nGetY();
294 
295  // Store the local slope of the consolidated sediment, this is just for output display purposes
296  m_pRasterGrid->m_Cell[nX][nY].SetLocalConsSlope(dVConsSlope[i]);
297 
298  // dDeltaZ is zero or -ve: if dDeltaZ is zero then do nothing, if -ve then remove some sediment from this cell
299  if (dDeltaZ < 0)
300  {
301  // If there has already been potential erosion on this cell, then it must be a shared line segment (i.e. has co-incident profiles)
302  double dPrevPotentialErosion = -m_pRasterGrid->m_Cell[nX][nY].dGetPotentialPlatformErosion();
303  if (dPrevPotentialErosion < 0)
304  {
305  // Average the two values
306 // LogStream << m_ulIter << ": [" << nX << "][" << nY << "] under profile " << nProfile << " has previous potential platform erosion = " << dPrevPotentialErosion << endl;
307  dDeltaZ = ((dDeltaZ + dPrevPotentialErosion) / 2);
308  }
309 
310  // Constrain the lowering so we don't get negative slopes or +ve erosion amounts (dDeltaZ must be -ve), this is implicit in SCAPE
311  dDeltaZ = tMax(dDeltaZ, -dVConsZDiff[i]);
312  dDeltaZ = tMin(dDeltaZ, 0.0);
313  dVChangeElevZ[i] = dDeltaZ;
314 
315  // Set the potential (unconstrained) erosion for this cell, is a +ve value
316  m_pRasterGrid->m_Cell[nX][nY].SetPotentialPlatformErosion(-dDeltaZ);
317 
318  // Update this-timestep totals
319  m_ulThisIterNumPotentialPlatformErosionCells++;
320  m_dThisIterPotentialPlatformErosion -= dDeltaZ; // Since dDeltaZ is a -ve value
321 // assert(isfinite(m_dThisIterPotentialPlatformErosion));
322 // assert(m_dThisIterPotentialPlatformErosion >= 0);
323 
324  // Increment the check values
325  m_ulTotPotentialPlatformErosionOnProfiles++;
326  m_dTotPotentialPlatformErosionOnProfiles -= dDeltaZ;
327  }
328 
329  // Finally, calculate the beach protection factor, this will be used in estimating actual (supply-limited) erosion
330  double dBeachProtectionFactor = dCalcBeachProtectionFactor(nX, nY, dBreakingWaveHeight);
331  m_pRasterGrid->m_Cell[nX][nY].SetBeachProtectionFactor(dBeachProtectionFactor);
332  }
333 
334  // If desired, save this coastline-normal profile data for checking purposes
335  if (m_bOutputProfileData)
336  {
337  int nRet = nSaveProfile(nProfile, nCoast, nProfSize, &VdProfileDistXY, &dVConsProfileZ, &dVProfileDepthOverDB, &dVProfileErosionPotential, &dVConsSlope, &dVRecessionXY, &dVChangeElevZ, pProfile->pPtiVGetCellsInProfile(), &dVSCAPEXY);
338  if (nRet != RTN_OK)
339  return nRet;
340  }
341 
342  return RTN_OK;
343 }

433  // LogStream << m_ulIter << ": coast " << nCoast << ", LEAVING LOOP since hit another profile at nThisPointOnCoast = " << nThisPointOnCoast << " while doing potential platform erosion " << (nDirection == DIRECTION_DOWNCOAST ? 453  nParCoastXLast = nParCoastX;
454  nParCoastYLast = nParCoastY;
455 // LogStream << m_ulIter << ": parallel profile abandoned since too short, starts at [" << nParCoastX << "][" << nParCoastY << "] coastline point " << nThisPointOnCoast << ", length = " << nParProfSize << endl;
456  continue;
457  }
458 
459  // This parallel profile is OK, so calculate potential erosion along it. First calculate the length of the parallel profile in external CRS units
460  double const dParProfileLenXY = dGetDistanceBetween(&PtVExtCRSParProfile[0], &PtVExtCRSParProfile[nParProfSize-1]);
461 
462  // Next calculate the distance between profile points, again in external CRS units. Assume that the sample points are equally spaced along the parallel profile (not quite true)
463  double const dParSpacingXY = dParProfileLenXY / (nParProfSize - 1);
464 // LogStream << "dParSpacingXY = " << dParSpacingXY << endl;
465 
466  vector<double>
467  dVParProfileZ(nParProfSize, 0), // Initial (pre-erosion) elevation of both consolidated and unconsolidated sediment for cells 'under' the parallel profile
468  dVParProfileDistXY(nParProfSize, 0), // Along-profile distance measured from the coast, in external CRS units
469  dVParConsProfileZ(nParProfSize, 0), // Initial (pre-erosion) elevation of consolidated sediment only for cells 'under' the parallel profile
470  dVParConsZDiff(nParProfSize, 0),
471  dVParConsSlope(nParProfSize, 0);
472 
473  for (int i = 0; i < nParProfSize; i++)
474  {
475  int const
476  nXPar = PtiVGridParProfile[i].nGetX(),
477  nYPar = PtiVGridParProfile[i].nGetY();
478 
479  // Is this a sea cell?
480  if (! m_pRasterGrid->m_Cell[nXPar][nYPar].bIsInundated())
481  {
482  // It isn't so move along, nothing to do here
483 // LogStream << m_ulIter << " : [" << nXPar << "][" << nYPar << "] is not inundated" << endl;
484  continue;
485  }
486 
487  // Is this cell in a polygon?
488  int nPolyID = m_pRasterGrid->m_Cell[nXPar][nYPar].nGetPolygonID();
489  if (nPolyID == INT_NODATA)
490  {
491  // It isn't. This can happen at the seaward end of polygons TODO Is it a problem?
492 // LogStream << m_ulIter << " : [" << nXPar << "][" << nYPar << "] = {" << dGridCentroidXToExtCRSX(nXPar) << ", " << dGridCentroidYToExtCRSY(nYPar) << "} is not in a polygon" << endl;
493  continue;
494  }
495 
496  // Get the number of the highest layer with non-zero thickness
497  int const nTopLayer = m_pRasterGrid->m_Cell[nXPar][nYPar].nGetTopNonZeroLayerAboveBasement();
498 
499  // Safety check
500  if (nTopLayer == INT_NODATA)
501  return RTN_ERR_NO_TOP_LAYER;
502 
503  if (nTopLayer == NO_NONZERO_THICKNESS_LAYERS)
504  // TODO we are down to basement, check with Andres what to do
505  return RTN_OK;
506 
507  // Get the elevation for consolidated sediment only on this cell
508  dVParConsProfileZ[i] = m_pRasterGrid->m_Cell[nXPar][nYPar].dGetConsSedTopForLayerAboveBasement(nTopLayer);
509 
510  // Get the elevation for both consolidated and unconsolidated sediment on this cell
511  dVParProfileZ[i] = m_pRasterGrid->m_Cell[nXPar][nYPar].dGetSedimentTopElev();
512 
513  // And store the X-Y plane distance from the start of the profile
514  dVParProfileDistXY[i] = i * dParSpacingXY;
515  }
516 
517  for (int i = 0; i < nParProfSize-1; i++)
518  {
519  // For the consolidated-only profile, get the Z differences (already in external CRS units)
520  dVParConsZDiff[i] = dVParConsProfileZ[i] - dVParConsProfileZ[i+1];
521 
522  // Calculate dZ/dXY, the Z slope (i.e. rise over run) in the XY direction. Note that we use the elevation difference on the seaward side of 'this' point
523  dVParConsSlope[i] = dVParConsZDiff[i] / dParSpacingXY;
524  }
525 
526  // Sort out the final slope value
527  dVParConsSlope[nParProfSize-1] = dVParConsSlope[nParProfSize-2];
528 
529  if (m_nProfileSmoothWindow > 0)
530  {
531  // Smooth the vector of slopes for the consolidated-only profile
532  dVParConsSlope = dVSmoothProfileSlope(&dVParConsSlope);
533  }
534 
535  // Initialize the parallel profile vector with depth / m_dWaveBreakingDepth
536  vector<double>
537  dVParProfileDepthOverDB(nParProfSize, 0), // Depth / wave breaking depth at the parallel profile sample points
538  dVParProfileErosionPotential(nParProfSize, 0); // Erosion potential at the parallel profile sample points
539 
540  // Calculate the erosion potential along this profile using the shape function which we read in earlier
541  double dTotalErosionPotential = 0;
542 
543  for (int i = 0; i < nParProfSize; i++)
544  {
545  // Use the actual depth of water here (i.e. the depth to the top of the unconsolidated sediment, including the thickness of consolidated sediment beneath it)
546  dVParProfileDepthOverDB[i] = m_dThisIterSWL - dVParProfileZ[i];
547  dVParProfileDepthOverDB[i] /= dDepthOfBreaking;
548 
549  // Constrain dDepthOverDB[i] to be between 0 (can get small -ve due to rounding errors) and m_dDepthOverDBMax
550  dVParProfileDepthOverDB[i] = tMax(dVParProfileDepthOverDB[i], 0.0);
551  dVParProfileDepthOverDB[i] = tMin(dVParProfileDepthOverDB[i], m_dDepthOverDBMax);
552 
553  // And then use the look-up table to find the value of erosion potential at this point on the profile
554  dVParProfileErosionPotential[i] = dLookUpErosionPotential(dVParProfileDepthOverDB[i]);
555 
556  // If erosion potential (a -ve value) is tiny, set it to zero
557  if (dVParProfileErosionPotential[i] > -SEDIMENT_ELEV_TOLERANCE)
558  dVParProfileErosionPotential[i] = 0;
559 
560  // Keep track of the total erosion potential for this profile
561  dTotalErosionPotential += dVParProfileErosionPotential[i];
562  }
563 
564  // Constrain erosion potential at every point on the profile, so that the integral of erosion potential on the whole profile is unity (Walkden and Hall 2005). Note that here, erosion potential is -ve so we must constrain to -1
565  for (int i = 0; i < nParProfSize; i++)
566  {
567  if (dTotalErosionPotential < 0)
568  dVParProfileErosionPotential[i] /= (-dTotalErosionPotential);
569  }
570 
571  vector<double>
572  dVParRecessionXY(nParProfSize, 0),
573  dVParSCAPEXY(nParProfSize, 0);
574 
575  // Calculate recession at every point on the parallel profile
576  for (int i = 0; i < nParProfSize; i++)
577  {
578  // dRecession = dForce * (dBeachProtection / dR) * dErosionPotential * dSlope * dTime
579  // where:
580  // dVRecession [m] is the landward migration distance defined in the profile relative (XY) CRS
581  // dForce is given by Equation 4 in Walkden & Hall, 2005
582  // dVBeachProtection [1] is beach protection factor [1, 0] = [no protection, fully protected] (This is calculated later, see dCalcBeachProtectionFactor())
583  // dVR [m^(9/4)s^(2/3)] is the material strength and some hydrodynamic constant
584  // dVProfileErosionPotential [?] is the erosion potential at each point along the profile
585  // dVSlope [1] is the along-profile slope
586  // m_dTimeStep * 3600 [s] is the time interval in seconds
587  //
588  // dRecession is horizontal recession (along the XY direction):
589  //
590  // dVRecessionXY[i] = (dForce * dVBeachProtection[i] * dVErosionPotentialFunc[i] * dVSlope[i] * m_dTimeStep * 3600) / dVR[i]
591  //
592  //
593  // XY recession must be -ve or zero. If it is +ve then it represents accretion not erosion, which must be described by a different set of equations. So we also need to constrain XY recession to be <= 0
594  dVParRecessionXY[i] = tMin(m_VCoast[nCoast].dGetWaveEnergyatBreaking(nThisPointOnCoast) * dVParProfileErosionPotential[i] * dVParConsSlope[i] / m_dR, 0.0);
595  dVParSCAPEXY[i] = dVParProfileDistXY[i] - dVParRecessionXY[i];
596 
597 // LogStream << m_ulIter << ": [" << nXPar << "][" << nYPar << "] = {" << dGridCentroidXToExtCRSX(nXPar) << ", " << dGridCentroidYToExtCRSY(nYPar) << "} wave energy = " << m_VCoast[nCoast].dGetWaveEnergyatBreaking(nThisPointOnCoast) << " erosion potential = " << dVParProfileErosionPotential[i] << " slope = " << dVParProfileSlope[i] << " dVParZDiff[i] = " << dVParZDiff[i] << " nParProfSize = " << nParProfSize << endl;
598  }
599 
600  vector<double> dVParDeltaZ(nParProfSize, 0);
601 
602  // We have calculated the XY-plane recession at every point on the profile, so now convert this to a change in Z-plane elevation at every inundated point on the profile (not the coast point). Again we use the elevation difference on the seaward side of 'this' point
603  for (int i = 1; i < nParProfSize-1; i++)
604  {
605  // Vertical lowering dZ = dXY * tan(a), where tan(a) is the slope of the SCAPE profile in the XY direction
606  double
607  dSCAPEHorizDist = dVParSCAPEXY[i+1] - dVParSCAPEXY[i],
608  dSCAPEVertDist = dVParConsProfileZ[i] - dVParConsProfileZ[i+1],
609  dSCAPESlope = dSCAPEVertDist / dSCAPEHorizDist,
610  dDeltaZ = dVParRecessionXY[i] * dSCAPESlope;
611 
612  int const
613  nXPar = PtiVGridParProfile[i].nGetX(),
614  nYPar = PtiVGridParProfile[i].nGetY();
615 
616  // Store the local slope of the consolidated sediment, this is just for output display purposes
617  m_pRasterGrid->m_Cell[nXPar][nYPar].SetLocalConsSlope(dVParConsSlope[i]);
618 
619  // dDeltaZ is zero or -ve: if dDeltaZ is zero then do nothing, if -ve then remove some sediment from this cell
620  if (dDeltaZ < 0)
621  {
622  // Has this cell already been eroded during this timestep?
623  if (m_pRasterGrid->m_Cell[nXPar][nYPar].bPotentialPlatformErosion())
624  {
625  // It has
626  double const dPrevPotentialErosion = -m_pRasterGrid->m_Cell[nXPar][nYPar].dGetPotentialPlatformErosion();
627 
628 // LogStream << m_ulIter << ": [" << nXPar << "][" << nYPar << "] parallel profile " << nDistFromProfile << " coast points " << (nDirection == DIRECTION_DOWNCOAST ? "down" : "up") << "-coast from profile " << nProfile << " has previous potential platform erosion = " << dPrevPotentialErosion << ", current potential platform erosion = " << dDeltaZ << ", max value = " << tMin(dPrevPotentialErosion, dDeltaZ) << endl;
629 
630  // Use the larger of the two -ve values
631  dDeltaZ = tMin(dPrevPotentialErosion, dDeltaZ);
632 
633  // Adjust this-timestep totals, since this cell has already been eroded
634  m_ulThisIterNumPotentialPlatformErosionCells--;
635  m_dThisIterPotentialPlatformErosion += dPrevPotentialErosion; // Since dPrevPotentialErosion is +ve
636 // assert(isfinite(m_dThisIterPotentialPlatformErosion));
637 // assert(m_dThisIterPotentialPlatformErosion >= 0);
638 
639  // And also adjust the check values
640  m_ulTotPotentialPlatformErosionBetweenProfiles--;
641  m_dTotPotentialPlatformErosionBetweenProfiles += dPrevPotentialErosion; // Since -ve
642  }
643 
644  // Constrain the lowering so we don't get negative slopes or +ve erosion amounts (dDeltaZ must be -ve), this is implicit in SCAPE
645  dDeltaZ = tMax(dDeltaZ, -dVParConsZDiff[i]);
646  dDeltaZ = tMin(dDeltaZ, 0.0);
647  dVParDeltaZ[i] = dDeltaZ;
648 
649  // Set the potential (unconstrained) erosion for this cell, it is a +ve value
650  m_pRasterGrid->m_Cell[nXPar][nYPar].SetPotentialPlatformErosion(-dDeltaZ);
651 // LogStream << "[" << nXPar << "][" << nYPar << "] = {" << dGridCentroidXToExtCRSX(nXPar) << ", " << dGridCentroidYToExtCRSY(nYPar) << "} has potential platform erosion = " << -dDeltaZ << endl;
652 
653  // Update this-timestep totals
654  m_ulThisIterNumPotentialPlatformErosionCells++;
655  m_dThisIterPotentialPlatformErosion -= dDeltaZ; // Since dDeltaZ is a -ve value
656 // assert(isfinite(m_dThisIterPotentialPlatformErosion));
657 // assert(m_dThisIterPotentialPlatformErosion >= 0);
658 
659  // Increment the check values
660  m_ulTotPotentialPlatformErosionBetweenProfiles++;
661  m_dTotPotentialPlatformErosionBetweenProfiles -= dDeltaZ; // Since -ve
662  }
663 
664  // Finally, calculate the beach protection factor, this will be used in estimating actual (supply-limited) erosion
665  double const dBeachProtectionFactor = dCalcBeachProtectionFactor(nXPar, nYPar, dBreakingWaveHeight);
666  m_pRasterGrid->m_Cell[nXPar][nYPar].SetBeachProtectionFactor(dBeachProtectionFactor);
667  }
668 
669  // If desired, save this parallel coastline-normal profile for checking purposes
670  if (m_bOutputParallelProfileData)
671  {
672  int const nRet = nSaveParProfile(nCoast, nProfile, nParProfSize, nDirection, nDistFromProfile, &dVParProfileDistXY, &dVParConsProfileZ, &dVParProfileDepthOverDB, &dVParProfileErosionPotential, &dVParConsSlope, &dVParRecessionXY, &dVParDeltaZ, pProfile->pPtiVGetCellsInProfile(), &dVParSCAPEXY);
673  if (nRet != RTN_OK)
674  return nRet;
675  }
676 
677  // Update for next time round the loop
678  nParCoastXLast = nParCoastX;
679  nParCoastYLast = nParCoastY;
680  }
681 
682  return RTN_OK;
683 }

686 /*==============================================================================================================================
687 
688  Calculates actual (constrained by available sediment) erosion of the consolidated shore platform on a single sea cell
689 
690 ==============================================================================================================================*/
691 void CSimulation::DoActualShorePlatformErosionOnCell(int const nX, int const nY)
692 {
693 // LogStream << m_ulIter << ": doing platform erosion on cell [" << nX << "][" << nY << "] = {" << dGridCentroidXToExtCRSX(nX) << ", " << dGridCentroidYToExtCRSY(nY) << "}" << endl;
694 
695  // Get the beach protection factor, which quantifies the extent to which unconsolidated sediment on the shore platform (beach) protects the shore platform
696  double const dBeachProtectionFactor = m_pRasterGrid->m_Cell[nX][nY].dGetBeachProtectionFactor();
697  if (dBeachProtectionFactor == 0)
698  // The beach is sufficiently thick to prevent any platform erosion (or we are down to basement)
699  return;
700 
701  // Get the potential depth of potential erosion, considering beach protection
702  double dThisPotentialErosion = m_pRasterGrid->m_Cell[nX][nY].dGetPotentialPlatformErosion() * dBeachProtectionFactor;
703 
704  // We will be eroding the topmost layer that has non-zero thickness
705  int nThisLayer = m_pRasterGrid->m_Cell[nX][nY].nGetTopNonZeroLayerAboveBasement();
706 
707  // Safety check
708  if (nThisLayer == INT_NODATA)
709  {
710  cerr << ERR << "no sediment layer in DoActualShorePlatformErosionOnCell()" << endl;
711  return;
712  }
713 
714  if (nThisLayer == NO_NONZERO_THICKNESS_LAYERS)
715  // No layer with non-zero thickness left, we are down to basement
716  return;
717 
718  // OK, we have a layer that can be eroded so find out how much consolidated sediment we have available on this cell
719  double
720  dExistingAvailableFine = m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(nThisLayer)->pGetConsolidatedSediment()->dGetFine(),
721  dExistingAvailableSand = m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(nThisLayer)->pGetConsolidatedSediment()->dGetSand(),
722  dExistingAvailableCoarse = m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(nThisLayer)->pGetConsolidatedSediment()->dGetCoarse();
723 
724  // Now partition the total lowering for this cell between the three size fractions: do this by relative erodibility
725  int
726  nFineWeight = (dExistingAvailableFine > 0 ? 1 : 0),
727  nSandWeight = (dExistingAvailableSand > 0 ? 1 : 0),
728  nCoarseWeight = (dExistingAvailableCoarse > 0 ? 1 : 0);
729 
730  double
731  dTotErodibility = (nFineWeight * m_dFineErodibilityNormalized) + (nSandWeight * m_dSandErodibilityNormalized) + (nCoarseWeight * m_dCoarseErodibilityNormalized),
732  dTotActualErosion = 0,
733  dSandEroded = 0,
734  dCoarseEroded = 0;
735 
736  if (nFineWeight)
737  {
738  // Erode some fine-sized consolidated sediment
739  double dFineLowering = (m_dFineErodibilityNormalized * dThisPotentialErosion) / dTotErodibility;
740 
741  // Make sure we don't get -ve amounts left on the cell
742  double dFineEroded = tMin(dExistingAvailableFine, dFineLowering);
743  double dRemaining = dExistingAvailableFine - dFineEroded;
744 
745  dTotActualErosion += dFineEroded;
746 
747  // Set the value for this layer
748  m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(nThisLayer)->pGetConsolidatedSediment()->SetFine(dRemaining);
749 
750  // And set the changed-this-timestep switch
751  m_bConsChangedThisIter[nThisLayer] = true;
752 
753  // And increment the per-timestep total
754  m_dThisIterActualPlatformErosionFine += dFineEroded;
755  }
756 
757  if (nSandWeight)
758  {
759  // Erode some sand-sized consolidated sediment
760  double dSandLowering = (m_dSandErodibilityNormalized * dThisPotentialErosion) / dTotErodibility;
761 
762  // Make sure we don't get -ve amounts left on the source cell
763  dSandEroded = tMin(dExistingAvailableSand, dSandLowering);
764  double dRemaining = dExistingAvailableSand - dSandEroded;
765 
766  dTotActualErosion += dSandEroded;
767 
768  // Set the value for this layer
769  m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(nThisLayer)->pGetConsolidatedSediment()->SetSand(dRemaining);
770 
771  // Set the changed-this-timestep switch
772  m_bConsChangedThisIter[nThisLayer] = true;
773 
774  // And increment the per-timestep total
775  m_dThisIterActualPlatformErosionSand += dSandEroded;
776  }
777 
778  if (nCoarseWeight)
779  {
780  // Erode some coarse-sized consolidated sediment
781  double dCoarseLowering = (m_dCoarseErodibilityNormalized * dThisPotentialErosion) / dTotErodibility;
782 
783  // Make sure we don't get -ve amounts left on the source cell
784  dCoarseEroded = tMin(dExistingAvailableCoarse, dCoarseLowering);
785  double dRemaining = dExistingAvailableCoarse - dCoarseEroded;
786 
787  dTotActualErosion += dCoarseEroded;
788 
789  // Set the value for this layer
790  m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(nThisLayer)->pGetConsolidatedSediment()->SetCoarse(dRemaining);
791 
792  // Set the changed-this-timestep switch
793  m_bConsChangedThisIter[nThisLayer] = true;
794 
795  // And increment the per-timestep total
796  m_dThisIterActualPlatformErosionCoarse += dCoarseEroded;
797  }
798 
799  // Did we erode anything?
800  if (dTotActualErosion > 0)
801  {
802  // We did, so set the actual erosion value for this cell
803  m_pRasterGrid->m_Cell[nX][nY].SetActualPlatformErosion(dTotActualErosion);
804 
805  // Recalculate the elevation of every layer
806  m_pRasterGrid->m_Cell[nX][nY].CalcAllLayerElevsAndD50();
807 
808  // And update the cell's sea depth
809  m_pRasterGrid->m_Cell[nX][nY].SetSeaDepth();
810 
811  // Update per-timestep totals
812  m_ulThisIterNumActualPlatformErosionCells++;
813 
814  // Eroded fine sediment goes onto suspension
815 
816  // Add eroded sand/coarse sediment for this cell to the polygon that contains the cell, ready for redistribution during beach erosion/deposition
817  int nPolyID = m_pRasterGrid->m_Cell[nX][nY].nGetPolygonID();
818  if (nPolyID == INT_NODATA)
819  {
820  // Can get occasional problems with polygon rasterization near the coastline, so also search the eight adjacent cells
821  int nDirection[] = {NORTH, NORTH_EAST, EAST, SOUTH_EAST, SOUTH, SOUTH_WEST, WEST, NORTH_WEST};
822  Rand1Shuffle(nDirection, 8);
823 
824  for (int n = 0; n < 8; n++)
825  {
826  if (nDirection[n] == NORTH)
827  {
828  int
829  nXAdj = nX,
830  nYAdj = nY-1;
831  if (nYAdj >= 0)
832  {
833  nPolyID = m_pRasterGrid->m_Cell[nXAdj][nYAdj].nGetPolygonID();
834  if (nPolyID != INT_NODATA)
835  break;
836  }
837  }
838  else if (nDirection[n] == NORTH_EAST)
839  {
840  int
841  nXAdj = nX+1,
842  nYAdj = nY-1;
843  if ((nXAdj < m_nXGridMax) && (nYAdj >= 0))
844  {
845  nPolyID = m_pRasterGrid->m_Cell[nXAdj][nYAdj].nGetPolygonID();
846  if (nPolyID != INT_NODATA)
847  break;
848  }
849  }
850  else if (nDirection[n] == EAST)
851  {
852  int
853  nXAdj = nX+1,
854  nYAdj = nY;
855  if (nXAdj < m_nXGridMax)
856  {
857  nPolyID = m_pRasterGrid->m_Cell[nXAdj][nYAdj].nGetPolygonID();
858  if (nPolyID != INT_NODATA)
859  break;
860  }
861  }
862  else if (nDirection[n] == SOUTH_EAST)
863  {
864  int
865  nXAdj = nX+1,
866  nYAdj = nY+1;
867  if ((nXAdj < m_nXGridMax) && (nYAdj < m_nYGridMax))

871  break;

875  {
876  int
877  nXAdj = nX,
878  nYAdj = nY+1;
879  if (nYAdj < m_nYGridMax)
880  {
881  nPolyID = m_pRasterGrid->m_Cell[nXAdj][nYAdj].nGetPolygonID();

887  {
888  int
889  nXAdj = nX-1,
890  nYAdj = nY+1;
891  if ((nXAdj >= 0) && (nXAdj < m_nXGridMax))
892  {
893  nPolyID = m_pRasterGrid->m_Cell[nXAdj][nYAdj].nGetPolygonID();
894  if (nPolyID != INT_NODATA)
895  break;
896  }
897  }

899  {
900  int
901  nXAdj = nX-1,
902  nYAdj = nY;
903  if (nXAdj >= 0)
904  {
905  nPolyID = m_pRasterGrid->m_Cell[nXAdj][nYAdj].nGetPolygonID();
906  if (nPolyID != INT_NODATA)
907  break;

912  int
913  nXAdj = nX-1,

915  if ((nXAdj >= 0) && (nYAdj >= 0))

918  if (nPolyID != INT_NODATA)
919  break;

921  }
922  }
923  }
924 
925  if (nPolyID != INT_NODATA)
926  {
