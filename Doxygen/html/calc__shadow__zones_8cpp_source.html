50  Determines whether the wave orientation at this point on a coast is onshore or offshore, and up-coast or down-coast
51 
52 ===============================================================================================================================*/
53 bool CSimulation::bOnOrOffShoreAndUpOrDownCoast(double const dCoastAngle, double const dWaveOrientation, int const nSeaHand, bool& bDownCoast)
54 {
55  bool bOnShore;
56  double dWaveToCoastAngle = fmod((dWaveOrientation - dCoastAngle + 360), 360);
57 
58  bDownCoast = ((dWaveToCoastAngle > 270) || (dWaveToCoastAngle < 90)) ? true : false;
59 
60  if (nSeaHand == RIGHT_HANDED)
61  {
62  // The sea is on the RHS travelling down-coast
63  bOnShore = dWaveToCoastAngle > 180 ? true : false;
64  }
65  else
66  {
67  // The sea is on the LHS travelling down-coast
68  bOnShore = dWaveToCoastAngle > 180 ? false : true;
69  }
70 
71  return bOnShore;
72 }

666 int CSimulation::nFloodFillShadowZone(int const nZone, CGeom2DIPoint const* pPtiCentroid, CGeom2DIPoint const* pPtiShadowBoundaryStart, CGeom2DIPoint const* pPtiShadowBoundaryEnd)

672  {
673  // No it isn't: so try to find a cell that is
674  bStartPointOK = false;

680

682  if (PtiFloodFillStart == *pPtiCentroid)
683  {
684  dWeight += 0.05;
685  continue;
686  }
687 
688  // Safety check
689  if (! bIsWithinValidGrid(&PtiFloodFillStart))
690  {
691  LogStream << m_ulIter << ": " << ERR << "start point [" << PtiFloodFillStart.nGetX() << "][" << PtiFloodFillStart.nGetY() << "] = {" << dGridCentroidXToExtCRSX(PtiFloodFillStart.nGetX()) << ", " << dGridCentroidYToExtCRSY(PtiFloodFillStart.nGetY()) << "} for flood fill of shadow zone is outside grid" << endl;

705

707  }
708  }
709  }
710 
711  if (! bStartPointOK)
712  {

719 
720  // All OK, so create an empty stack
721  stack<CGeom2DIPoint> PtiStack;
722

724  PtiStack.push(PtiFloodFillStart);
725 
726  // Then do the flood fill: loop until there are no more cell co-ords on the stack
727  while (! PtiStack.empty())

730  PtiStack.pop();
731 
732  int
733  nX = Pti.nGetX(),
734  nY = Pti.nGetY();
735

738 
739  nX++;
740

746  {

781 
782  Traverse the shadow zone, changing wave orientation and height, and the down-drift zone, changing only wave height. Do this by following the coast between the shadow boundary start point and end point, and following the downdrift boundary between the same points. At each step, trace a linking line, then move along this line and change wave properties
783 
784 ===============================================================================================================================*/
785 int CSimulation::nDoShadowZoneAndDownDriftZone(int const nCoast, int const nZone, int const nShadowBoundaryStartPoint, int const nShadowBoundaryEndPoint)
786 {
787  int
788  nCoastSeaHand = m_VCoast[nCoast].nGetSeaHandedness(),
789  nShadowZoneCoastToCapeSeaHand;
790 
791  if (nCoastSeaHand == LEFT_HANDED)
792  nShadowZoneCoastToCapeSeaHand = RIGHT_HANDED;
793  else
794  nShadowZoneCoastToCapeSeaHand = LEFT_HANDED;
795 
796  // We will traverse the coastline from the start point of the shadow zone line, going toward the end point. Which direction is this?
797  bool bSweepDownCoast = true;
798  if (nShadowBoundaryEndPoint < nShadowBoundaryStartPoint)
799  bSweepDownCoast = false;
800 
801  // Get the distance (in cells) from the shadow boundary start point to the shadow boundary end point, going along the coast
802  int nAlongCoastDistanceToShadowEndpoint = tAbs(nShadowBoundaryEndPoint - nShadowBoundaryStartPoint - 1);
803 
804  // Calculate the point on the coastline which is 2 * nAlongCoastDistanceToShadowEndpoint from the shadow boundary start point, this will be the end point of the downdrift zone. This point may be beyond the end of the coastline in either direction
805  int
806  nDownDriftEndPoint,
807  nTotAlongCoastDistanceToDownDriftEndpoint = 2 * nAlongCoastDistanceToShadowEndpoint;
808 
809  if (bSweepDownCoast)
810  nDownDriftEndPoint = nShadowBoundaryStartPoint + nTotAlongCoastDistanceToDownDriftEndpoint;
811  else
812  nDownDriftEndPoint = nShadowBoundaryStartPoint - nTotAlongCoastDistanceToDownDriftEndpoint;
813 
814  // Next find the actual (i.e. within-grid) end of the downdrift line
815  CGeom2DIPoint PtiDownDriftEndPoint;
816 
817  // Is the downdrift end point beyond the start or end of the coastline?
818  if (nDownDriftEndPoint < 0)
819  {
820  // Is beyond the start of the coastline
821  int nStartEdge = m_VCoast[nCoast].nGetStartEdge();
822  if (nStartEdge == NORTH)
823  {
824  PtiDownDriftEndPoint.SetX(m_VCoast[nCoast].pPtiGetCellMarkedAsCoastline(0)->nGetX());
825  PtiDownDriftEndPoint.SetY(nDownDriftEndPoint);
826  }
827  else if (nStartEdge == SOUTH)
828  {
829  PtiDownDriftEndPoint.SetX(m_VCoast[nCoast].pPtiGetCellMarkedAsCoastline(0)->nGetX());
830  PtiDownDriftEndPoint.SetY(m_nYGridMax - nDownDriftEndPoint - 1);
831  }
832  else if (nStartEdge == WEST)
833  {
834  PtiDownDriftEndPoint.SetX(nDownDriftEndPoint);
835  PtiDownDriftEndPoint.SetY(m_VCoast[nCoast].pPtiGetCellMarkedAsCoastline(0)->nGetY());
836  }
837  else if (nStartEdge == EAST)
838  {
839  PtiDownDriftEndPoint.SetX(m_nXGridMax - nDownDriftEndPoint - 1);
840  PtiDownDriftEndPoint.SetY(m_VCoast[nCoast].pPtiGetCellMarkedAsCoastline(0)->nGetY());
841  }
842  }
843  else if (nDownDriftEndPoint >= m_VCoast[nCoast].nGetCoastlineSize())
844  {
845  // Is beyond the end of the coastline
846  int
847  nEndEdge = m_VCoast[nCoast].nGetEndEdge(),
848  nCoastSize = m_VCoast[nCoast].nGetCoastlineSize();
849  if (nEndEdge == NORTH)
850  {
851  PtiDownDriftEndPoint.SetX(m_VCoast[nCoast].pPtiGetCellMarkedAsCoastline(nCoastSize-1)->nGetX());
852  PtiDownDriftEndPoint.SetY(-nDownDriftEndPoint);
853  }
854  else if (nEndEdge == SOUTH)
855  {
856  PtiDownDriftEndPoint.SetX(m_VCoast[nCoast].pPtiGetCellMarkedAsCoastline(nCoastSize-1)->nGetX());
857  PtiDownDriftEndPoint.SetY(m_nYGridMax + nDownDriftEndPoint);
858  }
859  else if (nEndEdge == WEST)
860  {
861  PtiDownDriftEndPoint.SetX(-nDownDriftEndPoint);
862  PtiDownDriftEndPoint.SetY(m_VCoast[nCoast].pPtiGetCellMarkedAsCoastline(nCoastSize-1)->nGetY());
863  }
864  else if (nEndEdge == EAST)
865  {
866  PtiDownDriftEndPoint.SetX(m_nXGridMax + nDownDriftEndPoint);
867  PtiDownDriftEndPoint.SetY(m_VCoast[nCoast].pPtiGetCellMarkedAsCoastline(nCoastSize-1)->nGetY());
868  }
869  }
870  else
871  {
872  // Is on the coastline, so get the location (grid CRS)
873  PtiDownDriftEndPoint = *m_VCoast[nCoast].pPtiGetCellMarkedAsCoastline(nDownDriftEndPoint);
874  }
875 
876  // Get the location (grid CRS) of the shadow boundary start point: this is also the start point of the downdrift boundary
877  CGeom2DIPoint* pPtiDownDriftBoundaryStartPoint = m_VCoast[nCoast].pPtiGetCellMarkedAsCoastline(nShadowBoundaryStartPoint);
878 
879  // Now trace the down-drift boundary line: interpolate between cells by a simple DDA line algorithm, see http://en.wikipedia.org/wiki/Digital_differential_analyzer_(graphics_algorithm) Note that Bresenham's algorithm gave occasional gaps
880  int
881  nXStart = pPtiDownDriftBoundaryStartPoint->nGetX(),
882  nYStart = pPtiDownDriftBoundaryStartPoint->nGetY();
883  double
884  dXStart = dGridCentroidXToExtCRSX(nXStart),
885  dYStart = dGridCentroidYToExtCRSY(nYStart),
886  dXEnd = dGridCentroidXToExtCRSX(PtiDownDriftEndPoint.nGetX()),
887  dYEnd = dGridCentroidYToExtCRSY(PtiDownDriftEndPoint.nGetY()),
888  dXInc = dXEnd - dXStart,
889  dYInc = dYEnd - dYStart,
890  dLength = tMax(tAbs(dXInc), tAbs(dYInc));
891 
892  dXInc /= dLength;
893  dYInc /= dLength;
894 
895  int nTotDownDriftBoundaryDistance = 0;
896  double
897  dX = nXStart,
898  dY = nYStart;
899 
900  CGeomLine LDownDriftBoundary;
901 
902  // Process each interpolated point
903  for (int m = 0; m <= nRound(dLength); m++)
904  {
905  int
906  nX = static_cast<int>(dX),
907  nY = static_cast<int>(dY);
908 
909  if (! bIsWithinValidGrid(nX, nY))
910  {
911  // Safety check
912  break;
913  }
914 
915  // OK, this is part of the downdrift boundary so store this co-ordinate and mark the cell
916  CGeom2DPoint PtThis(dGridCentroidXToExtCRSX(nX), dGridCentroidYToExtCRSY(nY));
917 
918  // Make sure we have not already stored this co-ordinate (can happen, due to rounding)
919  if ((LDownDriftBoundary.nGetSize() == 0) || (PtThis != LDownDriftBoundary.pPtBack()))
920  {
921  // Store this co-ordinate
922  LDownDriftBoundary.Append(&PtThis);
923 
924  // Mark the cell (a +ve number, same as the associated shadow zone number i.e. starting from 1)
925  m_pRasterGrid->m_Cell[nX][nY].SetDownDriftZoneNumber(nZone+1);
926 
927  // Increment the boundary length
928  nTotDownDriftBoundaryDistance++;
929 
930 // LogStream << "DownDrift boundary [" << nX << "][" << nY << "] = {" << dGridCentroidXToExtCRSX(nX) << ", " << dGridCentroidYToExtCRSY(nY) << "}" << endl;
931 
932  // And increment for next time
933  if (dXEnd > dXStart)
934  dX -= dXInc;
935  else
936  dX += dXInc;
937 
938  if (dYEnd > dYStart)
939  dY += dYInc;
940  else
941  dY -= dYInc;
942  }
943  }
944 
945  // Store the downdrift boundary (external CRS), with the start point first
946  m_VCoast[nCoast].AppendShadowDowndriftBoundary(LDownDriftBoundary);
947 
948  // Compare the lengths of the along-coast and the along-downdrift boundaries. The increment will be 1 for the smaller of the two, will be > 1 for the larger of the two
949  int nMaxDistance;
950  double
951  dAlongCoastIncrement = 1,
952  dDownDriftBoundaryIncrement = 1;
953 
954  if (nTotAlongCoastDistanceToDownDriftEndpoint < nTotDownDriftBoundaryDistance)
955  {
956  // The downdrift boundary distance is the larger, so change it
957  dDownDriftBoundaryIncrement = static_cast<double>(nTotDownDriftBoundaryDistance) / nTotAlongCoastDistanceToDownDriftEndpoint;
958  nMaxDistance = nTotDownDriftBoundaryDistance;
959  }
960  else
961  {
962  // The along-coast distance is the larger, so change it
963  dAlongCoastIncrement = static_cast<double>(nTotAlongCoastDistanceToDownDriftEndpoint) / nTotDownDriftBoundaryDistance;
964  nMaxDistance = nTotAlongCoastDistanceToDownDriftEndpoint;
965  }
966 
967  double
968  dCoastDistSoFar = 0,
969  dDownDriftBoundaryDistSoFar = 0;
970 
971  // Now traverse the along-coast line and the down-drift boundary line, but with different increments for each
972  for (int n = 1; n < nMaxDistance-1; n++)
973  {
974  dCoastDistSoFar += dAlongCoastIncrement;
975  dDownDriftBoundaryDistSoFar += dDownDriftBoundaryIncrement;
976 
977 // LogStream << "dDownDriftBoundaryDistSoFar = " << dDownDriftBoundaryDistSoFar << " nTotDownDriftBoundaryDistance = " << nTotDownDriftBoundaryDistance << endl;
978 
979  if ((dCoastDistSoFar >= nTotAlongCoastDistanceToDownDriftEndpoint) || (dDownDriftBoundaryDistSoFar >= nTotDownDriftBoundaryDistance))
980  break;
981 
982  bool bPastShadowEnd = false;
983  int nAlongCoast;
984  if (bSweepDownCoast)
985  {
986  nAlongCoast = nShadowBoundaryStartPoint + nRound(dCoastDistSoFar);
987 
988  if (nAlongCoast >= m_VCoast[nCoast].nGetCoastlineSize())
989  break;
990 
991  if (nAlongCoast >= nShadowBoundaryEndPoint)
992  bPastShadowEnd = true;
993  }
994  else
995  {
996  nAlongCoast = nShadowBoundaryStartPoint - nRound(dCoastDistSoFar);
997 
998  if (nAlongCoast < 0)
999  break;
1000 
1001  if (nAlongCoast <= nShadowBoundaryEndPoint)
1002  bPastShadowEnd = true;
1003  }
1004 
1005  int nAlongDownDriftBoundary = nRound(dDownDriftBoundaryDistSoFar);
1006 
1007 // LogStream << endl << m_ulIter << ": dCoastDistSoFar = " << dCoastDistSoFar << " (nTotAlongCoastDistanceToDownDriftEndpoint = " << nTotAlongCoastDistanceToDownDriftEndpoint << ") dDownDriftBoundaryDistSoFar = " << dDownDriftBoundaryDistSoFar << " (nTotDownDriftBoundaryDistance = " << nTotDownDriftBoundaryDistance << ")" << endl;
1008 
1009  // nAlongCoast = " << nAlongCoast << ", nShadowBoundaryEndPoint = " << nShadowBoundaryEndPoint << ", << ", nAlongDownDriftBoundary = " << nAlongDownDriftBoundary << ", << endl;
1010 
1011  // Get the two endpoints of the linking line
1012  CGeom2DIPoint* pPtiCoast = m_VCoast[nCoast].pPtiGetCellMarkedAsCoastline(nAlongCoast);
1013  int
1014  nCoastX = pPtiCoast->nGetX(),
1015  nCoastY = pPtiCoast->nGetY();
1016 
1017  int
1018  nDownDriftX = nRound(dExtCRSXToGridX(LDownDriftBoundary[nAlongDownDriftBoundary].dGetX())),
1019  nDownDriftY = nRound(dExtCRSYToGridY(LDownDriftBoundary[nAlongDownDriftBoundary].dGetY()));
1020 
1021  // Safety check, in case the two points are identical (can happen due to rounding)
1022  if ((nCoastX == nDownDriftX) && (nCoastY == nDownDriftY))
1023  {
1024  LogStream << m_ulIter << ": coast point and downdrift boundary point [" << nCoastX << "][" << nCoastY << "] = {" << dGridCentroidXToExtCRSX(nCoastX) << ", " << dGridCentroidYToExtCRSY(nCoastY) << "} are identical, ignoring" << endl;
1025 
1026  continue;
1027  }
1028 
1029  // Traverse the linking line, interpolating between cells by a simple DDA line algorithm, see http://en.wikipedia.org/wiki/Digital_differential_analyzer_(graphics_algorithm)
1030  dXInc = nDownDriftX - nCoastX;
1031  dYInc = nDownDriftY - nCoastY;
1032  double dLinkingLineLength = tMax(tAbs(dXInc), tAbs(dYInc));
1033 
1034  dXInc /= dLinkingLineLength;
1035  dYInc /= dLinkingLineLength;
1036 
1037  dX = nCoastX,
1038  dY = nCoastY;
1039 
1040  // Process each interpolated point along the linking line
1041  int
1042  nXLast = -1,
1043  nYLast = -1,
1044  nShadowZoneLength = 0;
1045  vector<int> VnShadowCellX, VnShadowCellY;
1046  for (int m = 0; m < dLinkingLineLength; m++)
1047  {
1048  int
1049  nX = nRound(dX),
1050  nY = nRound(dY);
1051 
1052  // Check to see if we just processed this point, can happen due to rounding
1053  if ((nX == nXLast) && (nY == nYLast))
1054  {
1055 // LogStream << m_ulIter << ": n = " << n << ", m = " << m << ", dLinkingLineLength = " << dLinkingLineLength << ", dCoastDistSoFar = " << dCoastDistSoFar << " (nTotAlongCoastDistanceToDownDriftEndpoint = " << nTotAlongCoastDistanceToDownDriftEndpoint << "), dDownDriftBoundaryDistSoFar = " << dDownDriftBoundaryDistSoFar << " (nTotDownDriftBoundaryDistance = " << nTotDownDriftBoundaryDistance << ") same as last point at [" << nX << "][" << nY << "] = {" << dGridCentroidXToExtCRSX(nX) << ", " << dGridCentroidYToExtCRSY(nY) << "}" << endl;
1056 
1057  // Set for next time
1058  nXLast = nX;
1059  nYLast = nY;
1060  dX += dXInc;
1061  dY += dYInc;
1062 
1063  continue;
1064  }
1065 
1066  // Outside valid grid?
1067  if (! bIsWithinValidGrid(nX, nY))
1068  {
1069 // LogStream << m_ulIter << ": n = " << n << ", m = " << m << ", dLinkingLineLength = " << dLinkingLineLength << ", dCoastDistSoFar = " << dCoastDistSoFar << " (nTotAlongCoastDistanceToDownDriftEndpoint = " << nTotAlongCoastDistanceToDownDriftEndpoint << "), dDownDriftBoundaryDistSoFar = " << dDownDriftBoundaryDistSoFar << " (nTotDownDriftBoundaryDistance = " << nTotDownDriftBoundaryDistance << ") outside valid grid at [" << nX << "][" << nY << "] = {" << dGridCentroidXToExtCRSX(nX) << ", " << dGridCentroidYToExtCRSY(nY) << "}" << endl;
1070 
1071  // Set for next time
1072  nXLast = nX;
1073  nYLast = nY;
1074  dX += dXInc;
1075  dY += dYInc;
1076 
1077  continue;
1078  }
1079 
1080  // Not a sea cell?
1081  if (! m_pRasterGrid->m_Cell[nX][nY].bIsInContiguousSea())
1082  {
1083  // Not a sea cell
1084 // LogStream << m_ulIter << ": n = " << n << ", m = " << m << ", dLinkingLineLength = " << dLinkingLineLength << ", dCoastDistSoFar = " << dCoastDistSoFar << " (nTotAlongCoastDistanceToDownDriftEndpoint = " << nTotAlongCoastDistanceToDownDriftEndpoint << "), dDownDriftBoundaryDistSoFar = " << dDownDriftBoundaryDistSoFar << " (nTotDownDriftBoundaryDistance = " << nTotDownDriftBoundaryDistance << ") not a sea cell at [" << nX << "][" << nY << "] = {" << dGridCentroidXToExtCRSX(nX) << ", " << dGridCentroidYToExtCRSY(nY) << "}" << endl;
1085 
1086  // Set for next time
1087  nXLast = nX;
1088  nYLast = nY;
1089  dX += dXInc;
1090  dY += dYInc;
1091 
1092  continue;
1093  }
1094 
1095  // Have we gone past the point where the shadow boundary meets the coast (i.e. the shadow boundary end point)?
1096  if (! bPastShadowEnd)
1097  {
1098  // We have not, so the linking line has two parts: one between the coast and the shadow boundary, one between the shadow boundary and the downdrift boundary
1099  bool bInShadowZone = true;
1100 
1101  if (! m_pRasterGrid->m_Cell[nX][nY].bIsinAnyShadowZone())
1102  {
1103  // We have left the shadow zone
1104 // LogStream << "[" << nX << "][" << nY << "] = {" << dGridCentroidXToExtCRSX(nX) << ", " << dGridCentroidYToExtCRSY(nY) << "} LEFT SHADOW ZONE" << endl;
1105 
1106  bInShadowZone = false;
1107 
1108  // Go back over stored cell coords and set their wave properties
1109  for (unsigned int mm = 0; mm < VnShadowCellX.size(); mm++)
1110  {
1111  // Process this shadow zone cell
1112  ProcessShadowZoneCell(VnShadowCellX[mm], VnShadowCellY[mm], nShadowZoneCoastToCapeSeaHand, pPtiCoast, VnShadowCellX.back(), VnShadowCellY.back(), nZone);
1113 
1114  // Also process adjacent cells
1115  if (mm > 0)
1116  {
1117  CGeom2DIPoint
1118  PtiLeft = PtiGetPerpendicular(VnShadowCellX[mm], VnShadowCellY[mm], VnShadowCellX[mm-1], VnShadowCellY[mm-1], 1, RIGHT_HANDED),
1119  PtiRight = PtiGetPerpendicular(VnShadowCellX[mm], VnShadowCellY[mm], VnShadowCellX[mm-1], VnShadowCellY[mm-1], 1, LEFT_HANDED);
1120 
1121  if (bIsWithinValidGrid(&PtiLeft))
1122  ProcessShadowZoneCell(PtiLeft.nGetX(), PtiLeft.nGetY(), nShadowZoneCoastToCapeSeaHand, pPtiCoast, VnShadowCellX.back(), VnShadowCellY.back(), nZone);
1123 
1124  if (bIsWithinValidGrid(&PtiRight))
1125  ProcessShadowZoneCell(PtiRight.nGetX(), PtiRight.nGetY(), nShadowZoneCoastToCapeSeaHand, pPtiCoast, VnShadowCellX.back(), VnShadowCellY.back(), nZone);
1126  }
1127  }
1128  }
1129 
1130  if (bInShadowZone)
1131  {
1132  // Save coords for later
1133  VnShadowCellX.push_back(nX);
1134  VnShadowCellY.push_back(nY);
1135 
1136  nShadowZoneLength++;
1137  }
1138  else
1139  {
1140  // In downdrift zone
1141 
1142 // LogStream << m_ulIter << ": n = " << n << ", m = " << m << ", dLinkingLineLength = " << dLinkingLineLength << ", dCoastDistSoFar = " << dCoastDistSoFar << " (nTotAlongCoastDistanceToDownDriftEndpoint = " << nTotAlongCoastDistanceToDownDriftEndpoint << "), dDownDriftBoundaryDistSoFar = " << dDownDriftBoundaryDistSoFar << " (nTotDownDriftBoundaryDistance = " << nTotDownDriftBoundaryDistance << ") has [" << nX << "][" << nY << "] = {" << dGridCentroidXToExtCRSX(nX) << ", " << dGridCentroidYToExtCRSY(nY) << "} in downdrift zone" << endl;
1143 
1144  // Process this downdrift cell
1145  ProcessDownDriftCell(nX, nY, (m - nShadowZoneLength), (dLinkingLineLength - nShadowZoneLength), nZone);
1146 
1147  // Also process adjacent cells
1148  CGeom2DIPoint
1149  PtiLeft = PtiGetPerpendicular(nX, nY, nXLast, nYLast, 1, RIGHT_HANDED),
1150  PtiRight = PtiGetPerpendicular(nX, nY, nXLast, nYLast, 1, LEFT_HANDED);
1151 
1152  if (bIsWithinValidGrid(&PtiLeft))
1153  ProcessDownDriftCell(PtiLeft.nGetX(), PtiLeft.nGetY(), (m - nShadowZoneLength), (dLinkingLineLength - nShadowZoneLength), nZone);
1154 
1155  if (bIsWithinValidGrid(&PtiRight))
1156  ProcessDownDriftCell(PtiRight.nGetX(), PtiRight.nGetY(), (m - nShadowZoneLength), (dLinkingLineLength - nShadowZoneLength), nZone);
1157  }
1158  }
1159  else
1160  {
1161  // We have, so the linking line has only one part: between the coast and the downdrift boundary.
1162 
1163 // LogStream << m_ulIter << ": n = " << n << ", m = " << m << ", dLinkingLineLength = " << dLinkingLineLength << ", dCoastDistSoFar = " << dCoastDistSoFar << " (nTotAlongCoastDistanceToDownDriftEndpoint = " << nTotAlongCoastDistanceToDownDriftEndpoint << "), dDownDriftBoundaryDistSoFar = " << dDownDriftBoundaryDistSoFar << " (nTotDownDriftBoundaryDistance = " << nTotDownDriftBoundaryDistance << ") has [" << nX << "][" << nY << "] = {" << dGridCentroidXToExtCRSX(nX) << ", " << dGridCentroidYToExtCRSY(nY) << "} in downdrift zone" << endl;
1164 
1165  // Process this downdrift cell
1166  ProcessDownDriftCell(nX, nY, m, dLinkingLineLength, nZone);
1167 
1168  // Also process adjacent cells
1169  if ((nXLast != -1) && (nYLast != -1))
1170  {
1171  CGeom2DIPoint
1172  PtiLeft = PtiGetPerpendicular(nX, nY, nXLast, nYLast, 1, RIGHT_HANDED),
1173  PtiRight = PtiGetPerpendicular(nX, nY, nXLast, nYLast, 1, LEFT_HANDED);
1174 
1175  if (bIsWithinValidGrid(&PtiLeft))
1176  ProcessDownDriftCell(PtiLeft.nGetX(), PtiLeft.nGetY(), m, dLinkingLineLength, nZone);
1177 
1178  if (bIsWithinValidGrid(&PtiRight))
1179  ProcessDownDriftCell(PtiRight.nGetX(), PtiRight.nGetY(), m, dLinkingLineLength, nZone);
1180  }
1181  }
1182 
1183  // Set for next time
1184  nXLast = nX;
1185  nYLast = nY;
1186  dX += dXInc;
1187  dY += dYInc;
1188  }
1189  }
1190 
1191  return RTN_OK;
1192 }

1235  return;

1252 /*===============================================================================================================================
1253 
1254  Process a single cell which is in the shadow zone, changing its wave height and orientation
1255 
1256  ===============================================================================================================================*/
1257 void CSimulation::ProcessShadowZoneCell(int const nX, int const nY, int const nShadowZoneCoastToCapeSeaHand, CGeom2DIPoint const* pPtiCoast, int const nShadowEndX, int const nShadowEndY, int const nZone)
1258 {
1259  int nZoneCode = m_pRasterGrid->m_Cell[nX][nY].nGetShadowZoneNumber();
1260  if (nZoneCode == (-nZone-1))
1261  {
1262  // OK, we are in the shadow zone and have not already processed this cell, so mark it (a +ve number, starting from 1)
1263  m_pRasterGrid->m_Cell[nX][nY].SetShadowZoneNumber(nZone+1);
1264 
1265  // Next calculate wave angle here: first calculate dOmega, the signed angle subtended between this end point and the start point, and this end point and the end of the shadow boundary
1266  CGeom2DIPoint
1267  PtiThis(nX, nY),
1268  PtiShadowBoundary(nShadowEndX, nShadowEndY);
1269  double dOmega = 180 * dAngleSubtended(pPtiCoast, &PtiThis, &PtiShadowBoundary) / PI;
1270 
1271  // If dOmega is 90 degrees or more in either direction, set both wave angle and wave height to zero
1272  if (tAbs(dOmega) >= 90)
1273  {
1274  m_pRasterGrid->m_Cell[nX][nY].SetWaveOrientation(0);
1275  m_pRasterGrid->m_Cell[nX][nY].SetWaveHeight(0);
1276 
1277 // LogStream << m_ulIter << ": on shadow linking line with coast end [" << pPtiCoast->nGetX() << "][" << pPtiCoast->nGetY() << "] = {" << dGridCentroidXToExtCRSX(pPtiCoast->nGetX()) << ", " << dGridCentroidYToExtCRSY(pPtiCoast->nGetY()) << "} and shadow boundary end [" << PtiShadowBoundary.nGetX() << "][" << PtiShadowBoundary.nGetY() << "] = {" << dGridCentroidXToExtCRSX(PtiShadowBoundary.nGetX()) << ", " << dGridCentroidYToExtCRSY(PtiShadowBoundary.nGetY()) << "}, this point [" << nX << "][" << nY << "] = {" << dGridCentroidXToExtCRSX(nX) << ", " << dGridCentroidYToExtCRSY(nY) << "}" << endl << "angle subtended = " << dOmega << " degrees, m_pRasterGrid->m_Cell[" << nX << "][" << nY << "].dGetDeepWaterWaveHeight() = " << m_pRasterGrid->m_Cell[nX][nY].dGetDeepWaterWaveHeight() << " degrees, wave orientation = 0 degrees, wave height = 0 m" << endl;
1278  }
1279  else
1280  {
1281  // Adapted from equation 12 in Hurst et al.
1282  double dDeltaShadowWaveAngle = 1.5 * dOmega;
1283 
1284  // Get the pre-existing (i.e. shore-parallel) wave orientation
1285  double dWaveOrientation = m_pRasterGrid->m_Cell[nX][nY].dGetWaveOrientation();
1286 
1287  double dShadowWaveOrientation;
1288  if (nShadowZoneCoastToCapeSeaHand == LEFT_HANDED)
1289  dShadowWaveOrientation = dWaveOrientation + dDeltaShadowWaveAngle;
1290  else
1291  dShadowWaveOrientation = dWaveOrientation - dDeltaShadowWaveAngle;
1292 
1293  // Set the shadow zone wave orientation
1294  m_pRasterGrid->m_Cell[nX][nY].SetWaveOrientation(dKeepWithin360(dShadowWaveOrientation));
1295 
1296  // Now calculate wave height within the shadow zone, use equation 13 from Hurst et al.
1297  double dKp = 0.5 * cos(dOmega * PI / 180);
