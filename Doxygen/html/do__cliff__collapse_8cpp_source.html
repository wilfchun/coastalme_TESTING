40 /*===============================================================================================================================
41 
42  Update accumulated wave energy in coastal landform objects
43 
44 ===============================================================================================================================*/
45 int CSimulation::nDoAllWaveEnergyToCoastLandforms(void)
46 {
47  int nRet;
48 
49  // First go along each coastline and update the total wave energy which it has experienced
50  for (int i = 0; i < static_cast<int>(m_VCoast.size()); i++)
51  {
52  for (int j = 0; j < m_VCoast[i].nGetCoastlineSize(); j++)
53  {
54  CACoastLandform* pCoastLandform = m_VCoast[i].pGetCoastLandform(j);
55 
56  // Update accumulated wave energy for the coastal landform object
57  double
58  dWaveHeightAtCoast = m_VCoast[i].dGetCoastWaveHeight(j),
59  dDeepWaterWavePeriod = m_VCoast[i].dGetDeepWaterWavePeriod(j),
60  dWaveErosiveForce = pow(dWaveHeightAtCoast, WALKDEN_HALL_PARAM_1) * pow(dDeepWaterWavePeriod, WALKDEN_HALL_PARAM_2),
61  dWaveEnergy = dWaveErosiveForce * m_dTimeStep * 3600;
62 
63 // assert(bIsFinite(dWaveEnergy));
64  pCoastLandform->IncTotAccumWaveEnergy(dWaveEnergy);
65 
66  // Now simulate how the coastal landform responds to this wave energy
67  int nCategory = pCoastLandform->nGetLandFormCategory();
68  if (nCategory == LF_CAT_CLIFF)
69  {
70  // This is a cliff
71  CRWCliff* pCliff = reinterpret_cast<CRWCliff*>(pCoastLandform);
72 
73  // Calculate this-timestep cliff notch erosion (is a length in external CRS units)
74  double dNotchExtension = dWaveEnergy / m_dCliffErosionResistance;
75 
76  // Constrain this-timestep notch extension if it is more than the length of one cell side (in external CRS units), since the most we can remove in a single timestep is one coastal cell
77  dNotchExtension = tMin(m_dCellSide, dNotchExtension);
78 
79  // Extend the cliff object's erosional notch as a result of wave energy during this timestep. Note that extension may be constrained, since this-timestep extension cannot exceed the depth of sediment remaining on the cell
80  dNotchExtension = pCliff->dErodeNotch(dNotchExtension); // Constrain
81 
82  // OK, is the notch now extended enough to cause collapse (either because the overhang is greater than the threshold overhang, or because there is no sediment remaining)?
83  if (pCliff->bReadyToCollapse(m_dNotchOverhangAtCollapse))
84  {
85  // It is ready to collapse
86  double
87  dFineCollapse = 0,
88  dSandCollapse = 0,
89  dCoarseCollapse = 0;
90 
91  // So do the cliff collapse
92  nRet = nDoCliffCollapse(pCliff, dNotchExtension, dFineCollapse, dSandCollapse, dCoarseCollapse);
93  if (nRet != RTN_OK)
94  LogStream << m_ulIter << WARN << " problem with cliff collapse, continuing however" << endl;
95 
96  // And put fine sediment into suspension, and deposit sand and/or coarse sediment as unconsolidated sediment
97  nRet = nDoCliffCollapseDeposition(pCliff, dFineCollapse, dSandCollapse, dCoarseCollapse);
98  if (nRet != RTN_OK)
99  return nRet;
100  }
101  }
102  }
103  }
104 
105  LogStream << endl << m_ulIter << ": cliff collapse (m^3) = " << (m_dThisTimestepCliffCollapseErosionFine + m_dThisTimestepCliffCollapseErosionSand + m_dThisTimestepCliffCollapseErosionCoarse) * m_dCellArea << " (fine = " << m_dThisTimestepCliffCollapseErosionFine * m_dCellArea << ", sand = " << m_dThisTimestepCliffCollapseErosionSand * m_dCellArea << ", coarse = " << m_dThisTimestepCliffCollapseErosionCoarse * m_dCellArea << "), talus deposition (m^3) = " << (m_dThisTimestepCliffDepositionSand + m_dThisTimestepCliffDepositionCoarse) * m_dCellArea << " (sand = " << m_dThisTimestepCliffDepositionSand * m_dCellArea << ", coarse = " << m_dThisTimestepCliffDepositionSand * m_dCellArea << ")" << endl;
106 
107  return RTN_OK;
108 }

117 {

119  int

121  nY = pCliff->pPtiGetCellMarkedAsLF()->nGetY();
122

124  double dNotchElev = pCliff->dGetNotchBaseElev() - m_dNotchBaseBelowSWL;
125 
126  // Get the index of the layer containing the notch (layer 0 being just above basement)
127  int nNotchLayer = m_pRasterGrid->m_Cell[nX][nY].nGetLayerAtElev(dNotchElev);
128  if (nNotchLayer == ELEV_ABOVE_SEDIMENT_TOP)
129  {
130  LogStream << endl << m_ulIter << ": " << ERR << " cell [" << nX << "][" << nY << "] has dNotchElev (" << dNotchElev << ") above sediment top elevation (" << m_pRasterGrid->m_Cell[nX][nY].dGetSedimentTopElev() << ")" << endl;
131 
132  return RTN_ERR_CLIFFNOTCH;
133  }
134 // else
135 // LogStream << endl << m_ulIter << ": for cell [" << nX << "][" << nY << "] dNotchElev = " << dNotchElev << " sediment top elevation = " << m_pRasterGrid->m_Cell[nX][nY].dGetSedimentTopElev() << endl;

141 
142  // Safety check

144  return RTN_ERR_NO_TOP_LAYER;
145 
146  double dRemaining = pCliff->dGetRemaining();
147  if (dRemaining <= 0)
148  {
149  // No cliff sediment left on this cliff object, so the cell which it occupies will no longer be a cliff in the next timestep
150  pCliff->SetAllSedimentGone();
151 
152  // Set the base of the collapse (see above)
153  pCliff->SetNotchBaseElev(dNotchElev);
154 
155  // Set flags to say that the top layer has changed
156  m_bConsChangedThisTimestep[nTopLayer] = true;
157  m_bUnconsChangedThisTimestep[nTopLayer] = true;

159 // int nX = pCliff->pPtiGetCellMarkedAsLF()->nGetX();
160 // int nY = pCliff->pPtiGetCellMarkedAsLF()->nGetY();

163

165  double dAboveNotch = m_pRasterGrid->m_Cell[nX][nY].dGetVolEquivSedTopElev() - dNotchElev;
166 
167  // In CoastalME, all depth equivalents are assumed to be a depth upon the whole of a cell i.e. upon the area of a whole cell. The vertical depth of sediment lost in each cliff collapse is a depth upon only part of a cell, i.e. upon a fraction of a cell's area. To keep the depth of cliff collapse consistent with all other depth equivalents, weight it by the fraction of the cell's area which is being removed
168  double
169  dNotchAreaFrac = dNotchDeepen / m_dCellSide,
170  dCollapseDepth = dAboveNotch * dNotchAreaFrac;
171 
172  // Update the cell's totals for cliff collapse
173  m_pRasterGrid->m_Cell[nX][nY].IncrCliffCollapse(dCollapseDepth);
174 
175  double
176  dAvailable = 0,
177  dLost;
178 
179 // LogStream << m_ulIter << ": cell [" << nX << "][" << nY << "] before removing sediment, dGetVolEquivSedTopElev() = " << m_pRasterGrid->m_Cell[nX][nY].dGetVolEquivSedTopElev() << ", dGetSedimentTopElev() = " << m_pRasterGrid->m_Cell[nX][nY].dGetSedimentTopElev() << endl;
180 
181  // Now update the cell's sediment. If there are sediment layers above the notched layer, we must remove sediment from the whole depth of each layer. Again, weight the depth lost by the fraction of the cell's area which is being removed
182  for (int n = nTopLayer; n > nNotchLayer; n--)
183  {
184  // Add in this layer's sediment, both consolidated and unconsolidated, and adjust what is left
185  dAvailable = m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(n)->pGetUnconsolidatedSediment()->dGetFine() - m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(n)->pGetUnconsolidatedSediment()->dGetNotchFineLost();
186  if (dAvailable > 0)
187  {
188  dLost = dAvailable * dNotchAreaFrac;
189  dFineCollapse += dLost;
190  m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(n)->pGetUnconsolidatedSediment()->IncrNotchFineLost(dLost);
191  }
192 
193  dAvailable = m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(n)->pGetUnconsolidatedSediment()->dGetSand() - m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(n)->pGetUnconsolidatedSediment()->dGetNotchSandLost();
194  if (dAvailable > 0)
195  {
196  dLost = dAvailable * dNotchAreaFrac;
197  dSandCollapse += dLost;
198  m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(n)->pGetUnconsolidatedSediment()->IncrNotchSandLost(dLost);
199  }
200 
201  dAvailable = m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(n)->pGetUnconsolidatedSediment()->dGetCoarse() - m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(n)->pGetUnconsolidatedSediment()->dGetNotchCoarseLost();
202  if (dAvailable > 0)
203  {
204  dLost = dAvailable * dNotchAreaFrac;
205  dCoarseCollapse += dLost;
206  m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(n)->pGetUnconsolidatedSediment()->IncrNotchCoarseLost(dLost);
207  }
208 
209  dAvailable = m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(n)->pGetConsolidatedSediment()->dGetFine() - m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(n)->pGetConsolidatedSediment()->dGetNotchFineLost();
210  if (dAvailable > 0)
211  {
212  dLost = dAvailable * dNotchAreaFrac;
213  dFineCollapse += dLost;
214  m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(n)->pGetConsolidatedSediment()->IncrNotchFineLost(dLost);
215  }
216 
217  dAvailable = m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(n)->pGetConsolidatedSediment()->dGetSand() - m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(n)->pGetConsolidatedSediment()->dGetNotchSandLost();
218  if (dAvailable > 0)
219  {
220  dLost = dAvailable * dNotchAreaFrac;
221  dSandCollapse += dLost;
222  m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(n)->pGetConsolidatedSediment()->IncrNotchSandLost(dLost);
223  }
224 
225  dAvailable = m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(n)->pGetConsolidatedSediment()->dGetCoarse() - m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(n)->pGetConsolidatedSediment()->dGetNotchCoarseLost();
226  if (dAvailable > 0)
227  {
228  dLost = dAvailable * dNotchAreaFrac;
229  dCoarseCollapse += dLost;
230  m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(n)->pGetConsolidatedSediment()->IncrNotchCoarseLost(dLost);
231  }
232  }
233 
234  // For the layer which contains the notch, remove only part of the sediment depth
235  double
236  dNotchLayerTop = m_pRasterGrid->m_Cell[nX][nY].dCalcLayerElev(nNotchLayer),
237  dNotchLayerThickness = m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(nNotchLayer)->dGetTotalThickness(),
238  dNotchLayerVertFracRemoved = (dNotchLayerTop - dNotchElev ) / dNotchLayerThickness;
239 
240  // Now calculate the fraction of the volume which is removed
241  double dNotchLayerFracRemoved = dNotchLayerVertFracRemoved * dNotchAreaFrac;
242 
243  // Sort out the notched layer's sediment, both consolidated and unconsolidated, for this cell
244  dAvailable = m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(nNotchLayer)->pGetUnconsolidatedSediment()->dGetFine() - m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(nNotchLayer)->pGetUnconsolidatedSediment()->dGetNotchFineLost();
245  if (dAvailable > 0)
246  {
247  dLost = dAvailable * dNotchLayerFracRemoved;
248  dFineCollapse += dLost;
249  m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(nNotchLayer)->pGetUnconsolidatedSediment()->IncrNotchFineLost(dLost);
250  }
251 
252  dAvailable = m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(nNotchLayer)->pGetUnconsolidatedSediment()->dGetSand() - m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(nNotchLayer)->pGetUnconsolidatedSediment()->dGetNotchSandLost();
253  if (dAvailable > 0)
254  {
255  dLost = dAvailable * dNotchLayerFracRemoved;
256  dSandCollapse += dLost;
257  m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(nNotchLayer)->pGetUnconsolidatedSediment()->IncrNotchSandLost(dLost);
258  }
259 
260  dAvailable = m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(nNotchLayer)->pGetUnconsolidatedSediment()->dGetCoarse() - m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(nNotchLayer)->pGetUnconsolidatedSediment()->dGetNotchCoarseLost();
261  if (dAvailable > 0)
262  {
263  dLost = dAvailable * dNotchLayerFracRemoved;
264  dCoarseCollapse += dLost;
265  m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(nNotchLayer)->pGetUnconsolidatedSediment()->IncrNotchCoarseLost(dLost);
266  }
267 
268  dAvailable = m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(nNotchLayer)->pGetConsolidatedSediment()->dGetFine() - m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(nNotchLayer)->pGetConsolidatedSediment()->dGetNotchFineLost();
269  if (dAvailable > 0)
270  {
271  dLost = dAvailable * dNotchLayerFracRemoved;
272  dFineCollapse += dLost;
273  m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(nNotchLayer)->pGetConsolidatedSediment()->IncrNotchFineLost(dLost);
274  }
275 
276  dAvailable = m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(nNotchLayer)->pGetConsolidatedSediment()->dGetSand() - m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(nNotchLayer)->pGetConsolidatedSediment()->dGetNotchSandLost();
277  if (dAvailable > 0)
278  {
279  dLost = dAvailable * dNotchLayerFracRemoved;
280  dSandCollapse += dLost;
281  m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(nNotchLayer)->pGetConsolidatedSediment()->IncrNotchSandLost(dLost);
282  }
283 
284  dAvailable = m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(nNotchLayer)->pGetConsolidatedSediment()->dGetCoarse() - m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(nNotchLayer)->pGetConsolidatedSediment()->dGetNotchCoarseLost();
285  if (dAvailable > 0)
286  {
287  dLost = dAvailable * dNotchLayerFracRemoved;
288  dCoarseCollapse += dLost;
289  m_pRasterGrid->m_Cell[nX][nY].pGetLayerAboveBasement(nNotchLayer)->pGetConsolidatedSediment()->IncrNotchCoarseLost(dLost);
290  }
291 
292  // Update the cell's layer elevations
293  m_pRasterGrid->m_Cell[nX][nY].CalcAllLayerElevsAndD50();
294 
295  // And update the cell's sea depth
296  m_pRasterGrid->m_Cell[nX][nY].SetSeaDepth();
297 
298  // The notch has gone
299  pCliff->SetNotchOverhang(0);
300 
301 // LogStream << m_ulIter << ": cell [" << nX << "][" << nY << "] after removing sediment, dGetVolEquivSedTopElev() = " << m_pRasterGrid->m_Cell[nX][nY].dGetVolEquivSedTopElev() << ", dGetSedimentTopElev() = " << m_pRasterGrid->m_Cell[nX][nY].dGetSedimentTopElev() << endl << endl;
302 
303  // And update the this-timestep totals and the grand totals for collapse
304  m_nNThisTimestepCliffCollapse++;
305  m_nNTotCliffCollapse++;
306 
307  m_dThisTimestepCliffCollapseErosionFine += dFineCollapse;
308  m_dThisTimestepCliffCollapseErosionSand += dSandCollapse;
309  m_dThisTimestepCliffCollapseErosionCoarse += dCoarseCollapse;
310 
311  return RTN_OK;
312 }

315 /*===============================================================================================================================
316 
317  Puts the fine sediment from a cliff collapse into suspension, and redistributes the sand-sized and coarse-sized sediment from a cliff collapse onto the foreshore, as unconsolidated talus
318 
319  The talus is added to the existing beach volume (i.e. to the unconsolidated sediment). The shoreline is iteratively advanced seaward until all this volume is accommodated under a Dean equilibrium profile. This equilibrium beach profile is h(y) = A * y^(2/3) where h(y) is the water depth at a distance y from the shoreline and A is a sediment-dependent scale parameter
320 
321 ===============================================================================================================================*/
322 int CSimulation::nDoCliffCollapseDeposition(CRWCliff* pCliff, double const dFineCollapse, double const dSandCollapse, double const dCoarseCollapse)
323 {
324  // Fine sediment goes into suspension
325  if (dFineCollapse > SEDIMENT_ELEV_TOLERANCE)
326  {
327  m_dThisTimestepFineSedimentToSuspension += dFineCollapse;
328  }
329 
330  // Do we have any sand- or coarse-sized sediment to deposit?
331  double dTotFromCollapse = dSandCollapse + dCoarseCollapse;
332  if (dTotFromCollapse < SEDIMENT_ELEV_TOLERANCE)
333  return RTN_OK;
334 
335  // OK, we have some sand- and/or coarse-sized sediment to deposit
336  int
337  nCoast = pCliff->nGetCoast(),
338  nStartPoint = pCliff->nGetPointOnCoast(),
339  nCoastSize = m_VCoast[nCoast].nGetCoastlineSize();
340 
341  double
342  dTotSandToDeposit = dSandCollapse,
343  dTotCoarseToDeposit = dCoarseCollapse,
344  dSandProp = dSandCollapse / dTotFromCollapse,
345  dCoarseProp = 1 - dSandProp;
346 
347 
348 // LogStream << "=====================================================================================================" << endl;
349 // LogStream << m_ulIter << ": coast = " << nCoast << ", point = " << nStartPoint << endl;
350 
351  // Calculate the proportion per planview collapse profile

357

360  for (int n = 0; n < m_nCliffCollapseTalusPlanviewWidth; n++)
361  {
362  nVWidthDistSigned[n] = nSigned++;
363  nVProfileLength[n] = nRound(m_dCliffDepositionPlanviewLength);
364  dVToDepositPerProfile[n] = (dTotSandToDeposit + dTotCoarseToDeposit) / m_nCliffCollapseTalusPlanviewWidth;
365  }

378 // for (int n = 0; n < m_nCliffCollapseTalusPlanviewWidth; n++)
379 // {
380 // LogStream << dVToDepositPerProfile[n] << " ";
381 // }
382 // LogStream << endl;
383 
384 // LogStream << "dSandCollapse = " << dSandCollapse << " dCoarseCollapse = " << dCoarseCollapse << " m_nCliffCollapseTalusPlanviewWidth = " << m_nCliffCollapseTalusPlanviewWidth << endl;
385 
386 // int nX = pCliff->pPtiGetCellMarkedAsLF()->nGetX();
387 // int nY = pCliff->pPtiGetCellMarkedAsLF()->nGetY();
388 // LogStream << "Cliff object is at cell[" << nX << "][" << nY << "] which is " << dGridCentroidXToExtCRSX(nX) << ", " << dGridCentroidYToExtCRSY(nY) << endl;
389 
390  for (int nAcross = 0; nAcross < m_nCliffCollapseTalusPlanviewWidth; nAcross++)
391  {
392  int nWidthDistSigned = nVWidthDistSigned[nAcross];
393 
394  // Get the start point of the deposition collapse profile
395  int nThisPoint = nStartPoint + nWidthDistSigned;
396 
397  // Is this start point valid?
398  if ((nThisPoint < 0) || (nThisPoint > (nCoastSize-1)))
399  {
400 // LogStream << endl << m_ulIter << ": ABANDONING PROFILE with nWidthDistSigned = " << nWidthDistSigned << endl;
401 // LogStream << "START point " << nThisPoint << " of profile would have been outside the grid, so " << dVToDepositPerProfile[nAcross] << " exported from grid" << endl;

405  m_dThisTimestepSandSedLostCliffCollapse += (dVToDepositPerProfile[nAcross] * dSandProp);
406  m_dThisTimestepCoarseSedLostCliffCollapse += (dVToDepositPerProfile[nAcross] * dCoarseProp);

813 int CSimulation::nRasterizeCliffCollapseProfile(vector<CGeom2DPoint> const* pVPointsIn, vector<CGeom2DIPoint>* pVIPointsOut) const
814 {
815  pVIPointsOut->clear();
816 
817  // The start point of the normal is the centroid of a coastline cell. Convert from the external CRS to grid CRS
818  double
819  dXStart = dExtCRSXToGridX(pVPointsIn->at(0).dGetX()),
820  dYStart = dExtCRSYToGridY(pVPointsIn->at(0).dGetY());
821 
822  // The end point of the normal, again convert from the external CRS to grid CRS. Note too that it could be off the grid
823  double
824  dXEnd = dExtCRSXToGridX(pVPointsIn->at(1).dGetX()),
825  dYEnd = dExtCRSYToGridY(pVPointsIn->at(1).dGetY());

828  double
829  dXInc = dXEnd - dXStart,
830  dYInc = dYEnd - dYStart,
831  dLength = tMax(tAbs(dXInc), tAbs(dYInc));
832 
833  dXInc /= dLength;
834  dYInc /= dLength;
835 
836  double
837  dX = dXStart,
838  dY = dYStart;
839 
840  // Process each interpolated point
841  int nLength = nRound(dLength);
842  for (int m = 0; m <= nLength; m++)
843  {
844  int
845  nX = static_cast<int>(dX),
846  nY = static_cast<int>(dY);
847 
848  // Make sure the interpolated point is within the raster grid (can get this kind of problem due to rounding)
849  if (! bIsWithinValidGrid(nX, nY))
850  KeepWithinValidGrid(nRound(dXStart), nRound(dYStart), nX, nY);
851 
852  // This point is fine, so append it to the output vector
853  pVIPointsOut->push_back(CGeom2DIPoint(nX, nY)); // In raster-grid co-ordinates
854 
855  // And increment for next time
856  dX += dXInc;
857  dY += dYInc;
