197 bool CGeomProfile::bInsertIntersection(double const dX, double const dY, int const nSeg)
198 {
199  // Safety check
200  if (nSeg >= nGetNumLineSegments())
201  return false;
202 
203  vector<CGeom2DPoint>::iterator it;
204  it = m_VPoints.begin();
205 
206  // Do the insertion
207  m_VPoints.insert(it+nSeg+1, CGeom2DPoint(dX, dY));
208 
209  // Now insert a line segment in the associated multi-line, this will inherit the profile/line seg details from the preceding line segment
210  CGeomMultiLine::InsertLineSegment(nSeg);
211 
212  return true;
213 }

424 int CGeomProfile::nGetCellGivenDepth(CGeomRasterGrid const* pGrid, double const dDepthIn)
425 {
426  int nIndex = INT_NODATA; // If not found, i.e. if every profile cell has sea depth less than dDepthIn
427 
428  for (unsigned int n = 0; n < m_VCellInProfile.size(); n++)
429  {
430  int
431  nX = m_VCellInProfile[n].nGetX(),
432  nY = m_VCellInProfile[n].nGetY();
433 
434  double dCellDepth = pGrid->m_Cell[nX][nY].dGetSeaDepth();
435  if (dCellDepth >= dDepthIn)
436  {
437  nIndex = n;
438 
439  if (n > 0)
440  nIndex = n-1; // Grid CRS units
441 
442  break;
443  }
444  }
445 
446  return nIndex;
447 }
