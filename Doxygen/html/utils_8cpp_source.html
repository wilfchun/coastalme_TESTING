162 void CSimulation::StartClock(void)
163 {
164  // First start the 'CPU time' clock ticking
165  if (static_cast<clock_t>(-1) == clock())
166  {
167  // There's a problem with the clock, but continue anyway
168  LogStream << INFO << "CPU time not available" << endl;
169  m_dCPUClock = -1;
170  }
171  else
172  {
173  // All OK, so get the time in m_dClkLast (this is needed to check for clock rollover on long runs)
174  m_dClkLast = static_cast<double>(clock());
175  m_dClkLast -= CLOCK_T_MIN; // necessary if clock_t is signed to make m_dClkLast unsigned
176  }
177 
178  // And now get the actual time we started
179  m_tSysStartTime = time(nullptr);
180 }

248 double CSimulation::dGetTimeMultiplier(string const* strIn)
249 {
250  // First decide what the time units are
251  int nTimeUnits = nDoTimeUnits(strIn);
252 
253  // Then return the correct multiplier, since m_dTimeStep is in hours
254  switch (nTimeUnits)
255  {
256  case TIME_UNKNOWN:
257  return TIME_UNKNOWN;
258  break;
259 
260  case TIME_HOURS:
261  return 1; // Multiplier for hours
262  break;
263 
264  case TIME_DAYS:
265  return 24; // Multiplier for days -> hours
266  break;
267 
268  case TIME_MONTHS:
269  return 24 * 30.416667; // Multiplier for months -> hours (assume 30 + 5/12 day months, no leap years)
270  break;
271 
272  case TIME_YEARS:
273  return 24 * 365.25; // Multiplier for years -> hours
274  break;
275  }
276 
277  return 0;
278 }

1255 /*==============================================================================================================================
1256 
1257  Checks to see if the simulation has gone on too long, amongst other things
1258 
1259 ==============================================================================================================================*/
1260 bool CSimulation::bTimeToQuit(void)
1261 {
1262  // Add timestep to the total time simulated so far
1263  m_dSimElapsed += m_dTimeStep;
1264 
1265  if (m_dSimElapsed >= m_dSimDuration)
1266  {
1267  // It is time to quit
1268  m_dSimElapsed = m_dSimDuration;
1269  AnnounceProgress();
1270  return true;
1271  }
1272 
1273  // Not quitting, so increment the timestep count, and recalc total timesteps
1274  m_ulIteration++;
1275  m_ulTotTimestep = static_cast<unsigned long>(dRound(m_dSimDuration / m_dTimeStep));
1276 
1277  // Check to see if we have done CLOCK_CHECK_ITERATION timesteps: if so, it is time to reset the CPU time running total in case the clock()
1278  // function later rolls over
1279  if (0 == m_ulIteration % CLOCK_CHECK_ITERATION)
1280  DoCPUClockReset();
1281 
1282  // Not yet time to quit
1283  return false;

1448 /*==============================================================================================================================

1450  Resets the CPU clock timer to prevent it 'rolling over', as can happen during long runs. This is a particularly problem under Unix systems where the value returned by clock() is defined in microseconds (for compatibility with systems that have CPU clocks with much higher resolution) i.e. CLOCKS_PER_SEC is 1000000 rather than the more usual 1000. In this case, the value returned from clock() will wrap around after accumulating only 2147 seconds of CPU time (about 36 minutes).
1451 
1452 ==============================================================================================================================*/
1453 void CSimulation::DoCPUClockReset(void)
1454 {
1455  if (static_cast<clock_t>(-1) == clock())
1456  {
1457  // Error
1458  LogStream << "CPU time not available" << endl;
1459  m_dCPUClock = -1;
1460  return;
1461  }
1462 
1463  // OK, so carry on
1464  double dClkThis = static_cast<double>(clock());
1465  dClkThis -= CLOCK_T_MIN; // necessary when clock_t is signed, to make dClkThis unsigned
1466 
1467  if (dClkThis < m_dClkLast)
1468  {
1469  // Clock has 'rolled over'
1470  m_dCPUClock += (CLOCK_T_RANGE + 1 - m_dClkLast); // this elapsed before rollover
1471  m_dCPUClock += dClkThis; // this elapsed after rollover
1472 
1473 #ifdef CLOCKCHECK
1474  // For debug purposes
1475  LogStream << "Rolled over: dClkThis=" << dClkThis << " m_dClkLast=" << m_dClkLast << endl << "\t" << " before rollover=" << (CLOCK_T_RANGE + 1 - m_dClkLast) << endl << "\t" << " after rollover=" << dClkThis << endl << "\t" << " ADDED=" << (CLOCK_T_RANGE + 1 - m_dClkLast + dClkThis) << endl;
1476 #endif
1477  }
1478  else
1479  {
1480  // No rollover
1481  m_dCPUClock += (dClkThis - m_dClkLast);
1482 
1483 #ifdef CLOCKCHECK
1484  // For debug purposes
1485  LogStream << "No rollover: dClkThis=" << dClkThis << " m_dClkLast=" << m_dClkLast << " ADDED=" << dClkThis - m_dClkLast << endl;
1486 #endif
1487  }
1488 
1489  // Reset for next time
1490  m_dClkLast = dClkThis;
1491 }
